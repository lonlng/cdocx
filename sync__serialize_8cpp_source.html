29void Document::merge_sections_from_physical() {
30 auto *doc_xml = get_document_xml();
31 if (!doc_xml) {
32 return;
33 }
34
35 auto body = doc_xml->child("w:document").child("w:body");
36 if (!body) {
37 return;
38 }
39
40 auto ranges = collect_section_ranges(body);
41
42 auto dom_sections = get_sections();
43 std::vector<std::shared_ptr<Section>> dom_sections_vec;
44 for (auto& s : dom_sections) {
45 dom_sections_vec.push_back(s);
46 }
47
48 // Ensure we have enough DOM sections
49 while (dom_sections_vec.size() < ranges.size()) {
50 auto section = std::make_shared<Section>(this);
51 section->set_first_section(dom_sections_vec.empty());
52 auto sect_body = std::make_shared<Body>(this);
53 section->set_body(sect_body);
54 append_child(section);
55 sections_cache_.push_back(section);
56 dom_sections_vec.push_back(section);
57 }
58
59 // Merge each XML section into the corresponding DOM section
60 for (size_t i = 0; i < ranges.size(); ++i) {
61 const auto& section = dom_sections_vec[i];
62 auto sect_body = section->get_body();
63 if (!sect_body) {
64 sect_body = std::make_shared<Body>(this);
65 section->set_body(sect_body);
66 }
67
68 // Collect existing DOM children
69 std::vector<std::shared_ptr<Node>> dom_children;
70 for (const auto& child : sect_body->get_children()) {
71 dom_children.push_back(child);
72 }
73
74 // Collect XML children
75 std::vector<std::shared_ptr<Node>> xml_children;
76 for (auto node = ranges[i].begin; node && node != ranges[i].end;
77 node = node.next_sibling()) {
78 const char* name = node.name();
79 if (is_para_node(name)) {
80 if (auto para = parse_paragraph_from_xml(node)) {
81 xml_children.push_back(para);
82 }
83 } else if (is_table_node(name)) {
84 if (auto table = parse_table_from_xml(node)) {
85 xml_children.push_back(table);
86 }
87 }
88 }
89
90 // Merge by position: replace dirty-linked paragraphs, keep others,
91 // append extra XML children, preserve extra DOM children
92 const size_t merge_count = std::min(dom_children.size(), xml_children.size());
93 for (size_t j = 0; j < merge_count; ++j) {
94 bool should_replace = false;
95 if (dom_children[j]->node_type() == NodeType::Paragraph) {
96 auto* para = dynamic_cast<Paragraph*>(dom_children[j].get());
97 if (para && para->get_current() &&
98 dirty_xml_paragraphs_.count(para->get_current()) > 0) {
99 should_replace = true;
100 }
101 }
102
103 if (should_replace) {
104 sect_body->insert_child(static_cast<int>(j), xml_children[j]);
105 sect_body->remove_child(dom_children[j]);
106 }
107 }
108
109 // Append extra XML children
110 for (size_t j = merge_count; j < xml_children.size(); ++j) {
111 sect_body->append_child(xml_children[j]);
112 }
113
114 // Update section properties
115 if (ranges[i].end) {
116 section->set_sect_pr_node(ranges[i].end);
117 section->load_properties();
118 }
119 }
120
121 sections_dirty_ = false;
122}

124void Document::sync_sections_to_physical() {
125 auto *doc_xml = get_document_xml();
126 if (!doc_xml) {
127 return;
128 }
129
130 auto body = doc_xml->child("w:document").child("w:body");
131 if (!body) {
132 return;
133 }
134
135 // If DOM has no sections, preserve existing XML
136 auto sections = get_sections();
137 if (sections.is_empty()) {
138 return;
139 }
140
141 // Count XML paragraph/table children
142 int xml_child_count = 0;
143 for (auto child = body.first_child(); child; child = child.next_sibling()) {
144 const char* name = child.name();
145 if (is_content_node(name)) {
146 ++xml_child_count;
147 }
148 }
149
150 // Count DOM paragraph/table children across all sections
151 int dom_child_count = 0;
152 for (auto& section : sections) {
153 if (auto sect_body = section->get_body()) {
154 for (const auto& child : sect_body->get_children()) {
155 if (child->node_type() == NodeType::Paragraph ||
156 child->node_type() == NodeType::Table) {
157 ++dom_child_count;
158 }
159 }
160 }
161 }
162
163 // Update DOM paragraphs whose XML was explicitly modified by DocumentBuilder
164 // or legacy API, so their content is not lost during DOM serialization.
165 for (auto& section : sections) {
166 if (auto sect_body = section->get_body()) {
167 for (const auto& child : sect_body->get_children()) {
168 if (child->node_type() == NodeType::Paragraph) {
169 auto* para = dynamic_cast<Paragraph*>(child.get());
170 if (para && para->get_current() &&
171 dirty_xml_paragraphs_.count(para->get_current()) > 0) {
172 for (auto xml_para = body.child("w:p"); xml_para;
173 xml_para = xml_para.next_sibling("w:p")) {
174 if (xml_para == para->get_current()) {
175 if (auto updated = parse_paragraph_from_xml(xml_para)) {
176 para->remove_all_children();
177 for (const auto& new_child : updated->get_children()) {
178 para->append_child(new_child);
179 }
180 }
181 para->set_current(pugi::xml_node());
182 break;
183 }
184 }
185 }
186 }
187 }
188 }
189 }
190 // If XML has more content than DOM, physical XML may have been modified
191 // directly (legacy API, DocumentBuilder). We only sync from physical when
192 // the XML child count has INCREASED since the last sync — that indicates
193 // new content was added to physical XML outside the DOM. If the count is
194 // unchanged, the DOM intentionally deleted content and we must trust it.
195 if (xml_child_count > dom_child_count &&
196 xml_child_count > last_synced_xml_child_count_) {
197 if (!dirty_xml_paragraphs_.empty()) {
198 merge_sections_from_physical();
199 } else {
200 sync_from_physical_tree();
201 }
202 sections = get_sections();
203 }
204 dirty_xml_paragraphs_.clear();
205
206 // Preserve unknown nodes (not paragraphs, tables, or section properties)
207 pugi::xml_document preserved_doc;
208 for (auto child = body.first_child(); child;) {
209 auto next = child.next_sibling();
210 const char* name = child.name();
211 if (!is_content_node(name) && !is_sectpr_node(name)) {
212 preserved_doc.append_copy(child);
213 }
214 child = next;
215 }
216
217 // Remove all existing children from body
218 while (body.first_child()) {
219 body.remove_child(body.first_child());
220 }
221
222 // Serialize each section
223 for (auto& section : sections) {
224 serialize_section_to_xml(body, section.get());
225 }
226
227 // Re-append preserved unknown nodes at the end
228 for (auto child = preserved_doc.first_child(); child; child = child.next_sibling()) {
229 body.append_copy(child);
230 }
231
232 // Record the synced child count so future calls can detect physical-only
233 // additions (e.g. legacy API direct XML manipulation).
234 int synced_count = 0;
235 for (auto child = body.first_child(); child; child = child.next_sibling()) {
236 const char* name = child.name();
237 if (is_content_node(name)) {
238 ++synced_count;
239 }
240 }
241 last_synced_xml_child_count_ = synced_count;
242
243 mark_modified("word/document.xml");
244}

246void serialize_font_to_r_pr(pugi::xml_node r_pr, const Font& font, bool add_sz_cs) {
247 serialize_shading_to_xml(r_pr, font.shading);
248 if (font.bold) {
249 r_pr.append_child("w:b");
250 }
251 if (font.italic) {
252 r_pr.append_child("w:i");
253 }
254 if (font.strikethrough) {
255 auto strike = r_pr.append_child("w:strike");
256 strike.append_attribute("w:val").set_value("true");
257 }
258 if (font.underline != UnderlineType::None) {
259 auto u = r_pr.append_child("w:u");
260 const char* val = underline_type_to_string(font.underline);
261 if (!val) {
262 val = "single";
263 }
264 u.append_attribute("w:val").set_value(val);
265 }
266 if (font.size > 0) {
267 auto sz = r_pr.append_child("w:sz");
268 sz.append_attribute("w:val").set_value(static_cast<int>(font.size * 2));
269 if (add_sz_cs) {
270 auto sz_cs = r_pr.append_child("w:szCs");
271 sz_cs.append_attribute("w:val").set_value(static_cast<int>(font.size * 2));
272 }
273 }
274 if (!font.name.empty()) {
275 auto r_fonts = r_pr.append_child("w:rFonts");
276 r_fonts.append_attribute("w:ascii").set_value(font.name.c_str());
277 r_fonts.append_attribute("w:hAnsi").set_value(font.name.c_str());
278 if (!font.name_far_east.empty()) {
279 r_fonts.append_attribute("w:eastAsia").set_value(font.name_far_east.c_str());
280 }
281 }
282 if (font.color != Color::auto_color()) {
283 auto color = r_pr.append_child("w:color");
284 color.append_attribute("w:val").set_value(font.color.to_hex_rgb().c_str());
285 }
286 if (font.script_type != ScriptType::Normal) {
287 auto v_align = r_pr.append_child("w:vertAlign");
288 v_align.append_attribute("w:val").set_value(script_type_to_string(font.script_type));
289 }
290 if (font.spacing != 0) {
291 auto sp = r_pr.append_child("w:spacing");
292 sp.append_attribute("w:val").set_value(ConvertUtil::point_to_twips(font.spacing));
293 }
294 if (font.scale != 100) {
295 auto w = r_pr.append_child("w:w");
296 w.append_attribute("w:val").set_value(font.scale);
297 }

300static void serialize_run_formatting_to_xml(pugi::xml_node run_xml,
301 const Font& font,
302 pugi::xml_node preserved_r_pr) {
303 const bool has_formatting = font.bold || font.italic || font.strikethrough ||
304 font.underline != UnderlineType::None || font.size != 12.0 ||
305 font.name != "Times New Roman" || font.color != Color::black() ||
306 font.script_type != ScriptType::Normal || font.shading.has_fill() ||
307 font.spacing != 0 || font.scale != 100;
308
309 if (!has_formatting && !preserved_r_pr) {
310 return;
311 }
312
313 auto r_pr = run_xml.child("w:rPr");
314 if (!r_pr) {
315 if (preserved_r_pr) {
316 r_pr = run_xml.prepend_copy(preserved_r_pr);
317 strip_whitespace_text_nodes(r_pr);
318 } else {
319 r_pr = run_xml.prepend_child("w:rPr");
320 }
321 }
322
323 // Remember whether the original had w:sz_cs before we remove managed children
324 const bool original_had_sz_cs = preserved_r_pr && preserved_r_pr.child("w:szCs");
325
326 // Remove managed children that we will re-serialize from DOM font state.
327 // w:r_fonts is handled separately: if font.name is empty we leave the
328 // original r_fonts untouched so that attributes like w:hint survive.
329 remove_managed_children(r_pr,
330 {"w:b",
331 "w:i",
332 "w:strike",
333 "w:u",
334 "w:sz",
335 "w:szCs",
336 "w:color",
337 "w:vertAlign",
338 "w:spacing",
339 "w:w",
340 "w:shd"});
341 if (!font.name.empty()) {
342 remove_managed_children(r_pr, {"w:rFonts"});
343 }
344
345 // Re-add managed children from current font state
346 serialize_font_to_r_pr(r_pr, font, original_had_sz_cs);
347
348 // If r_pr ended up empty, remove it entirely to avoid <w:r_pr/> bloat.
349 if (!r_pr.first_child()) {
350 run_xml.remove_child(r_pr);
351 }
352}

354static void serialize_run_to_xml(pugi::xml_node parent, const Run* run) {
355 if (!run) {
356 return;
357 }
358
359 auto run_xml = parent.append_child("w:r");
360 serialize_run_formatting_to_xml(
361 run_xml,
362 run->get_font(),
363 run->has_preserved_r_pr() ? run->get_preserved_r_pr() : pugi::xml_node());
364
365 const std::string& text = run->get_text();
366 if (!text.empty()) {
367 auto text_node = run_xml.append_child("w:t");
368 if (std::isspace(static_cast<unsigned char>(text.front())) ||
369 std::isspace(static_cast<unsigned char>(text.back()))) {
370 text_node.append_attribute("xml:space").set_value("preserve");
371 }
372 text_node.text().set(text.c_str());
373 }
374
375 // Serialize preserved children (e.g., w:drawing) for round-trip fidelity
376 if (run->has_preserved_children()) {
377 run->serialize_preserved_children(run_xml);
378 }

415static void serialize_hyperlink_to_xml(pugi::xml_node parent, Hyperlink* link) {
416 if (!link) {
417 return;
418 }
419
420 auto hyperlink_xml = parent.append_child("w:hyperlink");
421
422 const std::string address = link->get_address();
423 const std::string bookmark = link->get_bookmark_name();
424
425 Document* doc = link->get_document();
426 if (!address.empty() && doc) {
427 std::string rel_id = doc->find_relationship_id("word/_rels/document.xml.rels", address);
428 if (rel_id.empty()) {
429 rel_id = doc->add_relationship(
430 "word/_rels/document.xml.rels",
431 "http://schemas.openxmlformats.org/officeDocument/2006/relationships/hyperlink",
432 address,
433 "External");
434 }
435 hyperlink_xml.append_attribute("r:id").set_value(rel_id.c_str());
436 } else if (!bookmark.empty()) {
437 hyperlink_xml.append_attribute("w:anchor").set_value(bookmark.c_str());
438 }
439
440 if (!link->get_tooltip().empty()) {
441 hyperlink_xml.append_attribute("w:tooltip").set_value(link->get_tooltip().c_str());
442 }
443
444 auto run_xml = hyperlink_xml.append_child("w:r");
445 serialize_run_formatting_to_xml(
446 run_xml,
447 link->get_font(),
448 link->has_preserved_r_pr() ? link->get_preserved_r_pr() : pugi::xml_node());
449
450 const std::string result = link->get_result();
451 if (!result.empty()) {
452 auto text_node = run_xml.append_child("w:t");
453 text_node.text().set(result.c_str());
454 }
455}

470static void serialize_special_char_to_xml(pugi::xml_node parent, SpecialChar* sc) {
471 if (!sc) {
472 return;
473 }
474 const char16_t code = sc->get_char();
475 for (const auto& mapping : kSpecialCharMappings) {
476 if (mapping.code == code) {
477 auto node = parent.append_child(mapping.element_name);
478 if (mapping.break_type) {
479 node.append_attribute("w:type").set_value(mapping.break_type);
480 }
481 return;
482 }
483 }
484 // For other special chars, append as text in a run
485 auto run = parent.append_child("w:r");
486 auto t = run.append_child("w:t");
487 t.text().set(sc->get_text().c_str());
488}

644void serialize_paragraph_to_xml(pugi::xml_node parent, const Paragraph* para) {
645 if (!para) {
646 return;
647 }
648 auto para_xml = parent.append_child("w:p");
649
650 serialize_paragraph_format_to_xml(para_xml, para->get_paragraph_format());
651 serialize_list_format_to_xml(para_xml, para->get_list_format());
652
653 // Preserve unmanaged p_pr children and attributes that ParagraphFormat doesn't track.
654 if (para->has_preserved_p_pr()) {
655 const pugi::xml_node preserved_p_pr = para->get_preserved_p_pr();
656 pugi::xml_node p_pr = para_xml.child("w:pPr");
657 if (!p_pr) {
658 para_xml.prepend_copy(preserved_p_pr);
659 } else {
660 for (pugi::xml_node child = preserved_p_pr.first_child(); child;
661 child = child.next_sibling()) {
662 // Skip whitespace text nodes from pretty-printed XML
663 if (child.type() == pugi::node_pcdata || child.type() == pugi::node_cdata) {
664 continue;
665 }
666 const char* name = child.name();
667 pugi::xml_node existing = p_pr.child(name);
668 if (!existing) {
669 p_pr.append_copy(child);
670 } else {
671 // Merge any attributes that the DOM serialization didn't create
672 for (pugi::xml_attribute attr = child.first_attribute(); attr;
673 attr = attr.next_attribute()) {
674 if (!existing.attribute(attr.name())) {
675 existing.append_copy(attr);
676 }
677 }
678 }
679 }
680 }
681 }
682
683 std::vector<int> comment_ids_for_reference;
684 for (const auto& child : para->get_children()) {
685 switch (child->node_type()) {
686 case NodeType::Run:
687 serialize_run_to_xml(para_xml, dynamic_cast<Run*>(child.get()));
688 break;
689 case NodeType::SpecialChar:
690 serialize_special_char_to_xml(para_xml, dynamic_cast<SpecialChar*>(child.get()));
691 break;
692 case NodeType::BookmarkStart:
693 serialize_bookmark_start_to_xml(para_xml,
694 dynamic_cast<BookmarkStart*>(child.get()));
695 break;
696 case NodeType::BookmarkEnd:
697 serialize_bookmark_end_to_xml(para_xml, dynamic_cast<BookmarkEnd*>(child.get()));
698 break;
699 case NodeType::CommentRangeStart:
700 serialize_comment_range_start_to_xml(para_xml,
701 dynamic_cast<CommentRangeStart*>(child.get()));
702 break;
703 case NodeType::CommentRangeEnd:
704 serialize_comment_range_end_to_xml(para_xml,
705 dynamic_cast<CommentRangeEnd*>(child.get()));
706 if (auto* cre = dynamic_cast<CommentRangeEnd*>(child.get())) {
707 comment_ids_for_reference.push_back(cre->get_id());
708 }
709 break;
710 case NodeType::FieldStart:
711 serialize_field_to_xml(para_xml, dynamic_cast<Field*>(child.get()));
712 break;
713 case NodeType::FormField:
714 serialize_form_field_to_xml(para_xml, dynamic_cast<FormField*>(child.get()));
715 break;
716 case NodeType::Hyperlink:
717 serialize_hyperlink_to_xml(para_xml, dynamic_cast<Hyperlink*>(child.get()));
718 break;
719 case NodeType::FootnoteReference:
720 serialize_footnote_reference_to_xml(para_xml,
721 dynamic_cast<FootnoteReference*>(child.get()));
722 break;
723 case NodeType::EndnoteReference:
724 serialize_endnote_reference_to_xml(para_xml,
725 dynamic_cast<EndnoteReference*>(child.get()));
726 break;
727 case NodeType::FieldSeparator: // NOLINT(bugprone-branch-clone)
728 case NodeType::FieldEnd:
729 // These node types are not used as standalone DOM nodes
730 break;
731 default:
732 break;
733 }
734 }
735 // Add comment references after the paragraph content
736 for (const int comment_id : comment_ids_for_reference) {
737 serialize_comment_reference_to_xml(para_xml, comment_id);
738 }
739}

743static void serialize_cell_to_xml(pugi::xml_node parent, Cell* cell) {
744 if (!cell) {
745 return;
746 }
747 auto tc = parent.append_child("w:tc");
748
749 // Cell properties — always write tc_pr with at least a default tc_w so that
750 // Word has structural information for layout. Omitting tc_pr entirely can
751 // cause unexpected default behaviour in some consumers.
752 const CellFormat& fmt = cell->get_cell_format();
753 auto tc_pr = tc.append_child("w:tcPr");
754 {
755 auto tc_w = tc_pr.append_child("w:tcW");
756 if (fmt.width != 0) {
757 tc_w.append_attribute("w:w").set_value(ConvertUtil::point_to_twips(fmt.width));
758 tc_w.append_attribute("w:type").set_value(fmt.preferred_width ? "pct" : "dxa");
759 } else {
760 tc_w.append_attribute("w:w").set_value("0");
761 tc_w.append_attribute("w:type").set_value("auto");
762 }
763 }
764 if (fmt.vertical_alignment != CellVerticalAlignment::Top) {
765 auto v_align = tc_pr.append_child("w:vAlign");
766 v_align.append_attribute("w:val").set_value(
767 cell_vertical_alignment_to_string(fmt.vertical_alignment));
768 }
769 if (fmt.horizontal_merge > 1) {
770 auto grid_span = tc_pr.append_child("w:gridSpan");
771 grid_span.append_attribute("w:val").set_value(fmt.horizontal_merge);
772 }
773 if (fmt.vertical_merge) {
774 auto v_merge = tc_pr.append_child("w:vMerge");
775 if (fmt.vertical_merge_start) {
776 v_merge.append_attribute("w:val").set_value("restart");
777 }
778 }
779 serialize_borders_to_xml(tc_pr, "w:tcBorders", fmt.borders);
780 serialize_shading_to_xml(tc_pr, fmt.shading);
781
782 for (const auto& child : cell->get_children()) {
783 serialize_node_child_to_xml(tc, child.get());
784 }
785
786 // Ensure at least one paragraph
787 if (tc.child("w:p") == nullptr) {
788 tc.append_child("w:p").append_child("w:r").append_child("w:t");
789 }

823static void serialize_table_to_xml(pugi::xml_node parent, const Table* table) {
824 if (!table) {
825 return;
826 }
827 auto tbl = parent.append_child("w:tbl");
828
829 // Table properties
830 const TableFormat& fmt = table->get_table_format();
831 pugi::xml_node tbl_pr;
832 if (table->has_preserved_tbl_pr()) {
833 tbl_pr = tbl.append_copy(table->get_preserved_tbl_pr());
834 strip_whitespace_text_nodes(tbl_pr);
835 // Remove managed children so we can re-serialize current DOM state
836 remove_managed_children(
837 tbl_pr,
838 {"w:tblW", "w:tblLayout", "w:jc", "w:tblInd", "w:tblStyle", "w:shd", "w:tblBorders"});
839 } else {
840 tbl_pr = tbl.append_child("w:tblPr");
841 }
842
843 auto tbl_w = tbl_pr.append_child("w:tblW");
844 if (fmt.auto_fit_behavior == AutoFitBehavior::AutoFitToWindow) {
845 tbl_w.append_attribute("w:w").set_value("5000");
846 tbl_w.append_attribute("w:type").set_value("pct");
847 } else {
848 tbl_w.append_attribute("w:w").set_value("0");
849 tbl_w.append_attribute("w:type").set_value("auto");
850 }
851

853 if (fmt.auto_fit_behavior == AutoFitBehavior::FixedColumnWidth) {
854 tbl_layout.append_attribute("w:type").set_value("fixed");
855 } else {
856 tbl_layout.append_attribute("w:type").set_value("autofit");
857 }
858
859 if (fmt.alignment != TableAlignment::Left) {
860 auto jc = tbl_pr.append_child("w:jc");
861 jc.append_attribute("w:val").set_value(table_alignment_to_string(fmt.alignment));
862 }

865 auto tbl_ind = tbl_pr.append_child("w:tblInd");
866 tbl_ind.append_attribute("w:w").set_value(ConvertUtil::point_to_twips(fmt.left_indent));
867 tbl_ind.append_attribute("w:type").set_value("dxa");
868 }
869

871 if (!style_name.empty()) {
872 auto tbl_style = tbl_pr.append_child("w:tblStyle");
873 tbl_style.append_attribute("w:val").set_value(style_name.c_str());
874 }
875
876 serialize_shading_to_xml(tbl_pr, fmt.shading);
877 serialize_borders_to_xml(tbl_pr, "w:tblBorders", fmt.borders);

879 // Table grid
880 if (table->has_preserved_tbl_grid()) {
881 tbl.append_copy(table->get_preserved_tbl_grid());

890 tbl_grid.append_child("w:gridCol");
891 }
892 }
893 }
894 }
895
896 for (const auto& child : table->get_children()) {
897 if (child->node_type() == NodeType::Row) {
898 serialize_row_to_xml(tbl, dynamic_cast<Row*>(child.get()));
899 }
900 }
