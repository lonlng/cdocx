40void Footnote::set_text(const std::string& text) {
41 remove_all_children();
42
43 size_t start = 0;
44 while (start <= text.size()) {
45 size_t end = text.find('\n', start);
46 if (end == std::string::npos) {
47 end = text.size();
48 }
49 auto para = std::make_shared<Paragraph>();
50 if (start < end) {
51 para->append_run(text.substr(start, end - start));
52 }
53 append_child(para);
54 start = end + 1;
55 }
56}
