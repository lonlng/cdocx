18bool Document::load_content_types() {
19 auto* content_types = get_content_types();
20 if (!content_types) {
21 return false;
22 }
23
24 content_types_.clear();
25
26 auto root = content_types->child("Types");
27 if (!root) {
28 return false;
29 }
30
31 for (auto def = root.child("Default"); def; def = def.next_sibling("Default")) {
32 ContentType ct;
33 ct.extension = def.attribute("Extension").value();
34 ct.content_type = def.attribute("ContentType").value();
35 ct.is_default = true;
36 content_types_.push_back(ct);
37 }
38
39 for (auto override = root.child("Override"); override;
40 override = override.next_sibling("Override")) {
41 ContentType ct;
42 ct.part_name = override.attribute("PartName").value();
43 ct.content_type = override.attribute("ContentType").value();
44 ct.is_default = false;
45 content_types_.push_back(ct);
46 }
47
48 return true;
49}

51void Document::parse_relationships(const std::string& rels_path) {
52 auto* rels_doc = get_xml_part(rels_path);
53 if (!rels_doc) {
54 return;
55 }
56
57 auto root = rels_doc->child("Relationships");
58 if (!root) {
59 return;
60 }
61
62 std::vector<Relationship> rels;
63
64 for (auto rel = root.child("Relationship"); rel; rel = rel.next_sibling("Relationship")) {
65 Relationship r;
66 r.id = rel.attribute("Id").value();
67 r.type = rel.attribute("Type").value();
68 r.target = rel.attribute("Target").value();
69 r.target_mode = rel.attribute("TargetMode").value();
70 rels.push_back(r);
71 }
72
73 relationships_[rels_path] = rels;
74}

98std::string Document::add_relationship(const std::string& rels_path,
99 const std::string& type,
100 const std::string& target,
101 const std::string& target_mode) {
102 // Find next available rId
103 int max_id = 0;
104 for (const auto& rel : relationships_[rels_path]) {
105 if (rel.id.substr(0, 3) == "rId") {
106 const int id = std::stoi(rel.id.substr(3));
107 max_id = std::max(max_id, id);
108 }
109 }
110
111 std::string new_id = "rId" + std::to_string(max_id + 1);
112
113 Relationship r;
114 r.id = new_id;
115 r.type = type;
116 r.target = target;
117 r.target_mode = target_mode;
118 relationships_[rels_path].push_back(r);
119
120 modified_parts_.insert(rels_path);
121
122 return new_id;
123}

164void Document::update_relationships_xml(const std::string& rels_path) {
165 auto& rels = relationships_[rels_path];
166 if (rels.empty()) {
167 return;
168 }
169
170 // Create or update the relationships XML
171 pugi::xml_document* doc = nullptr;
172
173 if (has_xml_part(rels_path)) {
174 doc = get_xml_part(rels_path);
175 doc->reset();
176 } else {
177 doc = &create_xml_part(rels_path);
178 }
179
180 auto root = doc->append_child("Relationships");
181 root.append_attribute("xmlns").set_value(
182 "http://schemas.openxmlformats.org/package/2006/relationships");
183
184 for (const auto& rel : rels) {
185 auto rel_node = root.append_child("Relationship");
186 rel_node.append_attribute("Id").set_value(rel.id.c_str());
187 rel_node.append_attribute("Type").set_value(rel.type.c_str());
188 rel_node.append_attribute("Target").set_value(rel.target.c_str());
189 if (!rel.target_mode.empty()) {
190 rel_node.append_attribute("TargetMode").set_value(rel.target_mode.c_str());
191 }
192 }
193}

233void Document::update_content_types_xml() {
234 if (content_types_.empty()) {
235 return;
236 }
237
238 pugi::xml_document* doc = nullptr;
239
240 if (has_xml_part("[Content_Types].xml")) {
241 doc = get_content_types();
242 doc->reset();
243 } else {
244 doc = &create_xml_part("[Content_Types].xml");
245 }
246
247 auto root = doc->append_child("Types");
248 root.append_attribute("xmlns").set_value(
249 "http://schemas.openxmlformats.org/package/2006/content-types");
250
251 // Add defaults first
252 for (const auto& ct : content_types_) {
253 if (ct.is_default) {
254 auto def = root.append_child("Default");
255 def.append_attribute("Extension").set_value(ct.extension.c_str());
256 def.append_attribute("ContentType").set_value(ct.content_type.c_str());
257 }
258 }
259
260 // Then overrides
261 for (const auto& ct : content_types_) {
262 if (!ct.is_default) {
263 auto override = root.append_child("Override");
264 override.append_attribute("PartName").set_value(ct.part_name.c_str());
265 override.append_attribute("ContentType").set_value(ct.content_type.c_str());
266 }
267 }
