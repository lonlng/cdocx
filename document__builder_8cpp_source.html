258void DocumentBuilder::apply_formatting(pugi::xml_node run) const {
259 if (!run) {
260 return;
261 }
262
263 pugi::xml_node r_pr = run.child("w:rPr");
264 if (!r_pr) {
265 r_pr = run.prepend_child("w:rPr");
266 }
267
268 // Apply formatting flags
269 if (format_.bold) {
270 ensure_child(r_pr, "w:b");
271 }
272 if (format_.italic) {
273 ensure_child(r_pr, "w:i");
274 }
275 if (format_.underline) {
276 pugi::xml_node u = ensure_child(r_pr, "w:u");
277 u.append_attribute("w:val").set_value("single");
278 }
279 if (format_.strikethrough) {
280 ensure_child(r_pr, "w:strike");
281 }
282 if (!format_.font_name.empty()) {
283 pugi::xml_node r_fonts = r_pr.child("w:rFonts");
284 if (!r_fonts) {
285 r_fonts = r_pr.append_child("w:rFonts");
286 }
287 r_fonts.append_attribute("w:ascii").set_value(format_.font_name.c_str());
288 r_fonts.append_attribute("w:hAnsi").set_value(format_.font_name.c_str());
289 }
290 if (format_.font_size > 0) {
291 pugi::xml_node sz = r_pr.child("w:sz");
292 if (!sz) {
293 sz = r_pr.append_child("w:sz");
294 }
295 sz.append_attribute("w:val").set_value(format_.font_size * 2);
296 }
297 if (!format_.color.empty()) {
298 pugi::xml_node color = r_pr.child("w:color");
299 if (!color) {

331DocumentBuilder& DocumentBuilder::move_to_paragraph(size_t paragraph_index,
332 size_t character_index) {
333 const pugi::xml_node body = get_body();
334 size_t count = 0;
335
336 for (pugi::xml_node para = body.child("w:p"); para; para = para.next_sibling("w:p")) {
337 if (count == paragraph_index) {
338 current_paragraph_ = para;
339 // Find run containing the character index
340 size_t char_count = 0;
341 for (pugi::xml_node run = para.child("w:r"); run; run = run.next_sibling("w:r")) {
342 const pugi::xml_node t = run.child("w:t");
343 if (t) {
344 const size_t text_len = std::strlen(t.text().get());
345 if (char_count + text_len >= character_index) {
346 current_node_ = run;
347 return *this;
348 }
349 char_count += text_len;
350 }
351 }
352 current_node_ = para;
353 return *this;
354 }
355 count++;
356 }
357 return *this;

360DocumentBuilder& DocumentBuilder::move_to_section(size_t index) {
361 auto sections = doc_->get_sections();
362 if (index >= sections.get_count()) {
363 return *this;
364 }
365
366 auto body = sections[static_cast<int>(index)]->get_body();
367 if (!body) {
368 return *this;
369 }
370
371 pugi::xml_document* doc_xml = doc_->get_document_xml();
372 if (!doc_xml) {
373 return *this;
374 }
375
376 const pugi::xml_node xml_body = doc_xml->child("w:document").child("w:body");
377 size_t sect_count = 0;
378 for (pugi::xml_node para = xml_body.child("w:p"); para; para = para.next_sibling("w:p")) {
379 if (sect_count == index) {
380 current_paragraph_ = para;
381 current_node_ = para;
382 target_xml_doc_ = doc_xml;
383 return *this;
384 }
385 // Check if this paragraph ends a section
386 if (para.child("w:sectPr")) {
387 sect_count++;
388 if (sect_count == index) {
389 const pugi::xml_node next = para.next_sibling();
390 if (next && is_para_node(next.name())) {
391 current_paragraph_ = next;
392 current_node_ = next;
393 } else {
394 current_paragraph_ = para;
395 current_node_ = para;
396 }
397 target_xml_doc_ = doc_xml;
398 return *this;
399 }
400 }
401 }
402 // If no section breaks found, just move to first paragraph
403 if (index == 0) {
404 current_paragraph_ = xml_body.child("w:p");
405 current_node_ = current_paragraph_;
406 target_xml_doc_ = doc_xml;
407 }
408 return *this;
409}

448 if (!doc_xml) {

453

455 para = para.next_sibling("w:p")) {
456 for (pugi::xml_node run = para.child("w:r"); run; run = run.next_sibling("w:r")) {
457 const pugi::xml_node fld_char = run.child("w:fldChar");

460 std::string field_code;
461 if (walk_field_sequence(run, &field_code, nullptr)) {
462 const std::string trimmed = trim_whitespace(field_code);
463 if (trimmed.find(target_code) == 0) {
464 target_xml_doc_ = doc_xml;
465 current_paragraph_ = para;
466 current_node_ = run;
467 return true;
468 }
469 }
470 }
471 }
472 }
473 return false;

607DocumentBuilder& DocumentBuilder::insert_break(BreakType break_type) {
608 switch (break_type) {
609 case BreakType::ParagraphBreak: {
610 pugi::xml_node body = get_body();
611 const pugi::xml_node sect_pr = body.child("w:sectPr");
612 if (sect_pr) {

616 }

618 break;
619 }
620 case BreakType::PageBreak:
621 case BreakType::ColumnBreak:
622 case BreakType::SectionBreakNextPage:
623 case BreakType::SectionBreakContinuous:
624 case BreakType::SectionBreakEvenPage:
625 case BreakType::SectionBreakOddPage: {
626 ensure_paragraph();
627 pugi::xml_node run = current_paragraph_.append_child("w:r");
628 pugi::xml_node br = run.append_child("w:br");
629 if (const char* typeval = break_type_to_string(break_type)) {
630 br.append_attribute("w:type").set_value(typeval);
631 }
632 break;
633 }
634 case BreakType::LineBreak: {
635 ensure_paragraph();
636 pugi::xml_node run = current_paragraph_.append_child("w:r");
637 run.append_child("w:br");
638 break;
639 }
640 }
641 if (doc_) {
642 doc_->mark_xml_paragraph_dirty(current_paragraph_);
643 }
644 return *this;

679 pugi::xml_node mark_run = current_paragraph_.append_child("w:r");
680 const pugi::xml_node mark_t = mark_run.append_child("w:t");
681 mark_t.text().set(reference_mark.c_str());

1164// PNG signature
1165bool is_png(const std::vector<uint8_t>& data) {
1166 return data.size() >= 8 && data[0] == 0x89 && data[1] == 0x50 && data[2] == 0x4E &&
1167 data[3] == 0x47 && data[4] == 0x0D && data[5] == 0x0A && data[6] == 0x1A &&
1168 data[7] == 0x0A;
1169}
1170
1171// JPEG signature
1172bool is_jpeg(const std::vector<uint8_t>& data) {
1173 return data.size() >= 3 && data[0] == 0xFF && data[1] == 0xD8 && data[2] == 0xFF;
1174}
1175
1176// GIF signature
1177bool is_gif(const std::vector<uint8_t>& data) {
1178 return data.size() >= 6 && data[0] == 'G' && data[1] == 'I' && data[2] == 'F' &&
1179 data[3] == '8' && (data[4] == '7' || data[4] == '9') && data[5] == 'a';
1180}
1181
1182// BMP signature
1183bool is_bmp(const std::vector<uint8_t>& data) {
1184 return data.size() >= 2 && data[0] == 'B' && data[1] == 'M';
1185}

1187// Read PNG dimensions from data
1188bool read_png_dimensions(const std::vector<uint8_t>& data, int& width, int& height) {
1189 if (data.size() < 24) {
1190 return false;
1191 }
1192 // IHDR chunk: width at offset 16, height at offset 20 (big-endian)
1193 width = (data[16] << 24) | (data[17] << 16) | (data[18] << 8) | data[19];
1194 height = (data[20] << 24) | (data[21] << 16) | (data[22] << 8) | data[23];
1195 return width > 0 && height > 0 && width < 100000 && height < 100000;
1196}
1197
1198// Read JPEG dimensions from data
1199bool read_jpeg_dimensions(const std::vector<uint8_t>& data, int& width, int& height) {
1200 size_t pos = 2; // Skip SOI marker
1201 while (pos < data.size()) {
1202 if (data[pos] != 0xFF) {
1203 pos++;
1204 continue;
1205 }
1206 while (pos < data.size() && data[pos] == 0xFF) {
1207 pos++;
1208 }
1209 if (pos >= data.size()) {
1210 break;
1211 }
1212 const uint8_t marker = data[pos++];
1213 if (marker == 0xC0 || marker == 0xC2) { // SOF0 or SOF2
1214 if (pos + 9 >= data.size()) {
1215 return false;
1216 }
1217 height = (data[pos + 3] << 8) | data[pos + 4];
1218 width = (data[pos + 5] << 8) | data[pos + 6];
1219 return width > 0 && height > 0 && width < 100000 && height < 100000;
1220 }
1221 if (marker == 0xD9) {
1222 break; // EOI
1223 }
1224 if (pos + 2 > data.size()) {
1225 break;
1226 }
1227 const uint16_t len = (data[pos] << 8) | data[pos + 1];
1228 pos += len;
1229 }
1230 return false;
1231}

1233// Read GIF dimensions from data
1234bool read_gif_dimensions(const std::vector<uint8_t>& data, int& width, int& height) {
1235 if (data.size() < 10) {
1236 return false;
1237 }
1238 width = data[6] | (data[7] << 8);
1239 height = data[8] | (data[9] << 8);
1240 return width > 0 && height > 0 && width < 100000 && height < 100000;
1241}
1242
1243// Read BMP dimensions from data
1244bool read_bmp_dimensions(const std::vector<uint8_t>& data, int& width, int& height) {
1245 if (data.size() < 26) {
1246 return false;
1247 }
1248 // DIB header size at offset 14 (little-endian uint32_t)
1249 const uint32_t dib_size = data[14] | (data[15] << 8) | (data[16] << 16) | (data[17] << 24);
1250 if (dib_size == 12) { // BITMAPCOREHEADER
1251 if (data.size() < 22) {
1252 return false;
1253 }
1254 width = data[18] | (data[19] << 8);
1255 height = data[20] | (data[21] << 8);
1256 } else { // BITMAPINFOHEADER or later
1257 width = data[18] | (data[19] << 8) | (data[20] << 16) |
1258 (data[21] << 24);
1259 auto h = static_cast<int32_t>(data[22] | (data[23] << 8) | (data[24] << 16) |
1260 (data[25] << 24));
1261 height = std::abs(h);
1262 }
1263 return width > 0 && height > 0 && width < 100000 && height < 100000;
1264}

1295bool detect_image_size(const std::string& image_path, ImageSize& size) {
1296 std::ifstream file(image_path, std::ios::binary);
1297 if (!file) {
1298 return false;
1299 }
1300
1301 // Read file header
1302 std::vector<uint8_t> data(65536); // Read up to 64KB
1303 file.read(reinterpret_cast<char*>(data.data()), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1304 static_cast<std::streamsize>(data.size()));
1305 const size_t bytes_read = file.gcount();
1306 data.resize(bytes_read);
1307
1308 return detect_image_size_from_memory(data, size);
1309}

1311bool detect_image_size_from_memory(const std::vector<uint8_t>& data, ImageSize& size) {
1312 if (data.size() < 8) {
1313 return false;
1314 }
1315
1316 int width = 0;
1317 int height = 0;
1318 bool success = false;
1319
1320 for (const auto& m : kImageFormatMappings) {
1321 if (m.detector(data)) {
1322 success = m.reader(data, width, height);
1323 break;
1324 }
1325 }
1326
1327 if (success) {
1328 // Convert pixels to points (assuming 96 DPI)
1329 size = ImageSize(width * 72.0 / 96.0, height * 72.0 / 96.0);
1330 }
1331
1332 return success;
1333}

1335ImageFormatInfo validate_image_format_detailed(const std::string& image_path) {
1336 ImageFormatInfo info;
1337 std::ifstream file(image_path, std::ios::binary);
1338
1339 if (!file) {
1340 info.error_message = "File not found or cannot be opened";
1341 return info;
1342 }
1343
1344 // Check file size
1345 file.seekg(0, std::ios::end);
1346 const size_t file_size = file.tellg();
1347 file.seekg(0);
1348
1349 if (file_size > static_cast<size_t>(50) * 1024 * 1024) {
1350 info.error_message = "File too large (max 50MB)";
1351 return info;
1352 }
1353
1354 // Read header
1355 std::vector<uint8_t> data(65536);
1356 file.read(reinterpret_cast<char*>(data.data()), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1357 static_cast<std::streamsize>(data.size()));
1358 data.resize(file.gcount());
1359
1360 // Detect format
1361 const char* format_name = nullptr;
1362 const char* mime_type = nullptr;
1363 if (!detect_image_format_type(data, &format_name, &mime_type)) {
1364 info.error_message = "Unknown or unsupported image format";
1365 return info;
1366 }
1367 info.format = format_name;
1368 info.mime_type = mime_type;
1369
1370 // Try to read dimensions
1371 ImageSize size;
1372 if (detect_image_size_from_memory(data, size)) {
1373 info.width = static_cast<int>(size.width_pt * 96.0 / 72.0);
1374 info.height = static_cast<int>(size.height_pt * 96.0 / 72.0);
1375 info.is_valid = true;
1376 } else {
1377 info.error_message = "Cannot read image dimensions, file may be corrupted";
1378 }
1379
1380 return info;
1381}
