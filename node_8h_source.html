281// Out-of-class definition of Node::get_ancestor (requires CompositeNode to be complete)
282template <typename T>
283T* Node::get_ancestor() const {
284 Node* current = parent_;
285 while (current) {
286 if (auto typed = dynamic_cast<T*>(current)) {
287 return typed;
288 }
289 current = current->get_parent();
290 }
291 return nullptr;
