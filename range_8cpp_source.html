44bool Range::replace(const std::string& old_text, const std::string& new_text) {
45 if (old_text.empty()) {
46 return false;
47 }
48
49 // Fallback for DOM paragraphs without XML node binding
50 if (!is_valid() && doc_) {
51 auto paragraphs = doc_->get_paragraphs();
52 for (auto& para : paragraphs) {
53 if (!para) {
54 continue;
55 }
56 std::string para_text = para->get_text();
57 const size_t pos = para_text.find(old_text);
58 if (pos != std::string::npos) {
59 para_text.replace(pos, old_text.size(), new_text);
60 para->set_text(para_text);
61 return true;
62 }
63 }
64 return false;
65 }
66
67 pugi::xml_node current = start_para_;
68 while (current) {
69 std::string para_text = collect_text_from_runs(current);
70
71 const size_t pos = para_text.find(old_text);
72 if (pos != std::string::npos) {
73 para_text.replace(pos, old_text.size(), new_text);
74
75 // Clear all runs and set first run to new text
76 pugi::xml_node first_run;
77 pugi::xml_node run = current.child("w:r");
78 while (run) {
79 const pugi::xml_node next = run.next_sibling("w:r");
80 if (!first_run) {
81 first_run = run;
82 // Clear existing text nodes
83 for (pugi::xml_node t = run.child("w:t"); t; t = t.next_sibling("w:t")) {
84 run.remove_child(t);
85 }
86 pugi::xml_node new_t = run.append_child("w:t");
87 if (!para_text.empty() &&
88 (std::isspace(static_cast<unsigned char>(para_text.front())) ||
89 std::isspace(static_cast<unsigned char>(para_text.back())))) {
90 new_t.append_attribute("xml:space").set_value("preserve");
91 }
92 new_t.text().set(para_text.c_str());
93 } else {
94 current.remove_child(run);
95 }
96 run = next;
97 }
98
99 // If no run existed, create one
100 if (!first_run) {
101 pugi::xml_node new_run = current.append_child("w:r");
102 pugi::xml_node new_t = new_run.append_child("w:t");
103 if (!para_text.empty() &&
104 (std::isspace(static_cast<unsigned char>(para_text.front())) ||
105 std::isspace(static_cast<unsigned char>(para_text.back())))) {
106 new_t.append_attribute("xml:space").set_value("preserve");
107 }
108 new_t.text().set(para_text.c_str());
109 }
110
111 return true;
112 }
113
114 if (current == end_para_) {
115 break;
116 }
117 current = current.next_sibling();
118 }
119 return false;
120}

122int Range::replace_all(const std::string& old_text, const std::string& new_text) {
123 if (old_text.empty()) {
124 return 0;
125 }
126
127 // Fallback for DOM paragraphs without XML node binding
128 if (!is_valid() && doc_) {
129 int total = 0;
130 auto paragraphs = doc_->get_paragraphs();
131 for (auto& para : paragraphs) {
132 if (!para) {
133 continue;
134 }
135 std::string para_text = para->get_text();
136 int count = 0;
137 size_t pos = 0;
138 while ((pos = para_text.find(old_text, pos)) != std::string::npos) {
139 para_text.replace(pos, old_text.size(), new_text);
140 pos += new_text.size();
141 ++count;
142 }
143 if (count > 0) {
144 para->set_text(para_text);
145 total += count;
146 }
147 }
148 return total;
149 }
150
151 int total = 0;
152 pugi::xml_node current = start_para_;
153 while (current) {
154 std::string para_text = collect_text_from_runs(current);
155
156 int count = 0;
157 size_t pos = 0;
158 while ((pos = para_text.find(old_text, pos)) != std::string::npos) {
159 para_text.replace(pos, old_text.size(), new_text);
160 pos += new_text.size();
161 ++count;
162 }
163
164 if (count > 0) {
165 ++total;
166 // Clear all runs and set first run to new text
167 pugi::xml_node first_run;
168 pugi::xml_node run = current.child("w:r");
169 while (run) {
170 const pugi::xml_node next = run.next_sibling("w:r");
171 if (!first_run) {
172 first_run = run;
173 for (pugi::xml_node t = run.child("w:t"); t; t = t.next_sibling("w:t")) {
174 run.remove_child(t);
175 }
176 pugi::xml_node new_t = run.append_child("w:t");
177 if (!para_text.empty() &&
178 (std::isspace(static_cast<unsigned char>(para_text.front())) ||
179 std::isspace(static_cast<unsigned char>(para_text.back())))) {
180 new_t.append_attribute("xml:space").set_value("preserve");
181 }
182 new_t.text().set(para_text.c_str());
183 } else {
184 current.remove_child(run);
185 }
186 run = next;
187 }
188
189 if (!first_run) {
190 pugi::xml_node new_run = current.append_child("w:r");
191 pugi::xml_node new_t = new_run.append_child("w:t");
192 if (!para_text.empty() &&
193 (std::isspace(static_cast<unsigned char>(para_text.front())) ||
194 std::isspace(static_cast<unsigned char>(para_text.back())))) {
195 new_t.append_attribute("xml:space").set_value("preserve");
196 }
197 new_t.text().set(para_text.c_str());
198 }
199 }
200
201 if (current == end_para_) {
202 break;
203 }
204 current = current.next_sibling();
205 }
206 return total;

209bool Range::delete_content() {
210 if (!is_valid()) {
211 return false;
212 }
213
214 pugi::xml_node current = start_para_;
215 while (current) {
216 // Remove all run elements
217 pugi::xml_node run = current.child("w:r");
218 while (run) {
219 const pugi::xml_node next = run.next_sibling("w:r");
220 current.remove_child(run);
221 run = next;
222 }
223
224 if (current == end_para_) {
225 break;
226 }
227 current = current.next_sibling();
228 }
229 return true;
230}
