13void apply_formatting_flags(pugi::xml_node r_pr, FormattingFlag f) {
14 if (f & kBold) {
15 r_pr.append_child("w:b");
16 }
17 if (f & kItalic) {
18 r_pr.append_child("w:i");
19 }
20 if (f & kUnderline) {
21 r_pr.append_child("w:u").append_attribute("w:val").set_value("single");
22 }
23 if (f & kStrikethrough) {
24 r_pr.append_child("w:strike").append_attribute("w:val").set_value("true");
25 }
26 if (f & kSuperscript) {
27 r_pr.append_child("w:vertAlign").append_attribute("w:val").set_value("superscript");
28 } else if (f & kSubscript) {
29 r_pr.append_child("w:vertAlign").append_attribute("w:val").set_value("subscript");
30 }
31 if (f & kSmallCaps) {
32 r_pr.append_child("w:smallCaps").append_attribute("w:val").set_value("true");
33 }
34 if (f & kShadow) {
35 r_pr.append_child("w:shadow").append_attribute("w:val").set_value("true");
36 }

89std::string Paragraph::get_text() const {
90 std::string result;
91 // Legacy fallback: if bound to XML but has no DOM children, read from XML directly
92 if (current_ && get_children().empty()) {
93 for (auto run = current_.child("w:r"); run; run = run.next_sibling("w:r")) {
94 auto t = run.child("w:t");
95 if (t) {
96 result += t.text().get();
97 }
98 }
99 return result;
100 }
101 for (const auto& child : get_children()) {
102 result += child->get_text();
103 }
104 return result;

351int Paragraph::join_runs_with_same_formatting() {
352 int merged_count = 0;
353 auto children = get_children();
354
355 for (size_t i = 0; i + 1 < children.size();) {
356 auto run1 = std::dynamic_pointer_cast<Run>(children[i]);
357 auto run2 = std::dynamic_pointer_cast<Run>(children[i + 1]);
358
359 if (!run1 || !run2) {
360 ++i;
361 continue;
362 }
363
364 if (run1->get_font() == run2->get_font()) {
365 run1->append_text(run2->get_text());
366 remove_child(run2);
367 children.erase(children.begin() + i + 1);
368 ++merged_count;
369 } else {
370 ++i;
371 }
372 }
373
374 return merged_count;
375}

431Run& Paragraph::add_run(const char* text, FormattingFlag f) {
432 if (!current_) {
433 return run_;
434 }
435
436 // Create new run element
437 pugi::xml_node new_run = current_.append_child("w:r");
438
439 // Create run properties element
440 const pugi::xml_node meta = new_run.append_child("w:rPr");
441
442 apply_formatting_flags(meta, f);
443
444 // Create text element
445 pugi::xml_node new_run_text = new_run.append_child("w:t");
446
447 // Preserve spaces if text starts or ends with whitespace
448 if (*text != 0 && (std::isspace(static_cast<unsigned char>(text[0])) ||
449 std::isspace(static_cast<unsigned char>(text[std::strlen(text) - 1])))) {
450 new_run_text.append_attribute("xml:space").set_value("preserve");
451 }
452
453 // Set text content
454 new_run_text.text().set(text);
455 run_.set_current_xml(new_run);
456
457 return run_;
458}

467Run& Paragraph::add_run_with_bookmark(Document& doc,
468 const char* text,

472 return run_;

477

479 pugi::xml_node bookmark_start = current_.append_child("w:bookmarkStart");
480 bookmark_start.append_attribute("w:id").set_value(bookmark_id);
481 bookmark_start.append_attribute("w:name").set_value(bookmark_name.c_str());
482

484 pugi::xml_node new_run = current_.append_child("w:r");

488

490
491 // Create text element
492 pugi::xml_node new_run_text = new_run.append_child("w:t");
493
494 // Preserve spaces if text starts or ends with whitespace
495 if (*text != 0 && (std::isspace(static_cast<unsigned char>(text[0])) ||
496 std::isspace(static_cast<unsigned char>(text[std::strlen(text) - 1])))) {
497 new_run_text.append_attribute("xml:space").set_value("preserve");
498 }
499

501 new_run_text.text().set(text);
502
503 // Create bookmark_end element after the run

534bool Paragraph::clear() {
535 if (!current_) {
536 return false;
537 }
538
539 // Remove all run elements
540 pugi::xml_node run = current_.child("w:r");
541 while (run) {
542 const pugi::xml_node next = run.next_sibling("w:r");
543 current_.remove_child(run);

545 }

784
785 // If this paragraph is backed by physical XML, update it as well.
786 if (!current_) {
787 return true;
788 }
789
790 pugi::xml_node p_pr = get_or_create_p_pr(current_);
791
792 // Remove existing num_pr if any
793 const pugi::xml_node existing_num_pr = p_pr.child("w:numPr");
794 if (existing_num_pr) {
795 p_pr.remove_child(existing_num_pr);
796 }
797
798 // Create num_pr element
799 pugi::xml_node num_pr = p_pr.append_child("w:numPr");
800
801 // Add ilvl (level)
802 pugi::xml_node ilvl = num_pr.append_child("w:ilvl");
803 ilvl.append_attribute("w:val").set_value(numbering_level_to_int(level));
804
805 // Add num_id
806 pugi::xml_node num_id_node = num_pr.append_child("w:numId");
807 num_id_node.append_attribute("w:val").set_value(static_cast<unsigned int>(num_id));
808
809 return true;
