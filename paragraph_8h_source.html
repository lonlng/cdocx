201class ParagraphCollection {

203
204 public:
205 using iterator = std::vector<std::shared_ptr<Paragraph>>::iterator;
206 using const_iterator = std::vector<std::shared_ptr<Paragraph>>::const_iterator;
207
208 ParagraphCollection() = default;
209 explicit ParagraphCollection(const std::vector<std::shared_ptr<Paragraph>>& paragraphs)
210 : paragraphs_(paragraphs) {}
211
212 size_t get_count() const { return paragraphs_.size(); }
213 std::shared_ptr<Paragraph> get_item(int index) const {
214 if (index >= 0 && static_cast<size_t>(index) < paragraphs_.size()) {
215 return paragraphs_[index];
216 }
217 return nullptr;
218 }
219 std::shared_ptr<Paragraph> operator[](int index) const { return get_item(index); }
220
221 iterator begin() { return paragraphs_.begin(); }
222 iterator end() { return paragraphs_.end(); }
223 const_iterator begin() const { return paragraphs_.begin(); }
224 const_iterator end() const { return paragraphs_.end(); }
225
226 std::shared_ptr<Paragraph> first() const {
227 return paragraphs_.empty() ? nullptr : paragraphs_.front();
228 }
229 std::shared_ptr<Paragraph> last() const {
230 return paragraphs_.empty() ? nullptr : paragraphs_.back();
231 }
232
233 std::string get_text() const;
234};
