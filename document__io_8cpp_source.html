86bool Document::load_tree_from_zip() {
87 if (!zip_handle_) {
88 return false;
89 }
90
91 const int n = zip_entries_total(zip_handle_);
92 if (n < 0) {
93 return false;
94 }
95
96 tree_.clear();
97
98 zip_entry_openbyindex(zip_handle_, 0);
99
100 for (int i = 0; i < n; i++) {
101 zip_entry_openbyindex(zip_handle_, i);
102
103 const char* name = zip_entry_name(zip_handle_);
104 if (!name) {
105 zip_entry_close(zip_handle_);
106 continue;
107 }
108
109 const std::string entry_name(name);
110 const bool is_dir = zip_entry_isdir(zip_handle_);
111
112 if (is_dir) {
113 zip_entry_close(zip_handle_);
114 continue;
115 }
116
117 // Read entry data
118 void* buf = nullptr;
119 size_t bufsize = 0;
120
121 if (zip_entry_read(zip_handle_, &buf, &bufsize) < 0) {
122 zip_entry_close(zip_handle_);
123 continue;
124 }
125
126 std::vector<uint8_t> data(static_cast<uint8_t*>(buf), static_cast<uint8_t*>(buf) + bufsize);
127 free(buf);
128
129 // Add to tree
130 auto node = tree_.add_zip_entry(entry_name, data);
131
132 // Parse XML files
133 if (string_ends_with(entry_name, ".xml") || string_ends_with(entry_name, ".rels")) {
134 node->type = DocxNodeType::XmlFile;
135 node->xml_doc = std::make_shared<pugi::xml_document>();
136
137 const pugi::xml_parse_result result = node->xml_doc->load_buffer(
138 data.data(),
139 data.size(),
140 pugi::parse_default | pugi::parse_declaration | pugi::parse_ws_pcdata);
141
142 if (!result) {
143 // XML parse failed, treat as binary
144 node->type = DocxNodeType::BinaryFile;
145 node->xml_doc.reset();
146 }
147 } else if (entry_name.find("word/media/") != std::string::npos) {
148 node->type = DocxNodeType::MediaFile;
149 } else {
150 node->type = DocxNodeType::BinaryFile;
151 }
152
153 zip_entry_close(zip_handle_);
154 }
155
156 return true;
157}

159LoadResult Document::load_tree_with_result() {
160 LoadResult result;
161 last_load_stats_.start_time = std::chrono::high_resolution_clock::now();
162
163 if (!zip_handle_) {
164 result.success = false;
165 result.errors.emplace_back(LoadErrorType::ZipOpenFailed, filepath_, "ZIP handle is null");
166 result.integrity = DocumentIntegrity::Corrupted;
167 last_load_result_ = result;
168 return result;
169 }
170
171 const int n = zip_entries_total(zip_handle_);
172 if (n < 0) {
173 result.success = false;
174 result.errors.emplace_back(
175 LoadErrorType::ZipOpenFailed, filepath_, "Failed to get entry count");
176 result.integrity = DocumentIntegrity::Corrupted;
177 last_load_result_ = result;
178 return result;
179 }
180
181 result.total_files = static_cast<size_t>(n);
182 last_load_stats_.total_entries = result.total_files;
183
184 tree_.clear();
185
186 // Use parallel loading when enabled and threshold is met
187 const bool use_parallel = load_config_.enable_parallel_loading &&
188 static_cast<size_t>(n) >= load_config_.parallel_threshold &&
189 std::thread::hardware_concurrency() > 1;
190
191 if (use_parallel) {
192 const bool parallel_ok = load_tree_parallel(last_load_stats_);
193 if (parallel_ok) {
194 last_load_stats_.end_time = std::chrono::high_resolution_clock::now();
195 result.success = last_load_stats_.xml_files > 0;
196 result.loaded_files = last_load_stats_.processed_entries;
197 result.load_time_ms = last_load_stats_.get_elapsed_ms();
198
199 if (result.errors.empty()) {
200 result.integrity = DocumentIntegrity::Complete;
201 } else if (result.loaded_files > result.errors.size() * 2) {
202 result.integrity = DocumentIntegrity::Partial;
203 } else {
204 result.integrity = DocumentIntegrity::Corrupted;
205 }
206
207 last_load_result_ = result;
208 return result;
209 }
210 // Fall back to sequential on failure
211 tree_.clear();
212 }
213
214 for (int i = 0; i < n; i++) {
215 if (zip_entry_openbyindex(zip_handle_, i) != 0) {
216 result.errors.emplace_back(
217 LoadErrorType::ZipEntryReadFailed, "", "Failed to open entry " + std::to_string(i));
218 continue;
219 }
220
221 const char* name = zip_entry_name(zip_handle_);
222 if (!name) {
223 zip_entry_close(zip_handle_);
224 continue;
225 }
226
227 const std::string entry_name(name);
228
229 if (zip_entry_isdir(zip_handle_)) {
230 zip_entry_close(zip_handle_);
231 continue;
232 }
233
234 // Read entry data
235 void* buf = nullptr;
236 size_t bufsize = 0;
237
238 if (zip_entry_read(zip_handle_, &buf, &bufsize) < 0) {
239 result.errors.emplace_back(
240 LoadErrorType::ZipEntryReadFailed, entry_name, "Failed to read entry");
241 zip_entry_close(zip_handle_);
242 continue;
243 }
244
245 std::vector<uint8_t> data(static_cast<uint8_t*>(buf), static_cast<uint8_t*>(buf) + bufsize);
246 free(buf);
247
248 // Add to tree
249 auto node = tree_.add_zip_entry(entry_name, data);
250
251 // Parse XML files
252 if (string_ends_with(entry_name, ".xml") || string_ends_with(entry_name, ".rels")) {
253 node->type = DocxNodeType::XmlFile;
254 node->xml_doc = std::make_shared<pugi::xml_document>();
255
256 const pugi::xml_parse_result parse_result = node->xml_doc->load_buffer(
257 data.data(),
258 data.size(),
259 pugi::parse_default | pugi::parse_declaration | pugi::parse_ws_pcdata);
260
261 if (parse_result) {
262 last_load_stats_.xml_files++;
263 } else {
264 result.errors.emplace_back(
265 LoadErrorType::XmlParseFailed, entry_name, "Failed to parse XML");
266 node->type = DocxNodeType::BinaryFile;
267 node->xml_doc.reset();
268 }
269 } else if (entry_name.find("word/media/") != std::string::npos) {
270 node->type = DocxNodeType::MediaFile;
271 last_load_stats_.media_files++;
272 } else {
273 node->type = DocxNodeType::BinaryFile;
274 last_load_stats_.binary_files++;
275 }
276
277 last_load_stats_.processed_entries++;
278
279 // Report progress
280 if (load_config_.progress_callback) {
281 const int percent = ((i + 1) * 100 / n);
282 load_config_.progress_callback(percent, entry_name);
283 }
284
285 zip_entry_close(zip_handle_);
286 }
287
288 last_load_stats_.end_time = std::chrono::high_resolution_clock::now();
289
290 result.success = last_load_stats_.xml_files > 0;
291 result.loaded_files = last_load_stats_.processed_entries;
292 result.load_time_ms = last_load_stats_.get_elapsed_ms();
293
294 // Determine integrity
295 if (result.errors.empty()) {
296 result.integrity = DocumentIntegrity::Complete;
297 } else if (result.loaded_files > result.errors.size() * 2) {
298 result.integrity = DocumentIntegrity::Partial;
299 } else {
300 result.integrity = DocumentIntegrity::Corrupted;
301 }
302
303 last_load_result_ = result;
304 return result;
305}

307bool Document::load_tree_parallel(LoadStatistics& stats) {
308 if (!zip_handle_) {
309 return false;
310 }
311
312 const int total_entries = zip_entries_total(zip_handle_);
313 if (total_entries < 0) {
314 return false;
315 }
316
317 struct EntryInfo {
318 int index = 0;
319 std::string name;
320 bool is_dir = false;
321 };
322
323 std::vector<EntryInfo> entries;
324 entries.reserve(total_entries);
325
326 // Phase 1: Collect metadata sequentially (zip handles are not thread-safe)
327 for (int i = 0; i < total_entries; ++i) {
328 if (zip_entry_openbyindex(zip_handle_, i) != 0) {
329 continue;
330 }
331
332 const char* name = zip_entry_name(zip_handle_);
333 if (!name) {
334 zip_entry_close(zip_handle_);
335 continue;
336 }
337
338 entries.push_back({i, name, zip_entry_isdir(zip_handle_) != 0});
339 zip_entry_close(zip_handle_);
340 }
341
342 std::vector<EntryInfo> files_to_load;
343 for (const auto& entry : entries) {
344 if (!entry.is_dir) {
345 files_to_load.push_back(entry);
346 }
347 }
348
349 if (files_to_load.empty()) {
350 return true;
351 }
352
353 size_t num_threads = load_config_.max_threads > 0 ? load_config_.max_threads
354 : std::thread::hardware_concurrency();
355 if (num_threads == 0) {
356 num_threads = 2;
357 }
358 num_threads = std::min(num_threads, files_to_load.size());
359
360 std::atomic<size_t> processed{0};
361 std::atomic<size_t> error_count{0};
362 std::atomic<size_t> xml_count{0};
363 std::atomic<size_t> media_count{0};
364 std::atomic<size_t> binary_count{0};
365
366 const size_t batch_size = (files_to_load.size() + num_threads - 1) / num_threads;
367 std::vector<std::thread> threads;
368
369 for (size_t t = 0; t < num_threads; ++t) {
370 const size_t start = t * batch_size;
371 const size_t end = std::min(start + batch_size, files_to_load.size());
372 if (start >= end) {
373 break;
374 }
375
376 threads.emplace_back([&, start, end]() {
377 // Each thread opens its own zip handle for thread safety
378 zip_t* local_zip = zip_open(filepath_.c_str(), 0, 'r');
379 if (!local_zip) {
380 error_count += (end - start);
381 return;
382 }
383
384 for (size_t i = start; i < end; ++i) {
385 const auto& entry = files_to_load[i];
386
387 if (zip_entry_openbyindex(local_zip, entry.index) != 0) {
388 ++error_count;
389 continue;
390 }
391
392 void* data = nullptr;
393 size_t size = 0;
394 if (zip_entry_read(local_zip, &data, &size) < 0 || !data) {
395 zip_entry_close(local_zip);
396 ++error_count;
397 continue;
398 }
399
400 std::vector<uint8_t> buffer(size);
401 std::memcpy(buffer.data(), data, size);
402 free(data);
403 zip_entry_close(local_zip);
404
405 // DocxTree::add_zip_entry is internally synchronized
406 auto node = tree_.add_zip_entry(entry.name, buffer);
407 if (!node) {
408 ++error_count;
409 continue;
410 }
411
412 // Classify for stats
413 if (node->type == DocxNodeType::XmlFile) {
414 ++xml_count;
415 } else if (node->type == DocxNodeType::MediaFile) {
416 ++media_count;
417 } else {
418 ++binary_count;
419 }
420
421 const size_t current = ++processed;
422
423 // Throttled progress reporting
424 if (load_config_.progress_callback && current % 10 == 0) {
425 const int percent = static_cast<int>((current * 100) / files_to_load.size());
426 load_config_.progress_callback(percent, entry.name);
427 }
428 }
429
430 zip_close(local_zip);
431 });
432 }
433
434 for (auto& thread : threads) {
435 if (thread.joinable()) {
436 thread.join();
437 }
438 }
439
440 stats.processed_entries = processed.load();
441 stats.xml_files = xml_count.load();
442 stats.media_files = media_count.load();
443 stats.binary_files = binary_count.load();
444
445 return error_count.load() < files_to_load.size();
446}

464bool Document::save_to_zip(const std::string& output_path) {
465 // On Windows, opening a file for writing while it is already open
466 // for reading fails. Close our read handle first if we are about
467 // to overwrite the same file.
468 if (output_path == filepath_ && zip_handle_) {
469 close_zip();
470 }
471
472 zip_t* zip = zip_open(output_path.c_str(), 9, 'w');
473 if (!zip) {
474 // Windows may need a brief moment to fully release the file handle
475 // after close_zip(). Retry once after a short delay.
476 std::this_thread::sleep_for(std::chrono::milliseconds(50));
477 zip = zip_open(output_path.c_str(), 9, 'w');
478 if (!zip) {
479 std::cerr << "[cdocx debug] zip_open failed for: " << output_path << std::endl;
480 return false;
481 }
482 }
483
484 const bool success = save_tree_to_zip(zip);
485
486 zip_close(zip);
487 return success;
488}

490bool Document::save_tree_to_zip(zip_t* zip) {
491 if (!zip) {
492 return false;
493 }
494
495 // First pass: create directories
496 tree_.iterate_all([zip](const std::shared_ptr<DocxTreeNode>& node) {
497 if (node->is_directory() && !node->name.empty()) {
498 zip_entry_open(zip, (node->full_path + "/").c_str());
499 zip_entry_close(zip);
500 }
501 });
502
503 // Second pass: write files
504 tree_.iterate_files([this, zip](const std::shared_ptr<DocxTreeNode>& node) {
505 if (node->is_deleted) {
506 return;
507 }
508
509 zip_entry_open(zip, node->full_path.c_str());
510
511 if (node->type == DocxNodeType::XmlFile && node->xml_doc) {
512 // Serialize XML
513 std::vector<uint8_t> data = node->serialize_xml_to_binary();
514 zip_entry_write(zip, data.data(), data.size());
515 } else {
516 // Write binary data
517 zip_entry_write(zip, node->binary_data.data(), node->binary_data.size());
518 }
519
520 zip_entry_close(zip);
521 });
522
523 return true;
524}
