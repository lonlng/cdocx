20// Extract merge field name from field code like "MERGEFIELD Name \\* MERGEFORMAT"
21std::string parse_merge_field_name(const std::string& field_code) {
22 const std::string code = trim_whitespace(field_code);
23 std::istringstream iss(code);
24 std::string keyword;
25 if (!(iss >> keyword)) {
26 return "";
27 }
28
29 if (to_lower(keyword) != "mergefield") {
30 return "";
31 }
32
33 // The rest is the field name and optional switches
34 std::string rest;
35 std::getline(iss, rest);
36 rest = trim_whitespace(rest);
37
38 // Remove switches (anything starting with \\‍)
39 const size_t switch_pos = rest.find("\\\\");
40 if (switch_pos != std::string::npos) {
41 rest = trim_whitespace(rest.substr(0, switch_pos));
42 }
43
44 return rest;
45}

66// Check if a paragraph contains only whitespace/empty runs
67bool is_empty_paragraph(const Paragraph* para) {
68 if (!para) {
69 return true;
70 }
71 for (const auto& child : para->get_children()) {
72 if (child->node_type() == NodeType::Run) {
73 auto* run = dynamic_cast<Run*>(child.get());
74 if (run && !trim_whitespace(run->get_text()).empty()) {
75 return false;
76 }
77 } else if (!is_structural_node(child->node_type())) {
78 return false;
79 }
80 }
81 return true;
82}

105void MailMerge::execute_impl(const std::map<std::string, std::string>& data) {
106 if (!doc_) {
107 return;
108 }
109
110 // Round-trip sync to unify DOM and physical XML state.
111 // This ensures MERGEFIELDs created via DOM or DocumentBuilder are visible.
112 doc_->sync_to_physical_tree();
113 doc_->sync_from_physical_tree();
114
115 bool removed_any_field = false;
116
117 auto sections = doc_->get_sections();
118 for (auto& section : sections) {
119 if (auto body = section->get_body()) {
120 for (const auto& child : body->get_children()) {
121 if (child->node_type() != NodeType::Paragraph) {
122 continue;
123 }
124 auto* para = dynamic_cast<Paragraph*>(child.get());
125 if (!para) {
126 continue;
127 }
128
129 auto children = para->get_children();
130 for (size_t i = 0; i < children.size(); ++i) {
131 auto& node = children[i];
132 if (node->node_type() != NodeType::FieldStart) {
133 continue;
134 }
135 auto* field = dynamic_cast<Field*>(node.get());
136 if (!field) {
137 continue;
138 }
139
140 std::string field_name = parse_merge_field_name(field->get_field_code());
141 if (field_name.empty()) {
142 continue;
143 }
144
145 removed_any_field = true;
146
147 auto it = std::find_if(data.begin(), data.end(), [&field_name](const auto& kv) {
148 return iequals(kv.first, field_name);
149 });
150
151 if (it != data.end()) {
152 // Replace field with a Run containing the value
153 auto run = std::make_shared<Run>(doc_, it->second);
154 para->insert_child(static_cast<int>(i), run);
155 para->remove_child(node);
156 } else {
157 // No value found: remove field entirely
158 para->remove_child(node);
159 }
160 }
161 }
162 }
163 }
164
165 if (removed_any_field) {
166 apply_cleanup();
167 doc_->sync_to_physical_tree();
168 }
169}

269void MailMerge::apply_cleanup() {
270 if (static_cast<std::uint8_t>(cleanup_options_ &
271 MailMergeCleanupOptions::RemoveEmptyParagraphs) == 0) {
272 return;
273 }
274
275 auto sections = doc_->get_sections();
276 for (auto& section : sections) {
277 if (auto body = section->get_body()) {
278 auto children = body->get_children();
279 for (auto& child : children) {
280 if (child->node_type() != NodeType::Paragraph) {
281 continue;
282 }
283 auto* para = dynamic_cast<Paragraph*>(child.get());
284 if (para && is_empty_paragraph(para)) {
285 body->remove_child(child);
