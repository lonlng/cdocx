97AbstractNumberingDefinition AbstractNumberingDefinition::make_numbered_list(NumberStyle style) {
98 AbstractNumberingDefinition def;
99 def.type = NumberingType::MultiLevel;
100
101 // Set up 9 levels with increasing indentation
102 for (int i = 0; i < 9; ++i) {
103 def.levels[i] = LevelDefinition::make_numbered_level(style, i);
104
105 // Set different styles for different levels in multi-level lists
106 if (i == 0) {
107 def.levels[i].number_style = style;
108 } else if (i == 1) {
109 def.levels[i].number_style =
110 (style == NumberStyle::Decimal) ? NumberStyle::LowerLetter : NumberStyle::Decimal;
111 } else if (i == 2) {
112 def.levels[i].number_style = NumberStyle::LowerRoman;
113 } else {
114 def.levels[i].number_style = NumberStyle::Decimal;
115 }
116
117 // Update format based on style
118 def.levels[i] = LevelDefinition::make_numbered_level(def.levels[i].number_style, i);
119 }
120
121 return def;
122}

148AbstractNumberingDefinition AbstractNumberingDefinition::make_outline_list() {
149 AbstractNumberingDefinition def;
150 def.type = NumberingType::HybridMultiLevel;
151
152 // Standard outline format: 1., 1.1, 1.1.1, etc.
153 for (int i = 0; i < 9; ++i) {
154 def.levels[i] = LevelDefinition::make_numbered_level(NumberStyle::Decimal, i);
155
156 // Build level text
157 std::stringstream ss;
158 for (int j = 0; j <= i; ++j) {
159 if (j > 0) {
160 ss << ".";
161 }
162 ss << "%" << (j + 1);
163 }
164 def.levels[i].level_text = ss.str();
165 def.levels[i].number_format = def.levels[i].level_text;
166 }
167
168 return def;
169}

256void NumberingManager::load_from_xml(pugi::xml_node numbering_root) {
257 clear();
258
259 // Parse abstract numbering definitions
260 for (const auto& abstract_num : numbering_root.children("w:abstractNum")) {
261 AbstractNumberingDefinition def;
262
263 // Parse abstract_numId
264 auto id_attr = abstract_num.attribute("w:abstractNumId");
265 if (id_attr) {
266 def.id = id_attr.as_uint();
267 if (def.id >= next_abstract_id_) {
268 next_abstract_id_ = def.id + 1;
269 }
270 }
271
272 // Parse levels
273 for (const auto& lvl : abstract_num.children("w:lvl")) {
274 auto ilvl = lvl.attribute("w:ilvl");
275 if (!ilvl) {
276 continue;
277 }
278
279 const int level = ilvl.as_int();
280 if (level < 0 || level >= 9) {
281 continue;
282 }
283
284 LevelDefinition level_def;
285
286 // Parse number format
287 auto num_fmt = lvl.child("w:numFmt");
288 if (num_fmt) {
289 const char* val = num_fmt.attribute("w:val").value();
290 level_def.number_style = string_to_number_style(val);
291 }
292
293 // Parse level text
294 auto lvl_text = lvl.child("w:lvlText");
295 if (lvl_text) {
296 level_def.level_text = lvl_text.attribute("w:val").value();
297 }
298
299 // Parse start number
300 auto start = lvl.child("w:start");
301 if (start) {
302 level_def.start_number = start.attribute("w:val").as_uint();
303 }
304
305 def.levels[level] = level_def;
306 }
307
308 abstract_definitions_[def.id] = def;
309 }
310
311 // Parse numbering instances
312 for (const auto& num : numbering_root.children("w:num")) {
313 NumberingDefinition def;
314
315 auto id_attr = num.attribute("w:numId");
316 if (id_attr) {
317 def.id = id_attr.as_uint();
318 if (def.id >= next_num_id_) {
319 next_num_id_ = def.id + 1;
320 }
321 }
322
323 auto abstract_id = num.child("w:abstractNumId");
324 if (abstract_id) {
325 def.abstract_id = abstract_id.attribute("w:val").as_uint();
326 }
327
328 num_definitions_[def.id] = def;
329 }
330
331 modified_ = false;
332}

334void NumberingManager::save_to_xml(pugi::xml_node numbering_root) {
335 // Clear existing content
336 for (const auto& child : numbering_root.children()) {
337 numbering_root.remove_child(child);
338 }
339
340 // Add namespace
341 numbering_root.append_attribute("xmlns:w").set_value(
342 "http://schemas.openxmlformats.org/wordprocessingml/2006/main");
343
344 // Save abstract numbering definitions
345 for (const auto& [id, def] : abstract_definitions_) {
346 auto abstract_num = numbering_root.append_child("w:abstractNum");
347 abstract_num.append_attribute("w:abstractNumId").set_value(static_cast<unsigned int>(id));
348
349 // Add multi-level type if applicable
350 if (def.type == NumberingType::MultiLevel || def.type == NumberingType::HybridMultiLevel) {
351 auto multi_level_type = abstract_num.append_child("w:multi_level_type");
352 multi_level_type.append_attribute("w:val").set_value(
353 def.type == NumberingType::HybridMultiLevel ? "hybridMultilevel" : "multilevel");
354 }
355
356 // Save levels
357 for (int i = 0; i < 9; ++i) {
358 const auto& level = def.levels[i];
359
360 auto lvl = abstract_num.append_child("w:lvl");
361 lvl.append_attribute("w:ilvl").set_value(i);
362
363 // Start number
364 auto start = lvl.append_child("w:start");
365 start.append_attribute("w:val").set_value(static_cast<unsigned int>(level.start_number));
366
367 // Number format
368 auto num_fmt = lvl.append_child("w:numFmt");
369 num_fmt.append_attribute("w:val").set_value(number_style_to_string(level.number_style));
370
371 // Level text
372 auto lvl_text = lvl.append_child("w:lvlText");
373 lvl_text.append_attribute("w:val").set_value(level.level_text.c_str());
374
375 // Level justification
376 auto lvl_jc = lvl.append_child("w:lvlJc");
377 lvl_jc.append_attribute("w:val").set_value(pp_alignment_to_string(level.number_alignment));
378 }
379 }
380
381 // Save numbering instances
382 for (const auto& [id, def] : num_definitions_) {
383 auto num = numbering_root.append_child("w:num");
384 num.append_attribute("w:numId").set_value(static_cast<unsigned int>(id));
385
386 auto abstract_num_id = num.append_child("w:abstractNumId");
387 abstract_num_id.append_attribute("w:val").set_value(
388 static_cast<unsigned int>(def.abstract_id));
389
390 // Save level overrides
391 for (const auto& [level, level_def] : def.level_overrides) {
392 auto lvl_override = num.append_child("w:lvlOverride");
393 lvl_override.append_attribute("w:ilvl").set_value(numbering_level_to_int(level));
394
395 auto lvl = lvl_override.append_child("w:lvl");
396 lvl.append_attribute("w:ilvl").set_value(numbering_level_to_int(level));
397
398 // Override properties would be saved here
399 (void)level_def; // Avoid unused warning
400 }
401 }
402}

441const char* number_style_to_string(NumberStyle style) {
442 for (const auto& m : kNumberStyleMappings) {
443 if (m.style == style) {
444 return m.xml_value;
445 }
446 }
447 return "decimal";
448}

450NumberStyle string_to_number_style(const char* val) {
451 if (!val || !*val) {
452 return NumberStyle::Decimal;
453 }
454 for (const auto& m : kNumberStyleMappings) {
455 if (std::strcmp(m.xml_value, val) == 0) {
456 return m.style;
457 }
458 }
459 return NumberStyle::Decimal;
460}
