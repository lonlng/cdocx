68bool Template::try_replace_in_text(std::string& text) {
69 bool replaced = false;
70
71 for (const auto& [key, value] : placeholders_) {
72 const std::string pattern = pattern_prefix_ + key + pattern_suffix_;
73 size_t pos = 0;
74 while ((pos = text.find(pattern, pos)) != std::string::npos) {
75 text.replace(pos, pattern.length(), value);
76 pos += value.length();
77 replaced = true;
78 }
79 }
80
81 return replaced;
82}

88bool Template::try_replace_single_run(Run& r, bool first_only) {
89 std::string text = r.get_text();
90 if (first_only) {
91 // Find the earliest match across all keys (text order, not key order)
92 size_t best_pos = std::string::npos;
93 std::string best_value;
94 std::string best_pattern;
95 for (const auto& [key, value] : placeholders_) {
96 const std::string pattern = pattern_prefix_ + key + pattern_suffix_;
97 const size_t pos = text.find(pattern);
98 if (pos != std::string::npos &&
99 (best_pos == std::string::npos || pos < best_pos)) {
100 best_pos = pos;
101 best_value = value;
102 best_pattern = pattern;
103 }
104 }
105 if (best_pos != std::string::npos) {
106 text.replace(best_pos, best_pattern.length(), best_value);
107 r.set_text(text);
108 return true;
109 }
110 return false;
111 }
112 if (try_replace_in_text(text)) {
113 r.set_text(text);
114 return true;
115 }
116 return false;
117}

129bool Template::try_replace_placeholder(const PlaceholderContext& ctx, Paragraph& p) {
130 (void)p;
131 if (!ctx.first_run) {
132 return false;
133 }
134
135 size_t best_pos = std::string::npos;
136 std::string best_value;
137 std::string best_pattern;
138
139 for (const auto& [key, value] : placeholders_) {
140 const std::string pattern = pattern_prefix_ + key + pattern_suffix_;
141 const size_t pos = ctx.collected_text.find(pattern);
142 if (pos != std::string::npos) {
143 if (best_pos == std::string::npos || pos < best_pos) {
144 best_pos = pos;
145 best_value = value;
146 best_pattern = pattern;
147 }
148 }
149 }
150
151 if (best_pos == std::string::npos) {
152 return false;
153 }
154
155 // =========================================================================
156 // Format preservation: copy the format from the run(s) containing the
157 // actual key text (between prefix and suffix) to the first run.
158 // =========================================================================
159 const size_t key_start_in_collected = best_pos + pattern_prefix_.length();
160 const size_t key_end_in_collected =
161 best_pos + best_pattern.length() - pattern_suffix_.length();
162
163 if (key_start_in_collected < key_end_in_collected) {
164 const size_t first_portion_len =
165 ctx.first_run->get_text().length() - ctx.prefix_pos;
166
167 if (key_start_in_collected >= first_portion_len) {
168 // Key text starts in collected runs; find the first overlapping run
169 // and copy its formatting to the first run.
170 size_t offset = first_portion_len;
171 for (Run* run : ctx.runs_to_delete) {
172 const size_t run_len = run->get_text().length();
173 if (offset + run_len > key_start_in_collected &&
174 offset < key_end_in_collected) {
175 ctx.first_run->get_font() = run->get_font();
176 if (run->has_preserved_r_pr()) {
177 ctx.first_run->preserve_r_pr(run->get_preserved_r_pr());
178 }
179 break;
180 }
181 offset += run_len;
182 }
183 }
184 }
185
186 const size_t pattern_end = best_pos + best_pattern.length();
187 const std::string trailing = ctx.collected_text.substr(pattern_end);
188
189 const std::string first_run_text = ctx.first_run->get_text();
190 const std::string new_text = first_run_text.substr(0, ctx.prefix_pos) + best_value + trailing;
191 ctx.first_run->set_text(new_text);
192 return true;
193}

206bool Template::process_paragraph(Paragraph& p, bool stop_after_first) {
207 PlaceholderContext ctx;
208 ctx.clear();
209
210 int i = 0;
211 while (i < static_cast<int>(p.get_child_count())) {
212 auto child = p.get_child(i);
213 auto run = std::dynamic_pointer_cast<Run>(child);
214 if (!run) {
215 if (ctx.first_run) {
216 ctx.clear();
217 }
218 ++i;
219 continue;
220 }
221
222 std::string text = run->get_text();
223
224 if (!ctx.first_run) {
225 // Idle state: try single-run replacement first
226 if (try_replace_single_run(*run, stop_after_first)) {
227 if (stop_after_first) {
228 return true;
229 }
230 text = run->get_text();
231 }
232
233 const size_t prefix_start = text.rfind(pattern_prefix_);
234 if (prefix_start != std::string::npos) {
235 const std::string candidate = text.substr(prefix_start);
236 bool has_complete = false;
237 for (const auto& [key, value] : placeholders_) {
238 (void)value;
239 const std::string pattern = pattern_prefix_ + key + pattern_suffix_;
240 if (candidate.find(pattern) != std::string::npos) {
241 has_complete = true;
242 break;
243 }
244 }
245 if (!has_complete) {
246 transition_to_collecting_state(ctx, *run, text, prefix_start);
247 }
248 }
249 ++i;
250 } else {
251 // Collecting state: accumulate text across runs
252 ctx.collected_text += text;
253 ctx.runs_to_delete.push_back(run.get());
254
255 if (try_replace_placeholder(ctx, p)) {
256 delete_collected_runs(ctx, p);
257 if (stop_after_first) {
258 return true;
259 }
260 ctx.clear();
261 i = 0;
262 continue;
263 }
264 ++i;
265 }
266 }
267
268 if (ctx.first_run) {
269 ctx.clear();
270 }
271 return false;
272}

283bool Template::replace_image_in_run(const std::shared_ptr<Run>& run) {
284 if (!run || !doc_) {
285 return false;
286 }
287
288 const std::string text = run->get_text();
289 for (const auto& [key, image_path] : image_placeholders_) {
290 const std::string pattern = pattern_prefix_ + key + pattern_suffix_;
291 if (text != pattern) {
292 continue;
293 }
294
295 if (!std::filesystem::exists(image_path)) {
296 continue;
297 }
298
299 ImageSize size;
300 if (!detect_image_size(image_path, size)) {
301 size = ImageSize(400, 300);
302 }
303
304 const std::string rel_id = doc_->add_media_with_rel(image_path, nullptr);
305 if (rel_id.empty()) {
306 continue;
307 }
308
309 run->set_text("");
310
311 pugi::xml_document drawing_doc;
312 auto drawing = append_image_drawing(
313 drawing_doc, rel_id, size, ImageAlignment::Center,
314 image_id_counter_++, image_path);
315
316 run->preserve_child(drawing);
317 return true;
318 }
319 return false;
320}

322bool Template::replace_in_paragraph(const std::shared_ptr<Paragraph>& para) {
323 if (!para) {
324 return false;
325 }
326
327 bool replaced = false;
328 // First pass: handle image placeholders (exact single-run match)
329 for (const auto& child : para->get_children()) {
330 if (auto run = std::dynamic_pointer_cast<Run>(child)) {
331 if (replace_image_in_run(run)) {
332 replaced = true;
333 }
334 }
335 }
336
337 // Second pass: handle text placeholders with FSM (cross-run support)
338 if (process_paragraph(*para)) {
339 replaced = true;
340 }
341 return replaced;
