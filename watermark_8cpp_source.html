41static std::string escape_xml(const std::string& text) {
42 std::string result;
43 result.reserve(text.size());
44 for (const char c : text) {
45 bool found = false;
46 for (const auto& mapping : kXmlEscapeMappings) {
47 if (mapping.c == c) {
48 result += mapping.replacement;
49 found = true;
50 break;
51 }
52 }
53 if (!found) {
54 result += c;
55 }
56 }
57 return result;
58}

72static void remove_watermark_nodes(pugi::xml_node header) {
73 if (!header) {
74 return;
75 }
76 for (pugi::xml_node para = header.child("w:p"); para;) {
77 bool is_watermark = false;
78 for (pugi::xml_node run = para.child("w:r"); run; run = run.next_sibling("w:r")) {
79 for (pugi::xml_node pict = run.child("w:pict"); pict;
80 pict = pict.next_sibling("w:pict")) {
81 const pugi::xml_node shape = pict.child("v:shape");
82 if (shape) {
83 if (is_word_watermark_shape(shape)) {
84 is_watermark = true;
85 break;
86 }
87 }
88 }
89 if (is_watermark) {
90 break;
91 }
92 for (pugi::xml_node drawing = run.child("w:drawing"); drawing;
93 drawing = drawing.next_sibling("w:drawing")) {
94 const pugi::xml_node doc_pr =
95 drawing.child("wp:anchor").child("a:graphic").child("a:graphicData");
96 if (doc_pr) {
97 const pugi::xml_attribute name_attr = doc_pr.attribute("uri");
98 if (name_attr && std::strstr(name_attr.value(), "watermark") != nullptr) {
99 is_watermark = true;
100 break;
101 }
102 }
103 }
104 if (is_watermark) {
105 break;
106 }
107 }
108 const pugi::xml_node next_para = para.next_sibling("w:p");
109 if (is_watermark) {
110 header.remove_child(para);
111 }
112 para = next_para;
113 }
114}

286void Watermark::insert_image_watermark_into_header(const std::string& image_path,
287 const ImageWatermarkOptions& options) {
288 if (!document_) {
289 return;
290 }
291
292 // Read image file into memory
293 std::ifstream file(image_path, std::ios::binary);
294 if (!file) {
295 return;
296 }
297 const std::vector<uint8_t> data((std::istreambuf_iterator<char>(file)),
298 std::istreambuf_iterator<char>());
299 file.close();
300 if (data.empty()) {
301 return;
302 }
303
304 // Determine media filename and ensure uniqueness
305 std::string filename = std::filesystem::path(image_path).filename().string();
306 if (document_->has_media(filename)) {
307 filename = document_->generate_unique_image_name(filename);
308 }
309
310 // Add media to document
311 if (!document_->add_media_from_memory(filename, data)) {
312 return;
313 }
314
315 // Find the first header and its rels path
316 auto header_names = document_->get_header_names();
317 if (header_names.empty()) {
318 return;
319 }
320
321 const std::string& header_path = header_names[0];
322 std::string rels_path;
323 if (header_path.find('/') != std::string::npos) {
324 rels_path = header_path.substr(0, header_path.rfind('/') + 1) + "_rels/" +
325 header_path.substr(header_path.rfind('/') + 1) + ".rels";
326 } else {
327 rels_path = "word/_rels/" + header_path + ".rels";
328 }
329
330 // Create relationship in header rels
331 const std::string rel_id = document_->add_relationship(
332 rels_path,
333 "http://schemas.openxmlformats.org/officeDocument/2006/relationships/image",
334 "media/" + filename);
335 if (rel_id.empty()) {
336 return;
337 }
338
339 pugi::xml_node header = get_first_header_xml(document_);
340 if (!header) {
341 return;
342 }
343
344 auto para = header.append_child("w:p");
345 auto p_pr = para.append_child("w:pPr");
346 auto jc = p_pr.append_child("w:jc");
347 jc.append_attribute("w:val").set_value("center");
348
349 auto run = para.append_child("w:r");
350 auto pict = run.append_child("w:pict");
351 auto shape = pict.append_child("v:shape");
352 shape.append_attribute("id").set_value("PowerPlusWaterMarkObject");
353 shape.append_attribute("style").set_value(
354 "position:absolute;margin-left:0;margin-top:0;width:400pt;height:300pt;z-index:-251658752;"
355 "mso-position-horizontal:center;mso-position-horizontal-relative:margin;mso-position-"
356 "vertical:center;mso-position-vertical-relative:margin");
357 shape.append_attribute("o:allowincell").set_value("f");
358 shape.append_attribute("stroked").set_value("f");
359
360 // Apply scale if explicitly set (> 0)
361 if (options.scale > 0) {
362 const int width_pt = static_cast<int>(400.0 * options.scale / 100.0);
363 const int height_pt = static_cast<int>(300.0 * options.scale / 100.0);
364 const std::string style =
365 "position:absolute;margin-left:0;margin-top:0;width:" + std::to_string(width_pt) +
366 "pt;height:" + std::to_string(height_pt) +
367 "pt;z-index:-251658752;"
368 "mso-position-horizontal:center;mso-position-horizontal-relative:margin;"
369 "mso-position-vertical:center;mso-position-vertical-relative:margin";
370 shape.attribute("style").set_value(style.c_str());
371 }
372
373 // Apply washout effect (grayscale + reduced contrast)
374 if (options.washout) {
375 auto fill = shape.append_child("v:fill");
376 fill.append_attribute("type").set_value("frame");
377 fill.append_attribute("opacity").set_value("0.5");
378 auto image = shape.append_child("v:image");
379 image.append_attribute("o:title").set_value("Watermark");
380 image.append_attribute("cropleft").set_value("f");
381 }
382
383 auto imagedata = shape.append_child("v:imagedata");
384 imagedata.append_attribute("r:id").set_value(rel_id.c_str());
385 imagedata.append_attribute("o:title").set_value("Watermark");
386}
