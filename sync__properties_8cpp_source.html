28 set_text_child(root, "dc:description", builtin_properties_.comments);
29 set_text_child(root,

31 builtin_properties_.manager); // using manager as lastModifiedBy
32 set_text_child(root, "cp:category", builtin_properties_.category);
33 set_text_child(root, "cp:contentStatus", builtin_properties_.content_status);
34 set_text_child(root, "cp:contentType", builtin_properties_.content_type);
35 set_text_child(root, "cp:revision", builtin_properties_.revision);

38 auto created = get_or_create_child(root, "dcterms:created");
39 created.append_attribute("xsi:type").set_value("dcterms:W3CDTF");
40 created.append_child(pugi::node_pcdata)

44 auto modified = get_or_create_child(root, "dcterms:modified");

46 modified.append_child(pugi::node_pcdata)
47 .set_value(time_to_w3cdtf(builtin_properties_.modified).c_str());
48 }
49 if (builtin_properties_.last_printed != 0) {
50 set_text_child(
51 root, "cp:lastPrinted", time_to_w3cdtf(builtin_properties_.last_printed));
52 }
53 }
54 mark_modified("docProps/core.xml");
55 }

174void Document::sync_custom_properties_to_physical() {
175 if (custom_properties_.custom_properties.empty()) {
176 // Remove custom.xml if no custom properties
177 if (has_xml_part("docProps/custom.xml")) {
178 remove_xml_part("docProps/custom.xml");
179 // Remove relationship
180 const std::string rel_id = find_relationship_id("_rels/.rels", "docProps/custom.xml");
181 if (!rel_id.empty()) {
182 remove_relationship("_rels/.rels", rel_id);
183 }
184 }
185 return;
186 }
187
188 pugi::xml_document* custom_doc = get_xml_part("docProps/custom.xml");
189 if (!custom_doc) {
190 custom_doc = &create_xml_part("docProps/custom.xml");
191 add_content_type_override(
192 "/docProps/custom.xml",
193 "application/vnd.openxmlformats-officedocument.custom-properties+xml");
194
195 const std::string rel_id = find_relationship_id("_rels/.rels", "docProps/custom.xml");
196 if (rel_id.empty()) {
197 add_relationship("_rels/.rels",
198 "http://schemas.openxmlformats.org/officeDocument/2006/relationships/"
199 "custom-properties",
200 "docProps/custom.xml");
201 }
202 }
203
204 custom_doc->reset();
205 auto root = custom_doc->append_child("Properties");
206 root.append_attribute("xmlns").set_value(
207 "http://schemas.openxmlformats.org/officeDocument/2006/custom-properties");
208 root.append_attribute("xmlns:vt")
209 .set_value("http://schemas.openxmlformats.org/officeDocument/2006/docPropsVTypes");
210
211 int pid = 2;
212 for (const auto& kv : custom_properties_.custom_properties) {
213 auto prop = root.append_child("property");
214 prop.append_attribute("fmtid").set_value("{D5CDD505-2E9C-101B-9397-08002B2CF9AE}");
215 prop.append_attribute("pid").set_value(pid++);
216 prop.append_attribute("name").set_value(kv.first.c_str());
217 auto val = prop.append_child("vt:lpwstr");
218 val.append_child(pugi::node_pcdata).set_value(kv.second.c_str());
219 }
220
221 mark_modified("docProps/custom.xml");
222}

224void Document::sync_custom_properties_from_physical() {
225 custom_properties_.custom_properties.clear();
226
227 pugi::xml_document* custom_doc = get_xml_part("docProps/custom.xml");
228 if (!custom_doc) {
229 return;
230 }
231
232 auto root = custom_doc->child("Properties");
233 if (!root) {
234 return;
235 }
236
237 for (auto prop = root.child("property"); prop; prop = prop.next_sibling("property")) {
238 const pugi::xml_attribute name_attr = prop.attribute("name");
239 if (!name_attr) {
240 continue;
241 }
242
243 std::string value;
244 if (auto lpwstr = prop.child("vt:lpwstr").first_child()) {
245 value = lpwstr.value();
246 } else if (auto text = prop.first_child().first_child()) {
247 value = text.value();
248 }
249
250 custom_properties_.custom_properties[name_attr.value()] = value;
251 }
252}
