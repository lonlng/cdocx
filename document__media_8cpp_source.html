22bool Document::add_media(const std::string& image_path, const std::string* image_name) {
23 if (!is_open()) {
24 return false;
25 }
26 if (!std::filesystem::exists(image_path)) {
27 return false;
28 }
29
30 // Determine the filename in the document
31 std::string filename;
32 if (image_name && !image_name->empty()) {
33 filename = *image_name;
34 } else {
35 filename = std::filesystem::path(image_path).filename().string();
36 }
37
38 // Generate unique name if already exists
39 std::string media_path = "word/media/" + filename;
40 if (tree_.find_node(media_path)) {
41 filename = generate_unique_image_name(filename);
42 media_path = "word/media/" + filename;
43 }
44
45 // Read image file
46 std::ifstream file(image_path, std::ios::binary);
47 if (!file) {
48 return false;
49 }
50
51 const std::vector<uint8_t> data((std::istreambuf_iterator<char>(file)),
52 std::istreambuf_iterator<char>());
53 file.close();
54
55 if (data.empty()) {
56 return false;
57 }
58
59 // Add to tree
60 auto node = tree_.add_zip_entry(media_path, data);
61 if (!node) {
62 return false;
63 }
64
65 node->type = DocxNodeType::MediaFile;
66 node->content_type = get_mime_type(filename);
67 node->is_new = true;
68 node->is_modified = true;
69 media_files_cache_[media_path] = node;
70
71 // Register content type
72 add_content_type_override("/" + media_path, get_mime_type(filename));
73
74 return true;
75}

353std::string Document::get_mime_type(const std::string& filename) const {
354 std::string ext = std::filesystem::path(filename).extension().string();
355 if (!ext.empty() && ext[0] == '.') {
356 ext = ext.substr(1);
357 }
358 ext = to_lower(ext);
359
360 static const std::map<std::string, std::string> kMimeMap = {
361 {"png", "image/png"},
362 {"jpg", "image/jpeg"},
363 {"jpeg", "image/jpeg"},
364 {"gif", "image/gif"},
365 {"bmp", "image/bmp"},
366 {"tiff", "image/tiff"},
367 {"tif", "image/tiff"},
368 {"wmf", "image/x-wmf"},
369 {"emf", "image/x-emf"},
370 {"svg", "image/svg+xml"},
371 };
372
373 auto it = kMimeMap.find(ext);
374 if (it != kMimeMap.end()) {
375 return it->second;
376 }
377 return "application/octet-stream";
378}

399std::string Document::generate_unique_image_name(const std::string& base_name) const {
400 const std::filesystem::path path(base_name);
401 const std::string stem = path.stem().string();
402 const std::string ext = path.extension().string();
403
404 std::string name = base_name;
405 int counter = 1;
406
407 while (has_media(name)) {
408 name = stem;
409 name += "_";
410 name += std::to_string(counter);
411 name += ext;
412 counter++;
413 }
414
415 return name;
416}
