47std::shared_ptr<Node> Node::get_previous_node_in_document() const {
48 auto prev = prev_sibling_.lock();
49 if (prev) {
50 // Pre-order: previous sibling's deepest last descendant
51 auto current = prev;
52 while (current->is_composite()) {
53 auto composite = std::dynamic_pointer_cast<CompositeNode>(current);
54 if (!composite || !composite->has_children()) {
55 break;
56 }
57 current = composite->get_last_child();
58 }
59 return current;
60 }
61 // No previous sibling: return parent
62 if (parent_) {
63 return static_cast<Node*>(parent_)->shared_from_this();
64 }
65 return nullptr;
66}

68std::shared_ptr<Node> Node::get_next_node_in_document() const {
69 // Pre-order: first child, then next sibling, then ancestor's next sibling
70 if (is_composite()) {
71 const auto* composite = dynamic_cast<const CompositeNode*>(this);
72 if (composite && composite->has_children()) {
73 return composite->get_first_child();
74 }
75 }
76 auto next = next_sibling_.lock();
77 if (next) {
78 return next;

80 auto* current = parent_;
81 while (current) {
82 if (current->get_next_sibling()) {
83 return current->get_next_sibling();
84 }
85 current = current->get_parent();
86 }
87 return nullptr;

169std::shared_ptr<Node> CompositeNode::insert_child(int index, std::shared_ptr<Node> child) {
170 if (!child || index < 0 || static_cast<size_t>(index) > children_.size()) {
171 return nullptr;
172 }
173
174 child->set_parent(this);
175 child->set_document(document_);
176
177 if (index == 0) {
178 if (!children_.empty()) {
179 child->set_next_sibling(children_.front());
180 children_.front()->set_previous_sibling(child);
181 }
182 children_.insert(children_.begin(), child);
183 } else {
184 auto it = children_.begin() + index;
185 child->set_previous_sibling(*(it - 1));
186 child->set_next_sibling(*it);
187 (*(it - 1))->set_next_sibling(child);
188 (*it)->set_previous_sibling(child);
189 children_.insert(it, child);
190 }
191
192 return child;
193}

223void CompositeNode::remove_child(const std::shared_ptr<Node>& child) {
224 if (!child) {
225 return;
226 }
227
228 for (auto it = children_.begin(); it != children_.end(); ++it) {
229 if (*it == child) {
230 // Update sibling links
231 if (child->get_previous_sibling()) {
232 child->get_previous_sibling()->set_next_sibling(child->get_next_sibling());
233 }
234 if (child->get_next_sibling()) {
235 child->get_next_sibling()->set_previous_sibling(child->get_previous_sibling());
236 }
237
238 child->set_parent(nullptr);
239 children_.erase(it);
240 return;
241 }
242 }
243}
