169 // Get or create r_pr node

172 r_pr = run_node.prepend_child("w:rPr");

174
175 // Font settings
176 if (font) {
177 pugi::xml_node r_fonts = r_pr.child("w:rFonts");
178 if (!r_fonts) {
179 r_fonts = r_pr.append_child("w:rFonts");
180 }
181 if (!font->ascii.empty()) {
182 r_fonts.append_attribute("w:ascii").set_value(font->ascii.c_str());
183 }
184 if (!font->east_asia.empty()) {

186 }
187 if (!font->h_ansi.empty()) {
188 r_fonts.append_attribute("w:hAnsi").set_value(font->h_ansi.c_str());
189 }
190 if (!font->cs.empty()) {
191 r_fonts.append_attribute("w:cs").set_value(font->cs.c_str());
192 }
193 // Font hint
194 if (font->hint != Font::Hint::Default) {
195 if (const char* hint_str = font_hint_to_string(font->hint)) {
196 r_fonts.append_attribute("w:hint").set_value(hint_str);

199 }
200
201 // Font style (bold/italic)
202 if (font_style.bold) {
203 r_pr.append_child("w:b");
204 }
205 if (font_style.italic) {
206 r_pr.append_child("w:i");

209 // Font size
210 if (font_size > 0) {
211 pugi::xml_node sz = r_pr.append_child("w:sz");
212 sz.append_attribute("w:val").set_value(font_size);
213 pugi::xml_node sz_cs = r_pr.append_child("w:szCs");
214 sz_cs.append_attribute("w:val").set_value(font_size);
215 }
216
217 // Color
218 if (!color.empty()) {
219 pugi::xml_node color_node = r_pr.append_child("w:color");
220 color_node.append_attribute("w:val").set_value(color.c_str());
221 }
222
223 // Underline
224 if (underline.style != UnderlineStyle::None) {
225 pugi::xml_node u = r_pr.append_child("w:u");
226 const char* style_str = "single";
227 for (const auto& m : kUnderlineStyleMappings) {
228 if (m.style == underline.style) {
229 style_str = m.xml_value;
230 break;
231 }
232 }
233 u.append_attribute("w:val").set_value(style_str);
234 if (underline.color != "auto") {
235 u.append_attribute("w:color").set_value(underline.color.c_str());
236 }
237 }
238
239 // Strikethrough
240 if (strike == StrikeStyle::Single) {
241 r_pr.append_child("w:strike");
242 } else if (strike == StrikeStyle::Double) {
243 pugi::xml_node strike_node = r_pr.append_child("w:dstrike");
244 strike_node.append_attribute("w:val").set_value("true");
245 }
246
247 // Vertical align

249 pugi::xml_node v_align = r_pr.append_child("w:vertAlign");
250 v_align.append_attribute("w:val").set_value(align_str);

252
253 // Highlight

255 if (const char* hl_str = highlight_to_string(highlight)) {
256 pugi::xml_node highlight_node = r_pr.append_child("w:highlight");
257 highlight_node.append_attribute("w:val").set_value(hl_str);
258 }
259 }
260
261 // Scale
262 if (scale != 100) {
263 pugi::xml_node w_node = r_pr.append_child("w:w");
264 w_node.append_attribute("w:val").set_value(scale);
265 }
266
267 // Spacing
268 if (spacing.type != SpacingType::Normal) {
269 pugi::xml_node spacing_node = r_pr.append_child("w:spacing");
270 spacing_node.append_attribute("w:val").set_value(
271 spacing.type == SpacingType::Expanded ? spacing.value : -spacing.value);
272 }
273
274 // Position
275 if (position.type != PositionType::Normal) {
276 pugi::xml_node pos_node = r_pr.append_child("w:position");
277 const int val = (position.type == PositionType::Raised) ? position.value : -position.value;
278 pos_node.append_attribute("w:val").set_value(val);

346 }

350 props.vert_align = string_to_vert_align(v_align.attribute("w:val").value());
351 }
352
353 // Extract highlight

393 }
394
395 // Outline level
396 if (outline_level != OutlineLevel::BodyText) {
397 const int level = static_cast<int>(outline_level);
398 pugi::xml_node outline = p_pr.append_child("w:outlineLvl");
399 outline.append_attribute("w:val").set_value(level);

401
402 // Indentation
403 if (indent) {
404 pugi::xml_node ind = p_pr.append_child("w:ind");

407 const char* attr =
408 (indent->left.type == Indentation::Type::Character) ? "w:firstLineChars" : "w:left";
409 ind.append_attribute(attr).set_value(indent->left.value);
410 }
411 // Right
412 if (indent->right.value != 0) {
413 ind.append_attribute("w:right").set_value(indent->right.value);
414 }
415 // Special
416 if (indent->special.kind != Indentation::Special::Kind::None) {
417 const char* attr = (indent->special.kind == Indentation::Special::Kind::FirstLine)
418 ? "w:firstLine"
419 : "w:hanging";
420 ind.append_attribute(attr).set_value(indent->special.value);
421 }
422 }
423
424 // Spacing
425 if (spacing) {
426 pugi::xml_node sp = p_pr.append_child("w:spacing");
427 // Before
428 if (spacing->before.type != Spacing::Type::Auto || spacing->before.value != 0) {
429 sp.append_attribute("w:before").set_value(spacing->before.value);
430 }
431 // After
432 if (spacing->after.type != Spacing::Type::Auto || spacing->after.value != 0) {
433 sp.append_attribute("w:after").set_value(spacing->after.value);
434 }
435 // Line spacing
436 if (spacing->line_spacing.type == Spacing::LineSpacingType::AtLeast) {
437 sp.append_attribute("w:lineRule").set_value("atLeast");
438 } else if (spacing->line_spacing.type == Spacing::LineSpacingType::Exactly) {
439 sp.append_attribute("w:lineRule").set_value("exact");
440 }
441 sp.append_attribute("w:line").set_value(spacing->line_spacing.value);
442 }
443
444 // Page break control
445 if (keep_next) {
446 p_pr.append_child("w:keep_next");
447 }
448 if (keep_lines) {
449 p_pr.append_child("w:keep_lines");
450 }
451 if (page_break_before) {
452 p_pr.append_child("w:page_break_before");
453 }
454 if (page_break_after) {
455 pugi::xml_node pb = p_pr.append_child("w:page_break_after");
456 pb.append_attribute("w:val").set_value("true");
457 }
458
459 // Borders
460 if (borders) {
461 pugi::xml_node p_bdr = p_pr.append_child("w:pBdr");
462 auto add_border = [&p_bdr](const char* name, const std::optional<Border>& border) {
463 if (border) {
464 pugi::xml_node b = p_bdr.append_child(name);
465 b.append_attribute("w:val").set_value(border_style_to_string(border->style));

475 }

477 // Tab stops
478 if (tab_stops) {
479 tab_stops->apply_to(para_node);
480 }

667TabStopCollection TabStopCollection::extract_from(pugi::xml_node para_node) {
668 TabStopCollection collection;
669 if (!para_node) {
670 return collection;
671 }
672
673 const pugi::xml_node p_pr = para_node.child("w:pPr");
674 if (!p_pr) {
675 return collection;
676 }
677
678 const pugi::xml_node tabs = p_pr.child("w:tabs");
679 if (!tabs) {
680 return collection;
681 }
682
683 for (pugi::xml_node tab = tabs.child("w:tab"); tab; tab = tab.next_sibling("w:tab")) {
684 TabStop ts;
685 ts.alignment = string_to_tab_alignment(tab.attribute("w:val").value());
686 ts.position = ConvertUtil::twips_to_point(tab.attribute("w:pos").as_int());
687
688 const pugi::xml_attribute leader_attr = tab.attribute("w:leader");
689 if (leader_attr) {
690 ts.leader = string_to_tab_leader(leader_attr.value());
691 }
692
693 collection.add(ts);
694 }
695
696 return collection;

733 const char* typestr = "auto";
734 if (width.type == WidthType::Percent) {
735 typestr = "pct";
736 } else if (width.type == WidthType::Absolute) {
737 typestr = "dxa";
738 }
739 ensure_attr(tbl_w, "w:type").set_value(typestr);
740 ensure_attr(tbl_w, "w:w").set_value(width.value);

742 // Table alignment
743 if (alignment != ParagraphProperties::Alignment::Left) {
744 pugi::xml_node jc = tbl_pr.child("w:jc");
745 if (!jc) {
746 jc = tbl_pr.append_child("w:jc");
747 }
748 ensure_attr(jc, "w:val").set_value(pp_alignment_to_string(alignment));
749 } else {
750 tbl_pr.remove_child("w:jc");
751 }
752
753 // Table borders
754 if (borders.top || borders.left || borders.bottom || borders.right) {
755 pugi::xml_node tbl_borders = tbl_pr.child("w:tblBorders");
756 if (!tbl_borders) {
757 tbl_borders = tbl_pr.append_child("w:tblBorders");
758 }
759
760 auto apply_border = [&tbl_borders, &ensure_attr](
761 const char* name,
762 const std::optional<ParagraphProperties::Border>& border) {
763 if (!border) {
764 tbl_borders.remove_child(name);
765 return;
766 }
767 pugi::xml_node b = tbl_borders.child(name);
768 if (!b) {
769 b = tbl_borders.append_child(name);
770 }
771 ensure_attr(b, "w:val").set_value(border_style_to_string(border->style));
772 ensure_attr(b, "w:sz").set_value(border->size);
773 ensure_attr(b, "w:color").set_value(border->color.c_str());
774 ensure_attr(b, "w:space").set_value(border->space);
775 };
776
777 apply_border("w:top", borders.top);
778 apply_border("w:left", borders.left);
779 apply_border("w:bottom", borders.bottom);
780 apply_border("w:right", borders.right);
781 } else {
782 tbl_pr.remove_child("w:tblBorders");
783 }
784
785 // Cell margins
786 if (cell_margin.top > 0 || cell_margin.left > 0 || cell_margin.bottom > 0 ||
787 cell_margin.right > 0) {
788 pugi::xml_node tbl_cell_mar = tbl_pr.child("w:tblCellMar");
789 if (!tbl_cell_mar) {
790 tbl_cell_mar = tbl_pr.append_child("w:tblCellMar");
791 }
792
793 auto apply_margin = [&tbl_cell_mar, &ensure_attr](const char* name, int value) {
794 if (value <= 0) {
795 tbl_cell_mar.remove_child(name);
796 return;
797 }
798 pugi::xml_node m = tbl_cell_mar.child(name);
799 if (!m) {
800 m = tbl_cell_mar.append_child(name);
801 }
802 ensure_attr(m, "w:w").set_value(value);
803 ensure_attr(m, "w:type").set_value("dxa");
804 };
805
806 apply_margin("w:top", cell_margin.top);
807 apply_margin("w:left", cell_margin.left);
808 apply_margin("w:bottom", cell_margin.bottom);
809 apply_margin("w:right", cell_margin.right);
810 } else {
811 tbl_pr.remove_child("w:tblCellMar");
812 }
