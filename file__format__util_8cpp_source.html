88static std::string read_zip_entry(zip_t* za, const std::string& entry_name) {
89 if (zip_entry_open(za, entry_name.c_str()) != 0) {
90 return "";
91 }
92 void* buf = nullptr;
93 size_t bufsize = 0;
94 const ssize_t read = zip_entry_read(za, &buf, &bufsize);
95 zip_entry_close(za);
96 if (read <= 0 || !buf) {
97 return "";
98 }
99 std::string result(static_cast<const char*>(buf), static_cast<size_t>(read));
100 free(buf); // NOLINT: zip_entry_read allocates with malloc
101 return result;

104static std::shared_ptr<FileFormatInfo> detect_from_zip(zip_t* za) {
105 auto info = std::make_shared<FileFormatInfo>();
106
107 const std::string content_types = read_zip_entry(za, "[Content_Types].xml");
108 if (content_types.empty()) {
109 info->set_load_format(LoadFormat::Unknown);
110 return info;
111 }
112
113 const std::string ct_lower = to_lower(content_types);
114
115 // Detect macros
116 if (contains(ct_lower, "macros") || contains(ct_lower, "vnd.ms-office.vba")) {
117 info->set_has_macros(true);
118 }
119
120 // Detect digital signatures
121 if (contains(ct_lower, "digital-signature") || contains(ct_lower, "office.digitalsignature")) {
122 info->set_has_digital_signature(true);
123 }
124
125 // Detect encryption
126 if (contains(ct_lower, "encrypted-package") || contains(ct_lower, "encryption")) {
127 info->set_is_encrypted(true);
128 }
129
130 // Determine exact Word format via lookup table
131 struct ContentTypeMapping {
132 const char* pattern;
133 LoadFormat format;
134 };
135 static const ContentTypeMapping kContentTypeMappings[] = {
136 {"wordprocessingml.document.macroenabled.main+xml", LoadFormat::Docm},
137 {"wordprocessingml.template.macroenabled.main+xml", LoadFormat::Dotm},
138 {"wordprocessingml.template.main+xml", LoadFormat::Dotx},
139 {"wordprocessingml.document.main+xml", LoadFormat::Docx},
140 };
141
142 bool found = false;
143 for (const auto& m : kContentTypeMappings) {
144 if (contains(ct_lower, m.pattern)) {
145 info->set_load_format(m.format);
146 found = true;
147 break;
148 }
149 }
150
151 if (!found) {
152 if (contains(ct_lower, "spreadsheetml") || contains(ct_lower, "presentationml")) {
153 info->set_load_format(LoadFormat::Unknown);
154 } else {
155 // Could be ODT - check mimetype
156 const std::string mimetype = read_zip_entry(za, "mimetype");
157 const std::string mime_lower = to_lower(mimetype);
158 if (contains(mime_lower, "application/vnd.oasis.opendocument.text")) {
159 info->set_load_format(LoadFormat::Odt);
160 } else {
161 info->set_load_format(LoadFormat::Unknown);
162 }
163 }
164 }
165
166 return info;
167}

181std::shared_ptr<FileFormatInfo> FileFormatUtil::detect_file_format(std::istream& stream) {
182 auto info = std::make_shared<FileFormatInfo>();
183
184 // Remember position
185 auto start_pos = stream.tellg();
186
187 // Read first 8KB for sniffing
188 std::vector<uint8_t> buffer(8192);
189 stream.read(reinterpret_cast<char*>(buffer.data()), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
190 static_cast<std::streamsize>(buffer.size()));
191 const std::streamsize bytes_read = stream.gcount();
192 buffer.resize(static_cast<size_t>(bytes_read));
193
194 // Restore position
195 stream.seekg(start_pos);
196
197 if (buffer.empty()) {
198 return info;
199 }
200
201 // ZIP-based formats (DOCX, DOTX, DOCM, DOTM, ODT)
202 if (starts_with(buffer, "PK\x03\x04", 4) || starts_with(buffer, "PK\x05\x06", 4)) {
203 zip_t* za =
204 zip_stream_open(reinterpret_cast<const char*>(buffer.data()), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
205 buffer.size(), 0, 'r');
206 if (za) {
207 info = detect_from_zip(za);
208 zip_stream_close(za);
209 }
210 return info;
211 }
212
213 // RTF
214 if (starts_with_string(buffer, "{\\rtf")) {
215 info->set_load_format(LoadFormat::Rtf);
216 return info;
217 }
218
219 // HTML
220 {
221 auto count = std::min(static_cast<std::ptrdiff_t>(buffer.size()),
222 static_cast<std::ptrdiff_t>(256));
223 const std::string head(buffer.begin(), buffer.begin() + count);
224 std::string lower = to_lower(head);
225 if (starts_with_string(buffer, "<!doctype html") || contains(lower, "<html") ||
226 contains(lower, "<head")) {
227 info->set_load_format(LoadFormat::Html);
228 // Try to detect encoding from meta tag
229 const size_t charset_pos = lower.find("charset=");
230 if (charset_pos != std::string::npos) {
231 const size_t start = charset_pos + 8;
232 if (start < lower.size() && (lower[start] == '"' || lower[start] == '\'')) {
233 const char quote = lower[start];
234 const size_t end = lower.find(quote, start + 1);
235 if (end != std::string::npos) {
236 info->set_encoding(head.substr(start + 1, end - start - 1));
237 }
238 } else {
239 const size_t end = lower.find_first_of(" ;\">\r\n", start);
240 if (end != std::string::npos) {
241 info->set_encoding(head.substr(start, end - start));
242 }
243 }
244 }
245 return info;
246 }
247 }
248
249 // XML
250 if (starts_with_string(buffer, "<?xml")) {
251 info->set_load_format(LoadFormat::Xml);
252 return info;
253 }
254
255 // Markdown - look for common markers in first few lines
256 {
257 auto count = std::min(static_cast<std::ptrdiff_t>(buffer.size()),
258 static_cast<std::ptrdiff_t>(512));
259 const std::string head(buffer.begin(), buffer.begin() + count);
260 const std::string lower = to_lower(head);
261 if (contains(lower, "# ") || contains(lower, "## ") || contains(lower, "**") ||
262 contains(lower, "__") || contains(lower, "-") || contains(lower, "|")) {
263 info->set_load_format(LoadFormat::Markdown);
264 return info;
265 }
266 }
267
268 // Plain text - if no null bytes and mostly printable
269 bool has_null = false;
270 size_t printable = 0;
271 for (const uint8_t b : buffer) {
272 if (b == 0) {
273 has_null = true;
274 }
275 if (b >= 32 && b <= 126) {
276 printable++;
277 }
278 if (b == '\n' || b == '\r' || b == '\t') {
279 printable++;
280 }
281 }
282 if (!has_null && static_cast<double>(printable) > static_cast<double>(buffer.size()) * 0.8) {
283 info->set_load_format(LoadFormat::Text);
284 return info;
285 }
286
287 return info;
288}

303SaveFormat FileFormatUtil::extension_to_save_format(const std::string& extension) {
304 std::string ext = to_lower(extension);
305 if (!ext.empty() && ext[0] != '.') {
306 ext = "." + ext;
307 }
308 // Handle alias extensions via lookup table
309 struct AliasMapping {
310 const char* alias;
311 const char* canonical;
312 };
313 static const AliasMapping kAliasMappings[] = {
314 {".flatopc", ".fopc"},
315 {".htm", ".html"},
316 {".text", ".txt"},
317 {".markdown", ".md"},
318 {".wordml", ".xml"},
319 {".wml", ".xml"},
320 };
321 for (const auto& m : kAliasMappings) {
322 if (ext == m.alias) {
323 ext = m.canonical;
324 break;
325 }
326 }
327 const std::uint8_t value = extension_to_format_value(ext);
328 return static_cast<SaveFormat>(value);
329}
