98void Comment::set_text(const std::string& text) {
99 // Clear existing paragraphs
100 remove_all_children();
101
102 // Split by newlines and create paragraphs
103 size_t start = 0;
104 while (start <= text.size()) {
105 size_t end = text.find('\n', start);
106 if (end == std::string::npos) {
107 end = text.size();
108 }
109 auto para = std::make_shared<Paragraph>();
110 if (start < end) {
111 para->append_run(text.substr(start, end - start));
112 }
113 append_child(para);
114 start = end + 1;
115 }
116}
