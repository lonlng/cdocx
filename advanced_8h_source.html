80 }

87inline bool contains_ci(const std::string& str, const std::string& substr) {
88 if (substr.empty()) {
89 return true;
90 }
91 if (substr.size() > str.size()) {
92 return false;
93 }
94 auto it = std::search(str.begin(), str.end(), substr.begin(), substr.end(), [](char a, char b) {
95 return std::tolower(static_cast<unsigned char>(a)) ==
96 std::tolower(static_cast<unsigned char>(b));
97 });
98 return it != str.end();
99}
