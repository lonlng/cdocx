57std::shared_ptr<Paragraph> Body::append_paragraph(const std::string& text) {
58 // Remove the placeholder empty paragraph created by create_empty()
59 // when the body contains only that single unmodified placeholder.
60 auto children = get_children();
61 if (children.size() == 1 && children[0]->node_type() == NodeType::Paragraph) {
62 auto placeholder = std::static_pointer_cast<Paragraph>(children[0]);
63 auto para_children = placeholder->get_children();
64 if (para_children.size() == 1 && para_children[0]->node_type() == NodeType::Run &&
65 placeholder->get_text().empty()) {
66 remove_child(placeholder);
67 }
68 }
69
70 auto para = std::make_shared<Paragraph>(get_document());
71 if (!text.empty()) {
72 para->append_run(text);
73 }
74 append_child(para);
75 return para;
76}

123void Body::ensure_minimum() {
124 // Ensure at least one paragraph
125 bool has_content = false;
126 for (const auto& child : children_) {
127 if (child->node_type() == NodeType::Paragraph || child->node_type() == NodeType::Table) {
128 has_content = true;
129 break;
130 }
131 }
132
133 if (!has_content) {
134 append_paragraph();
135 }
136}
