555TextProperties Run::get_properties_xml() const {
556 TextProperties props;
557 if (!current_xml_) {
558 return props;
559 }
560
561 auto r_pr = current_xml_.child("w:rPr");
562 if (!r_pr) {
563 return props;
564 }
565
566 // Font
567 auto r_fonts = r_pr.child("w:rFonts");
568 if (r_fonts) {
569 TextProperties::Font font;
570 font.ascii = r_fonts.attribute("w:ascii").value();
571 font.h_ansi = r_fonts.attribute("w:hAnsi").value();
572 font.cs = r_fonts.attribute("w:cs").value();
573 font.east_asia = r_fonts.attribute("w:eastAsia").value();
574 props.font = font;
575 }
576
577 // Bold / Italic
578 props.font_style.bold = r_pr.child("w:b") != nullptr;
579 props.font_style.italic = r_pr.child("w:i") != nullptr;
580
581 // Font size
582 auto sz = r_pr.child("w:sz");
583 if (sz) {
584 props.font_size = sz.attribute("w:val").as_int();
585 }
586
587 // Color
588 auto color = r_pr.child("w:color");
589 if (color) {
590 props.color = color.attribute("w:val").value();
591 }
592
593 // Underline
594 auto u = r_pr.child("w:u");
595 if (u) {
596 auto& ul = props.underline;
597 ul.style = string_to_underline_style(u.attribute("w:val").value());
598 auto ucolor = u.attribute("w:color");
599 if (ucolor) {
600 ul.color = ucolor.value();
601 }
602 }
603
604 // Strike
605 if (r_pr.child("w:dstrike")) {
606 props.strike = TextProperties::StrikeStyle::Double;
607 } else if (r_pr.child("w:strike")) {
608 props.strike = TextProperties::StrikeStyle::Single;
609 }
610
611 // Highlight
612 auto hl = r_pr.child("w:highlight");
613 if (hl) {
614 props.highlight = string_to_highlight(hl.attribute("w:val").value());
615 }
616
617 // Scale
618 auto w = r_pr.child("w:w");
619 if (w) {
620 props.scale = w.attribute("w:val").as_int(100);
621 }
622
623 // Spacing
624 auto sp = r_pr.child("w:spacing");
625 if (sp) {
626 const int sval = sp.attribute("w:val").as_int();
627 if (sval >= 0) {
628 props.spacing.type = TextProperties::SpacingType::Expanded;
629 props.spacing.value = sval;
630 } else {
631 props.spacing.type = TextProperties::SpacingType::Condensed;
632 props.spacing.value = -sval;
633 }
634 }
635
636 // Position (vertAlign / position)
637 if (auto va = r_pr.child("w:vertAlign")) {
638 props.position.type = string_to_position_type(va.attribute("w:val").value());
639 if (auto pos = r_pr.child("w:position")) {
640 props.position.value = std::abs(pos.attribute("w:val").as_int());
641 }
642 }
643
644 return props;
645}

744 r_pr.remove_child("w:position");

759 set_attr(pos, "w:val", val);
760 } else {

762 }
