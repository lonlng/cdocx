21static int count_occurrences(const std::string& text, const std::string& pattern) {
22 if (pattern.empty() || text.empty() || pattern.size() > text.size()) {
23 return 0;
24 }
25 int count = 0;
26 size_t pos = 0;
27 while ((pos = text.find(pattern, pos)) != std::string::npos) {
28 ++count;
29 pos += pattern.size();
30 }
31 return count;
32}

117 for (auto& para : paragraphs) {
118 if (!para) {

166int DocumentSearch::find_and_process(Document& doc,

169 if (pattern.empty() || !callback) {
170 return 0;
171 }
172

175 for (auto& para : paragraphs) {
176 if (!para) {

178 }
179 const std::string para_text = para->get_text();
180 if (para_text.find(pattern) != std::string::npos) {
181 Range range(&doc, para->get_current_node(), para->get_current_node());

184 break;

187 }
