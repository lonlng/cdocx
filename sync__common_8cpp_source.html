32std::string trim_whitespace(const std::string& str) {
33 size_t start = 0;
34 while (start < str.size() && std::isspace(static_cast<unsigned char>(str[start]))) {
35 ++start;
36 }
37 size_t end = str.size();
38 while (end > start && std::isspace(static_cast<unsigned char>(str[end - 1]))) {
39 --end;
40 }
41 return str.substr(start, end - start);
42}

44bool iequals(const std::string& a, const std::string& b) {
45 if (a.size() != b.size()) {
46 return false;

49 if (std::tolower(static_cast<unsigned char>(a[i])) !=

614std::string time_to_w3cdtf(std::time_t t) {
615 if (t == 0) {
616 return "";
617 }
618 std::tm utc{};
619#ifdef _WIN32
620 if (gmtime_s(&utc, &t) != 0) {
621 return "";
622 }
623#else
624 if (gmtime_r(&t, &utc) == nullptr) {
625 return "";
626 }
627#endif
628 char buf[32];
629 if (std::strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &utc) == 0) {
630 return "";
631 }
632 return buf;
633}

643std::time_t w3cdtf_to_time(const std::string& s) {
644 if (s.empty() || s.size() < 19) {
645 return 0;
646 }
647 auto parse_component = [](const char* p, const char** end) -> int {
648 int value = 0;
649 auto result = std::from_chars(p, p + 4, value);
650 *end = result.ptr;
651 return value;
652 };
653 std::tm utc = {};
654 const char* p = s.c_str();
655 const char* end = nullptr;
656 utc.tm_year = parse_component(p, &end) - 1900;
657 p = end + 1; // skip '-'
658 utc.tm_mon = parse_component(p, &end) - 1;
659 p = end + 1; // skip '-'
660 utc.tm_mday = parse_component(p, &end);
661 p = end + 1; // skip 'T'
662 utc.tm_hour = parse_component(p, &end);
663 p = end + 1; // skip ':'
664 utc.tm_min = parse_component(p, &end);
665 p = end + 1; // skip ':'
666 utc.tm_sec = parse_component(p, &end);
667 return timegm_wrapper(&utc);
668}

882void serialize_ffdata_to_fld_char(pugi::xml_node fld_char, const FormField* field) {
883 if (!field) {
884 return;
885 }
886 auto ff_data = fld_char.append_child("w:ffData");
887 if (!field->get_name().empty()) {
888 ff_data.append_child("w:name").append_attribute("w:val").set_value(
889 field->get_name().c_str());
890 }
891 ff_data.append_child("w:enabled")
892 .append_attribute("w:val")
893 .set_value(field->get_enabled() ? "1" : "0");
894 ff_data.append_child("w:calcOnExit")
895 .append_attribute("w:val")
896 .set_value(field->get_calculate_on_exit() ? "1" : "0");
897 if (!field->get_status_text().empty()) {
898 ff_data.append_child("w:statusText")
899 .append_attribute("w:val")
900 .set_value(field->get_status_text().c_str());
901 }
902 if (!field->get_help_text().empty()) {
903 ff_data.append_child("w:helpText")
904 .append_attribute("w:val")
905 .set_value(field->get_help_text().c_str());
906 }
907 if (!field->get_entry_macro().empty()) {
908 ff_data.append_child("w:entryMacro")
909 .append_attribute("w:val")
910 .set_value(field->get_entry_macro().c_str());
911 }
912 if (!field->get_exit_macro().empty()) {
913 ff_data.append_child("w:exitMacro")
914 .append_attribute("w:val")
915 .set_value(field->get_exit_macro().c_str());
916 }
917
918 switch (field->get_form_field_type()) {
919 case FormFieldType::TextInput: {
920 auto text_input = ff_data.append_child("w:textInput");
921 const char* typeval = text_form_field_type_to_string(field->get_text_input_type());
922 text_input.append_child("w:type").append_attribute("w:val").set_value(typeval);
923 if (!field->get_text_input_default().empty()) {
924 text_input.append_child("w:default")
925 .append_attribute("w:val")
926 .set_value(field->get_text_input_default().c_str());
927 }
928 if (field->get_max_length() > 0) {
929 text_input.append_child("w:maxLength")
930 .append_attribute("w:val")
931 .set_value(field->get_max_length());
932 }
933 if (!field->get_text_input_format().empty()) {
934 text_input.append_child("w:format")
935 .append_attribute("w:val")
936 .set_value(field->get_text_input_format().c_str());
937 }
938 break;
939 }
940 case FormFieldType::CheckBox: {
941 auto check_box = ff_data.append_child("w:checkBox");
942 if (field->get_is_check_box_exact_size() && field->get_check_box_size() > 0) {
943 auto size = check_box.append_child("w:size");
944 size.append_attribute("w:val").set_value(
945 static_cast<int>(field->get_check_box_size() * 2));
946 } else {
947 check_box.append_child("w:sizeAuto");
948 }
949 check_box.append_child("w:default")
950 .append_attribute("w:val")
951 .set_value(field->get_default_value() ? "1" : "0");
952 check_box.append_child("w:checked")
953 .append_attribute("w:val")
954 .set_value(field->get_checked() ? "1" : "0");
955 break;
956 }
957 case FormFieldType::ComboBox: {
958 auto dd_list = ff_data.append_child("w:ddList");
959 for (const auto& item : field->get_drop_down_items()) {
960 dd_list.append_child("w:listEntry").append_attribute("w:val").set_value(
961 item.c_str());
962 }
963 if (field->get_drop_down_selected_index() >= 0) {
964 dd_list.append_child("w:default")
965 .append_attribute("w:val")
966 .set_value(field->get_drop_down_selected_index());
967 }
968 break;
969 }
970 }

1018 auto res_run = parent.append_child("w:r");

1020 text_node.text().set(result.c_str());
1021 }
1022
1023 auto end_run = parent.append_child("w:r");
1024 end_run.append_child("w:fldChar").append_attribute("w:fldCharType").set_value("end");
1025
1026 if (bookmark_id != 0) {
1027 auto bm_end = parent.append_child("w:bookmarkEnd");
1028 bm_end.append_attribute("w:id").set_value(bookmark_id);
1029 }
1030}
