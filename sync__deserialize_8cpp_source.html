93 }

131void Document::sync_sections_from_physical() {
132 auto *doc_xml = get_document_xml();
133 if (!doc_xml) {
134 return;
135 }
136
137 auto body = doc_xml->child("w:document").child("w:body");
138 if (!body) {
139 return;
140 }
141
142 // Clear existing DOM sections
143 remove_all_children();
144 sections_cache_.clear();
145 sections_dirty_ = false;
146
147 auto ranges = collect_section_ranges(body);
148
149 bool is_first = true;
150 for (const auto& range : ranges) {
151 auto section = std::make_shared<Section>(this);
152 section->set_first_section(is_first);
153 is_first = false;
154
155 sections_cache_.push_back(section);
156
157 auto sect_body = std::make_shared<Body>(this);
158
159 // Parse content nodes
160 parse_content_children(this, range.begin, range.end, sect_body.get());
161
162 // Load section properties
163 if (range.end) {
164 section->set_sect_pr_node(range.end);
165 section->load_properties();
166
167 // Parse header/footer references
168 for (auto ref = range.end.child("w:headerReference"); ref;
169 ref = ref.next_sibling("w:headerReference")) {
170 HeaderFooterRef hf_ref;
171 hf_ref.relationship_id = ref.attribute("r:id").value();
172 hf_ref.type = string_to_header_footer_type(ref.attribute("w:type").value());
173 section->add_header_ref(hf_ref);
174 }
175 for (auto ref = range.end.child("w:footerReference"); ref;
176 ref = ref.next_sibling("w:footerReference")) {
177 HeaderFooterRef hf_ref;
178 hf_ref.relationship_id = ref.attribute("r:id").value();
179 hf_ref.type = string_to_header_footer_type(ref.attribute("w:type").value());
180 section->add_footer_ref(hf_ref);
181 }
182
183 // Parse header/footer content from XML into DOM
184 for (auto& header : section->get_all_headers()) {
185 parse_header_footer_content(this, header.get());
186 }
187 for (auto& footer : section->get_all_footers()) {
188 parse_header_footer_content(this, footer.get());
189 }
190 }
191
192 section->set_body(sect_body);
193 append_child(section);
194 }
195
196 // If body is completely empty, create a default section
197 if (ranges.empty()) {
198 auto section = std::make_shared<Section>(this);
199 section->set_first_section(true);
200 auto sect_body = std::make_shared<Body>(this);
201 section->set_body(sect_body);
202 append_child(section);
203 }
204}

240std::shared_ptr<Run> Document::parse_run_from_xml(pugi::xml_node run_node) {
241 if (!run_node) {
242 return nullptr;
243 }
244
245 auto run = std::make_shared<Run>(this);
246
247 // Get text content
248 auto text_node = run_node.child("w:t");
249 if (text_node) {
250 run->set_text(text_node.text().get());
251 }
252
253 // Parse formatting
254 parse_run_format_from_xml(run.get(), run_node);
255
256 // Preserve unknown children (e.g., w:drawing) for round-trip fidelity.
257 // Skip whitespace text nodes introduced by pretty-printed XML.
258 for (auto child = run_node.first_child(); child; child = child.next_sibling()) {
259 if (child.type() == pugi::node_pcdata || child.type() == pugi::node_cdata) {
260 continue;
261 }
262 const char* name = child.name();
263 if (std::strcmp(name, "w:t") != 0 && std::strcmp(name, "w:rPr") != 0) {
264 run->preserve_child(child);
265 }
266 }
267
268 return run;
269}

338// Walks a field sequence starting from the given run (which should be the begin run).
339// Returns the node containing w:fld_char[end], or empty node if not found.
340pugi::xml_node walk_field_sequence(pugi::xml_node start_run,
341 std::string* out_instr_text,
342 std::string* out_resulttext) {
343 bool in_result = false;
344 for (auto node = start_run; node; node = node.next_sibling()) {
345 if (!is_run_node(node.name())) {
346 continue;
347 }
348
349 auto fc = node.child("w:fldChar");
350 if (fc) {
351 const char* fct = fc.attribute("w:fldCharType").value();
352 if (std::strcmp(fct, "separate") == 0) {
353 in_result = true;
354 continue;
355 }
356 if (std::strcmp(fct, "end") == 0) {
357 return node;
358 }
359 if (std::strcmp(fct, "begin") == 0) {
360 continue;
361 }
362 }
363
364 auto instr = node.child("w:instrText");
365 if (instr && !in_result && out_instr_text) {
366 *out_instr_text += instr.text().get();
367 }
368
369 auto text_node = node.child("w:t");
370 if (text_node && in_result && out_resulttext) {
371 *out_resulttext += text_node.text().get();
372 }
373 }
374 return pugi::xml_node{};
375}

377void parse_field_code_and_switches(const std::string& code, Field* field) {
378 std::string trimmed = trim_whitespace(code);
379 size_t switch_pos = std::string::npos;
380 for (size_t i = 0; i + 1 < trimmed.size(); ++i) {
381 if (trimmed[i] == ' ' && trimmed[i + 1] == '\\') {
382 switch_pos = i;
383 break;
384 }
385 }
386 if (switch_pos == std::string::npos) {
387 field->set_field_code(trimmed);
388 return;
389 }
390
391 std::string base = trimmed.substr(0, switch_pos);
392 while (!base.empty() && base.back() == ' ') {
393 base.pop_back();
394 }
395 field->set_field_code(base);
396
397 std::string rest = trimmed.substr(switch_pos + 1);
398 size_t sw_start = 0;
399 for (size_t i = 0; i + 1 < rest.size(); ++i) {
400 if (rest[i] == ' ' && rest[i + 1] == '\\') {
401 field->add_switch(rest.substr(sw_start, i - sw_start));
402 sw_start = i + 1;
403 }
404 }
405 if (sw_start < rest.size()) {
406 std::string last_switch = rest.substr(sw_start);
407 last_switch = trim_whitespace(last_switch);
408 if (!last_switch.empty()) {
409 field->add_switch(last_switch);
410 }
411 }
412}

414static void parse_hyperlink_from_xml(Document* doc,
415 pugi::xml_node hyperlink_node,
416 const std::shared_ptr<Paragraph>& para) {
417 auto hyperlink = std::make_shared<Hyperlink>(doc);
418
419 const char* rel_id = hyperlink_node.attribute("r:id").value();
420 if (rel_id && *rel_id) {
421 const std::string target =
422 doc->get_relationship_target("word/_rels/document.xml.rels", rel_id);
423 if (!target.empty()) {
424 hyperlink->set_address(target);
425 }
426 }
427
428 const char* anchor = hyperlink_node.attribute("w:anchor").value();
429 if (anchor && *anchor) {
430 hyperlink->set_bookmark_name(anchor);
431 }
432
433 const char* tooltip = hyperlink_node.attribute("w:tooltip").value();
434 if (tooltip && *tooltip) {
435 hyperlink->set_tooltip(tooltip);
436 }
437
438 std::string link_text;
439 bool formatting_parsed = false;
440 for (auto run_node = hyperlink_node.child("w:r"); run_node;
441 run_node = run_node.next_sibling("w:r")) {
442 auto text_node = run_node.child("w:t");
443 if (text_node) {
444 link_text += text_node.text().get();
445 }
446 if (!formatting_parsed) {
447 parse_run_format_from_xml(hyperlink.get(), run_node);
448 formatting_parsed = true;
449 }
450 }
451 hyperlink->set_result(link_text);
452 para->append_child(hyperlink);
453}

496std::shared_ptr<Paragraph> Document::parse_paragraph_from_xml(pugi::xml_node para_node) {
497 if (!para_node) {
498 return nullptr;
499 }
500
501 auto para = std::make_shared<Paragraph>(this);
502 para->set_current(para_node);
503
504 // Parse paragraph properties
505 auto p_pr = para_node.child("w:pPr");
506 if (p_pr) {
507 para->preserve_p_pr(p_pr);
508 parse_paragraph_format_children_from_xml(p_pr, para->get_paragraph_format());
509
510 auto p_style = p_pr.child("w:pStyle");
511 if (p_style) {
512 para->get_paragraph_format().style_name = p_style.attribute("w:val").value();
513 }
514
515 auto num_pr = p_pr.child("w:numPr");
516 if (num_pr) {
517 auto ilvl = num_pr.child("w:ilvl");
518 auto num_id = num_pr.child("w:numId");
519 if (num_id) {
520 para->get_list_format().list_id = num_id.attribute("w:val").as_uint();
521 para->get_list_format().level =
522 ilvl ? static_cast<NumberingLevel>(ilvl.attribute("w:val").as_int())
523 : NumberingLevel::Level1;
524 }
525 }
526 }
527
528 // Parse inline content
529 for (auto child = para_node.first_child(); child; child = child.next_sibling()) {
530 const char* name = child.name();
531 if (std::strcmp(name, "w:r") == 0) {
532 auto fld_char = child.child("w:fldChar");
533 if (fld_char &&
534 std::strcmp(fld_char.attribute("w:fldCharType").value(), "begin") == 0) {
535 // Check if this is a form field
536 auto ff_data = fld_char.child("w:ffData");
537 if (ff_data) {
538 auto form_field = parse_form_field_from_xml(this, fld_char, child);
539 if (form_field) {
540 para->append_child(form_field);
541 }
542 } else {
543 auto field = std::make_shared<Field>(this, FieldType::Unknown);
544 std::string field_code;
545 std::string field_result;
546 auto end_node = walk_field_sequence(child, &field_code, &field_result);
547 if (end_node) {
548 child = end_node;
549 }
550
551 parse_field_code_and_switches(field_code, field.get());
552 field->set_result(field_result);
553 para->append_child(field);
554 }
555 } else if (auto run = parse_run_from_xml(child)) {
556 para->append_child(run);
557 }
558 } else if (std::strcmp(name, "w:hyperlink") == 0) {
559 parse_hyperlink_from_xml(this, child, para);
560 } else if (std::strcmp(name, "w:br") == 0) {
561 const char* type = child.attribute("w:type").value();
562 bool found = false;
563 for (const auto& mapping : kBreakTypeMappings) {
564 if (std::strcmp(type, mapping.xml_value) == 0) {
565 para->append_child(mapping.factory());
566 found = true;
567 break;
568 }
569 }
570 if (!found) {
571 para->append_child(SpecialChar::line_break());
572 }
573 } else if (std::strcmp(name, "w:tab") == 0) {
574 para->append_child(SpecialChar::tab());
575 } else if (is_bookmark_start_node(name)) {
576 auto bookmark = std::make_shared<BookmarkStart>(this);
577 bookmark->set_id(child.attribute("w:id").as_int());
578 bookmark->set_name(child.attribute("w:name").value());
579 para->append_child(bookmark);
580 } else if (is_bookmark_end_node(name)) {
581 append_id_node<BookmarkEnd>(para.get(), this, child);
582 } else if (std::strcmp(name, "w:commentRangeStart") == 0) {
583 append_id_node<CommentRangeStart>(para.get(), this, child);
584 } else if (std::strcmp(name, "w:commentRangeEnd") == 0) {
585 append_id_node<CommentRangeEnd>(para.get(), this, child);
586 } else if (std::strcmp(name, "w:commentReference") == 0) {
587 // Skip comment references - they are auto-generated during serialization
588 continue;
589 } else if (std::strcmp(name, "w:footnoteReference") == 0) {
590 append_id_node<FootnoteReference>(para.get(), this, child);
591 } else if (std::strcmp(name, "w:endnoteReference") == 0) {
592 append_id_node<EndnoteReference>(para.get(), this, child);
593 }
594 }
595
596 return para;
597}

599std::shared_ptr<Table> Document::parse_table_from_xml(pugi::xml_node table_node) {
600 if (!table_node) {
601 return nullptr;
602 }
603
604 auto table = std::make_shared<Table>(this);
605
606 // Parse table properties
607 auto tbl_pr = table_node.child("w:tblPr");
608 if (tbl_pr) {
609 table->preserve_tbl_pr(tbl_pr);
610 auto jc = tbl_pr.child("w:jc");
611 if (jc) {
612 table->get_table_format().alignment =
613 string_to_table_alignment(jc.attribute("w:val").value());
614 }
615 auto tbl_ind = tbl_pr.child("w:tblInd");
616 if (tbl_ind) {
617 table->get_table_format().left_indent = ConvertUtil::twips_to_point(tbl_ind.attribute("w:w").as_int());
618 }
619 auto tbl_style = tbl_pr.child("w:tblStyle");
620 if (tbl_style) {
621 table->set_style(tbl_style.attribute("w:val").value());
622 }
623 parse_shading_from_xml(tbl_pr.child("w:shd"), table->get_table_format().shading);
624 parse_borders_from_xml(tbl_pr.child("w:tblBorders"), table->get_table_format().borders);
625 auto tbl_layout = tbl_pr.child("w:tblLayout");
626 if (tbl_layout) {
627 const char* layout_type = tbl_layout.attribute("w:type").value();
628 if (std::strcmp(layout_type, "fixed") == 0) {
629 table->get_table_format().auto_fit_behavior = AutoFitBehavior::FixedColumnWidth;
630 table->get_table_format().allow_auto_fit = false;
631 } else {
632 auto tbl_w = tbl_pr.child("w:tblW");
633 if (tbl_w) {
634 const char* width_type = tbl_w.attribute("w:type").value();
635 if (std::strcmp(width_type, "pct") == 0) {
636 table->get_table_format().auto_fit_behavior =
637 AutoFitBehavior::AutoFitToWindow;
638 } else {
639 table->get_table_format().auto_fit_behavior =
640 AutoFitBehavior::AutoFitToContents;
641 }
642 } else {
643 table->get_table_format().auto_fit_behavior =
644 AutoFitBehavior::AutoFitToContents;
645 }
646 table->get_table_format().allow_auto_fit = true;
647 }
648 }
649 }
650
651 // Preserve table grid for round-trip fidelity
652 auto tbl_grid = table_node.child("w:tblGrid");
653 if (tbl_grid) {
654 table->preserve_tbl_grid(tbl_grid);
655 }
656
657 for (auto tr = table_node.child("w:tr"); tr; tr = tr.next_sibling("w:tr")) {
658 auto row = std::make_shared<Row>(this);
659
660 // Parse row properties
661 auto tr_pr = tr.child("w:trPr");
662 if (tr_pr) {
663 auto tr_height = tr_pr.child("w:trHeight");
664 if (tr_height) {
665 row->get_row_format().height = ConvertUtil::twips_to_point(tr_height.attribute("w:val").as_int());
666 const char* rule = tr_height.attribute("w:hRule").value();
667 row->get_row_format().height_rule_exact = (std::strcmp(rule, "exact") == 0);
668 }
669 struct RowBoolFlagMapping {
670 const char* child_name;
671 bool RowFormat::*flag;
672 bool value;
673 };
674 static const RowBoolFlagMapping kRowBoolFlagMappings[] = {
675 {"w:tblHeader", &RowFormat::heading, true},
676 {"w:cantSplit", &RowFormat::allow_break_across_pages, false},
677 };
678 for (const auto& mapping : kRowBoolFlagMappings) {
679 if (tr_pr.child(mapping.child_name)) {
680 row->get_row_format().*mapping.flag = mapping.value;
681 }
682 }
683 }
684
685 for (auto tc = tr.child("w:tc"); tc; tc = tc.next_sibling("w:tc")) {
686 auto cell = std::make_shared<Cell>(this);
687
688 // Parse cell properties
689 auto tc_pr = tc.child("w:tcPr");
690 if (tc_pr) {
691 auto tc_w = tc_pr.child("w:tcW");
692 if (tc_w) {
693 cell->get_cell_format().width = ConvertUtil::twips_to_point(tc_w.attribute("w:w").as_int());
694 const char* typeval = tc_w.attribute("w:type").value();
695 cell->get_cell_format().preferred_width = (std::strcmp(typeval, "pct") == 0);
696 }
697 auto v_align = tc_pr.child("w:vAlign");
698 if (v_align) {
699 cell->get_cell_format().vertical_alignment =
700 string_to_cell_vertical_alignment(v_align.attribute("w:val").value());
701 }
702 auto grid_span = tc_pr.child("w:gridSpan");
703 if (grid_span) {
704 cell->get_cell_format().horizontal_merge =
705 grid_span.attribute("w:val").as_int(1);
706 cell->get_cell_format().horizontal_merge =
707 std::max(1, cell->get_cell_format().horizontal_merge);
708 }
709 auto v_merge = tc_pr.child("w:vMerge");
710 if (v_merge) {
711 cell->get_cell_format().vertical_merge = true;
712 const char* vmerge_val = v_merge.attribute("w:val").value();
713 cell->get_cell_format().vertical_merge_start =
714 (std::strcmp(vmerge_val, "restart") == 0);
715 }
716 parse_shading_from_xml(tc_pr.child("w:shd"), cell->get_cell_format().shading);
717 parse_borders_from_xml(tc_pr.child("w:tcBorders"), cell->get_cell_format().borders);
718 }
719
720 // Parse cell content
721 parse_content_children(this, tc.first_child(), pugi::xml_node(), cell.get());
722
723 cell->ensure_minimum();
724 row->append_child(cell);
725 }
726
727 table->append_child(row);
728 }
729
730 return table;
731}

733std::shared_ptr<Body> Document::parse_body_from_xml(pugi::xml_node body_node) {
734 if (!body_node) {
735 return nullptr;
736 }
737
738 auto body = std::make_shared<Body>(this);
739
740 // Parse paragraphs and tables (stop at sect_pr)
741 for (auto node = body_node.first_child(); node; node = node.next_sibling()) {
742 const char* name = node.name();
743 if (is_para_node(name)) {
744 if (auto para = parse_paragraph_from_xml(node)) {
745 body->append_child(para);
746 }
747 } else if (is_table_node(name)) {
748 if (auto table = parse_table_from_xml(node)) {
749 body->append_child(table);
750 }
751 } else if (is_sectpr_node(name)) {
752 break;
753 }
754 }
755
756 return body;
757}
