32static SectionProperties::Orientation string_to_orientation(const char* val) {
33 if (!val || !*val) {
34 return SectionProperties::Orientation::Portrait;
35 }
36 for (const auto& mapping : kOrientationMappings) {
37 if (std::strcmp(mapping.xml_value, val) == 0) {
38 return mapping.orientation;
39 }
40 }
41 return SectionProperties::Orientation::Portrait;

44static void remove_ref_from_node(pugi::xml_node sect_pr, HeaderFooterType type, bool is_header) {
45 if (!sect_pr) {
46 return;
47 }
48 const char* node_name = is_header ? "w:header_reference" : "w:footer_reference";
49 const char* typestr = header_footer_type_to_string(type);
50 for (auto child = sect_pr.child(node_name); child; child = child.next_sibling(node_name)) {
51 if (std::strcmp(child.attribute("w:type").value(), typestr) == 0) {
52 sect_pr.remove_child(child);
53 break;
54 }
55 }

87static void remove_header_footer(HeaderFooterType type,
88 std::vector<std::shared_ptr<HeaderFooter>>& collection,
89 std::vector<HeaderFooterRef>& refs,
90 pugi::xml_node sect_pr,
91 bool is_header) {
92 collection.erase(std::remove_if(collection.begin(),
93 collection.end(),
94 [type](const std::shared_ptr<HeaderFooter>& hf) {
95 return hf && hf->get_header_footer_type() == type;
96 }),
97 collection.end());
98 remove_ref_from_node(sect_pr, type, is_header);

102 refs.end());

105static void link_hf_to_previous(Document* doc,
106 const HeaderFooterRef* prev_ref,
107 const HeaderFooterRef* curr_ref,
108 std::vector<HeaderFooterRef>& refs,
109 std::vector<std::shared_ptr<HeaderFooter>>& collection,
110 pugi::xml_node sect_pr,
111 HeaderFooterType type,
112 bool is_header,
113 bool is_link_to_previous) {
114 const char* ref_tag = is_header ? "w:header_reference" : "w:footer_reference";
115 const char* part_prefix = is_header ? "word/header" : "word/footer";
116 const char* root_tag = is_header ? "w:hdr" : "w:ftr";
117 const char* rel_type =
118 is_header
119 ? "http://schemas.openxmlformats.org/officeDocument/2006/relationships/header"
120 : "http://schemas.openxmlformats.org/officeDocument/2006/relationships/footer";
121
122 auto add_ref = [&](const HeaderFooterRef& ref) {
123 auto node = sect_pr.append_child(ref_tag);
124 node.append_attribute("r:id").set_value(ref.relationship_id.c_str());
125 node.append_attribute("w:type").set_value(header_footer_type_to_string(type));
126 refs.push_back(ref);
127 auto hf = std::make_shared<HeaderFooter>(doc, type, is_header);
128 hf->set_part_path(ref.part_path);
129 hf->set_relationship_id(ref.relationship_id);
130 collection.push_back(hf);
131 };
132
133 if (is_link_to_previous) {
134 if (curr_ref) {
135 if (prev_ref && curr_ref->part_path == prev_ref->part_path) {
136 // already linked
137 } else if (prev_ref) {
138 remove_header_footer(type, collection, refs, sect_pr, is_header);
139 add_ref(*prev_ref);
140 } else {
141 remove_header_footer(type, collection, refs, sect_pr, is_header);
142 }
143 } else if (prev_ref) {
144 add_ref(*prev_ref);
145 }
146 } else if (prev_ref) {
147 const bool need_copy = (!curr_ref) || (curr_ref->part_path == prev_ref->part_path);
148 if (need_copy) {
149 const std::string new_part =
150 part_prefix +
151 std::to_string(is_header ? doc->get_next_header_number()
152 : doc->get_next_footer_number()) +
153 ".xml";
154
155 auto* prev_xml = doc->get_xml_part(prev_ref->part_path);
156 if (prev_xml) {
157 auto& new_xml = doc->create_xml_part(new_part);
158 new_xml.reset();
159 for (auto child = prev_xml->first_child(); child;
160 child = child.next_sibling()) {
161 new_xml.append_copy(child);
162 }
163 } else {
164 auto& new_xml = doc->create_xml_part(new_part);
165 auto root = new_xml.append_child(root_tag);
166 root.append_attribute("xmlns:w").set_value(
167 "http://schemas.openxmlformats.org/wordprocessingml/2006/main");
168 root.append_attribute("xmlns:r").set_value(
169 "http://schemas.openxmlformats.org/officeDocument/2006/relationships");
170 }
171
172 const std::string new_rel_id =
173 doc->add_relationship("word/_rels/document.xml.rels", rel_type,
174 new_part.substr(5));
175
176 remove_header_footer(type, collection, refs, sect_pr, is_header);
177
178 auto node = sect_pr.append_child(ref_tag);
179 node.append_attribute("r:id").set_value(new_rel_id.c_str());
180 node.append_attribute("w:type").set_value(header_footer_type_to_string(type));
181
182 HeaderFooterRef ref;
183 ref.type = type;
184 ref.relationship_id = new_rel_id;
185 ref.part_path = new_part;
186 refs.push_back(ref);
187 auto hf = std::make_shared<HeaderFooter>(doc, type, is_header);
188 hf->set_part_path(new_part);
189 hf->set_relationship_id(new_rel_id);
190 collection.push_back(hf);
191 }
192 }
193}

195static std::shared_ptr<HeaderFooter> add_header_footer_impl(
196 Document* doc,
197 HeaderFooterType type,
198 bool is_header,
199 std::vector<std::shared_ptr<HeaderFooter>>& collection,
200 std::vector<HeaderFooterRef>& refs,
201 pugi::xml_node sect_pr) {
202 // Check if already exists
203 if (auto existing = get_header_footer(collection, type)) {
204 return existing;
205 }
206
207 const char* part_prefix = is_header ? "word/header" : "word/footer";
208 const char* root_tag = is_header ? "w:hdr" : "w:ftr";
209 const char* ref_tag = is_header ? "w:header_reference" : "w:footer_reference";
210 const char* rel_type =
211 is_header
212 ? "http://schemas.openxmlformats.org/officeDocument/2006/relationships/header"
213 : "http://schemas.openxmlformats.org/officeDocument/2006/relationships/footer";
214
215 const std::string part_name =
216 part_prefix +
217 std::to_string(is_header ? doc->get_next_header_number() : doc->get_next_footer_number()) +
218 ".xml";
219
220 // Create XML
221 auto& xml_doc = doc->create_xml_part(part_name);
222 auto root = xml_doc.append_child(root_tag);
223 root.append_attribute("xmlns:w").set_value(
224 "http://schemas.openxmlformats.org/wordprocessingml/2006/main");
225 root.append_attribute("xmlns:r").set_value(
226 "http://schemas.openxmlformats.org/officeDocument/2006/relationships");
227
228 // Add paragraph
229 auto p = root.append_child("w:p");
230 auto r = p.append_child("w:r");
231 auto t = r.append_child("w:t");
232 t.text().set("");
233
234 // Add relationship
235 const std::string rel_id = doc->add_relationship(
236 "word/_rels/document.xml.rels", rel_type, part_name.substr(5));
237
238 // Add reference to section properties
239 if (sect_pr) {
240 auto ref_node = sect_pr.append_child(ref_tag);
241 ref_node.append_attribute("r:id").set_value(rel_id.c_str());
242 ref_node.append_attribute("w:type").set_value(header_footer_type_to_string(type));
243 }
244
245 // Create HeaderFooter object
246 auto hf = std::make_shared<HeaderFooter>(doc, type, is_header);
247 hf->set_part_path(part_name);
248 hf->set_relationship_id(rel_id);
249 collection.push_back(hf);
250
251 HeaderFooterRef ref;
252 ref.type = type;
253 ref.relationship_id = rel_id;
254 ref.part_path = part_name;
255 refs.push_back(ref);
256
257 return hf;
258}

379Paragraph* Section::add_paragraph(const std::string& text, FormattingFlag flag) {
380 if (!document_) {
381 return nullptr;
382 }
383
384 // Get the body element to append to
385 pugi::xml_node body;
386 if (body_node_) {
387 body = body_node_;
388 } else if (sect_pr_node_) {
389 body = sect_pr_node_.parent();
390 }
391
392 if (!body) {
393 return nullptr;
394 }
395
396 // Always insert before sect_pr when it exists so content stays inside the section;
397 // otherwise append to the body.
398 pugi::xml_node p;
399 if (sect_pr_node_) {
400 p = body.insert_child_before("w:p", sect_pr_node_);
401 } else {
402 p = body.append_child("w:p");
403 }
404
405 if (!p) {
406 return nullptr;
407 }
408
409 // Create paragraph object
410 paragraphs_.emplace_back(body, p);
411 Paragraph* para = &paragraphs_.back();
412
413 // Add initial text if provided
414 if (!text.empty()) {
415 para->add_run(text, flag);
416 }
417
418 return para;
419}

421Table* Section::add_table(size_t rows, size_t cols) {
422 if (!document_ || rows == 0 || cols == 0) {
423 return nullptr;
424 }
425
426 // Get the body element
427 pugi::xml_node body;
428 if (body_node_) {
429 body = body_node_;
430 } else if (sect_pr_node_) {
431 body = sect_pr_node_.parent();
432 }
433
434 if (!body) {
435 return nullptr;
436 }
437
438 // Always insert before sect_pr when it exists so content stays inside the section;
439 // otherwise append to the body.
440 pugi::xml_node tbl;
441 if (sect_pr_node_) {
442 tbl = body.insert_child_before("w:tbl", sect_pr_node_);
443 } else {
444 tbl = body.append_child("w:tbl");
445 }
446
447 if (!tbl) {
448 return nullptr;
449 }
450
451 // Add table properties
452 auto tbl_pr = tbl.append_child("w:tblPr");
453 auto tbl_w = tbl_pr.append_child("w:tblW");
454 tbl_w.append_attribute("w:w").set_value("0");
455 tbl_w.append_attribute("w:type").set_value("auto");
456
457 // Create rows and cells
458 for (size_t r = 0; r < rows; ++r) {
459 auto tr = tbl.append_child("w:tr");
460 tr.append_child("w:trPr");
461
462 for (size_t c = 0; c < cols; ++c) {
463 auto tc = tr.append_child("w:tc");
464 tc.append_child("w:tcPr");
465
466 // Add a paragraph to each cell
467 auto p = tc.append_child("w:p");
468 auto r_pr = p.append_child("w:r");
469 auto t = r_pr.append_child("w:t");
470 t.text().set("");
471 }
472 }
473
474 // Create table object
475 tables_.emplace_back(body, tbl);
476 return &tables_.back();
477}

640bool Section::is_linked_to_previous(HeaderFooterType type, bool is_header) const {
641 if (!document_ || is_first_section_) {
642 return false;
643 }
644 auto prev = document_->get_previous_section(this);
645 if (!prev) {
646 return false;
647 }
648
649 const HeaderFooterRef* prev_ref = is_header ? find_ref(prev->get_header_refs(), type)
650 : find_ref(prev->get_footer_refs(), type);
651 const HeaderFooterRef* curr_ref =
652 is_header ? find_ref(header_refs_, type) : find_ref(footer_refs_, type);
653
654 if (!prev_ref) {
655 return false;
656 }
657 if (!curr_ref) {
658 // If current section has no explicit ref but previous does,
659 // Word may still treat it as linked (inherited). We treat it as linked.
660 return true;
661 }
662 return curr_ref->part_path == prev_ref->part_path;
663}

689void HeaderFooter::accept(DocumentVisitor* visitor) {
690 if (!visitor) {
691 return;
692 }
693
694 // Determine visitor action based on header/footer type
695 VisitorAction action = VisitorAction::Continue;
696 if (is_header()) {
697 action = visitor->visit_header_start(*this);
698 } else {
699 action = visitor->visit_footer_start(*this);
700 }
701
702 if (action == VisitorAction::Continue) {
703 for (const auto& child : get_children()) {
704 child->accept(visitor);
705 }
706
707 if (is_header()) {
708 visitor->visit_header_end(*this);
709 } else {
710 visitor->visit_footer_end(*this);
711 }
712 }
713}

715std::shared_ptr<Node> HeaderFooter::clone(bool deep) const {
716 auto cloned = std::make_shared<HeaderFooter>(get_document(), type_, is_header_);
717 cloned->set_part_path(part_path_);
718 cloned->set_relationship_id(relationship_id_);
719 if (deep) {
