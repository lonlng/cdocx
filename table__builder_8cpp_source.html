100void TableBuilder::create_table_structure(pugi::xml_node tbl) const {
101
102
103 // Create tbl_pr (table properties)
104 pugi::xml_node tbl_pr = tbl.append_child("w:tblPr");
105
106 // Table width
107 pugi::xml_node tbl_w = tbl_pr.append_child("w:tblW");
108 tbl_w.append_attribute("w:w").set_value(auto_width_ ? 0 : width_);
109 tbl_w.append_attribute("w:type").set_value(auto_width_ ? "auto" : "dxa");
110
111 // Table indentation
112 pugi::xml_node tbl_ind = tbl_pr.append_child("w:tblInd");
113 tbl_ind.append_attribute("w:w").set_value(0);
114 tbl_ind.append_attribute("w:type").set_value("dxa");
115
116 // Borders
117 if (borders_.top || borders_.left || borders_.bottom || borders_.right || borders_.inside_h ||
118 borders_.inside_v) {
119 pugi::xml_node tbl_borders = tbl_pr.append_child("w:tblBorders");
120
121 auto add_border = [&](const char* name, bool enabled) {
122 if (enabled) {
123 pugi::xml_node border = tbl_borders.append_child(name);
124 border.append_attribute("w:val").set_value("single");
125 border.append_attribute("w:color").set_value(borders_.color.c_str());
126 border.append_attribute("w:sz").set_value(borders_.size);
127 border.append_attribute("w:space").set_value(0);
128 }
129 };
130
131 add_border("w:top", borders_.top);
132 add_border("w:left", borders_.left);
133 add_border("w:bottom", borders_.bottom);
134 add_border("w:right", borders_.right);
135 add_border("w:insideH", borders_.inside_h);
136 add_border("w:insideV", borders_.inside_v);
137 }
138
139 // Table layout
140 pugi::xml_node tbl_layout = tbl_pr.append_child("w:tblLayout");
141 tbl_layout.append_attribute("w:type").set_value("autofit");
142
143 // Cell margins
144 pugi::xml_node tbl_cell_mar = tbl_pr.append_child("w:tblCellMar");
145 auto add_margin = [&](const char* name, int value) {
146 pugi::xml_node m = tbl_cell_mar.append_child(name);
147 m.append_attribute("w:w").set_value(value);
148 m.append_attribute("w:type").set_value("dxa");
149 };
150 add_margin("w:top", cell_margin_top_);
151 add_margin("w:left", cell_margin_left_);
152 add_margin("w:bottom", cell_margin_bottom_);
153 add_margin("w:right", cell_margin_right_);
154
155 // Create tbl_grid (column definitions)
156 pugi::xml_node tbl_grid = tbl.append_child("w:tblGrid");
157 const int col_width = auto_width_ ? 2840 : (width_ / cols_);
158 for (int c = 0; c < cols_; c++) {
159 pugi::xml_node grid_col = tbl_grid.append_child("w:gridCol");
160 grid_col.append_attribute("w:w").set_value(col_width);
161 }
162}

164void TableBuilder::create_cell(pugi::xml_node tr, int row, int col) const {
165
166
167 const CellData& cell = cells_[row][col];
168
169 // Skip cells that are part of a merge but not the start
170 if (cell.col_span == 0 || cell.row_span == 0) {
171 return;
172 }
173
174 pugi::xml_node tc = tr.append_child("w:tc");
175
176 // Cell properties
177 pugi::xml_node tc_pr = tc.append_child("w:tcPr");
178
179 // Cell width
180 pugi::xml_node tc_w = tc_pr.append_child("w:tcW");
181 tc_w.append_attribute("w:w").set_value(2840); // Default width
182 tc_w.append_attribute("w:type").set_value("dxa");
183
184 // Horizontal merge (grid_span)
185 if (cell.col_span > 1) {
186 pugi::xml_node grid_span = tc_pr.append_child("w:gridSpan");
187 grid_span.append_attribute("w:val").set_value(cell.col_span);
188 }
189
190 // Vertical merge (v_merge)
191 if (cell.row_span > 1) {
192 pugi::xml_node v_merge = tc_pr.append_child("w:vMerge");
193 v_merge.append_attribute("w:val").set_value("restart");
194 }
195
196 // Vertical alignment
197 if (!cell.v_align.empty()) {
198 pugi::xml_node v_align = tc_pr.append_child("w:vAlign");
199 v_align.append_attribute("w:val").set_value(cell.v_align.c_str());
200 }
201
202 // Create paragraph with text
203 pugi::xml_node p = tc.append_child("w:p");
204 pugi::xml_node p_pr = p.append_child("w:pPr");
205 p_pr.append_child("w:rPr");
206
207 if (!cell.text.empty()) {
208 pugi::xml_node r = p.append_child("w:r");
209
210 // Run properties
211 pugi::xml_node r_pr = r.append_child("w:rPr");
212
213 if (cell.bold) {
214 r_pr.append_child("w:b");
215 }
216 if (cell.italic) {
217 r_pr.append_child("w:i");
218 }
219 if (cell.font_size > 0) {
220 pugi::xml_node sz = r_pr.append_child("w:sz");
221 sz.append_attribute("w:val").set_value(cell.font_size);
222 pugi::xml_node sz_cs = r_pr.append_child("w:szCs");
223 sz_cs.append_attribute("w:val").set_value(cell.font_size);
224 }
225
226 // Text
227 const pugi::xml_node t = r.append_child("w:t");
228 t.text().set(cell.text.c_str());
229 }

232pugi::xml_node TableBuilder::build(pugi::xml_document& doc_xml) const {
233 pugi::xml_node tbl = doc_xml.append_child("w:tbl");
234
235 create_table_structure(tbl);
236
237 // Create rows
238 for (int r = 0; r < rows_; r++) {
239 const pugi::xml_node tr = tbl.append_child("w:tr");
240
241 for (int c = 0; c < cols_; c++) {
242 create_cell(tr, r, c);
243 }
244 }
245
246 return tbl;
247}

249bool TableBuilder::insert_at_bookmark(Document& doc, const std::string& bookmark_name) const {
250 const BookmarkCollection bookmarks = doc.get_bookmarks();
251 auto bm = bookmarks.get(bookmark_name);
252 if (!bm) {
253 return false;
254 }
255
256 auto paras = bm->get_covered_paragraphs();
257 if (paras.empty()) {
258 return false;
259 }
260
261 const pugi::xml_node para = paras[0];
262 pugi::xml_node parent = para.parent();
263
264 // Build table XML
265 pugi::xml_document temp_doc;
266 const pugi::xml_node tbl = build(temp_doc);
267
268 // Import and insert after the paragraph
269 pugi::xml_node imported_tbl = parent.insert_child_after("w:tbl", para);
270 imported_tbl.set_name("w:tbl");
271
272 // Copy all children
273 for (pugi::xml_node child = tbl.first_child(); child; child = child.next_sibling()) {
274 imported_tbl.append_copy(child);
275 }
276
277 // Clear bookmark content but keep bookmark
278 bm->set_text("");
279
280 return true;
281}

299int count_existing_tables(Document* doc) {
300 if (!doc) {
301 return 0;
302 }
303
304 pugi::xml_document* doc_xml = doc->get_document_xml();
305 if (!doc_xml) {
306 return 0;
307 }
308
309 int count = 0;
310 const pugi::xml_node body = doc_xml->child("w:document").child("w:body");
311 if (!body) {
312 return 0;
313 }
314
315 for (pugi::xml_node para = body.child("w:p"); para; para = para.next_sibling("w:p")) {
316 if (is_table_caption(para)) {
317 count++;
318 }
319 }
320
321 return count;
322}
323
324bool is_table_caption(pugi::xml_node para) {
325 if (!para) {
326 return false;
327 }
328
329 // Extract all text from paragraph
330 std::string text;
331 for (pugi::xml_node run = para.child("w:r"); run; run = run.next_sibling("w:r")) {
332 const pugi::xml_node t = run.child("w:t");
333 if (t && t.text()) {
334 text += t.text().get();
335 }
336 }
337
338 // trim whitespace
339 const size_t start = text.find_first_not_of(" \t\n\r");
340 if (start == std::string::npos) {
341 return false;
342 }
343 text = text.substr(start);
344
345 // Check if starts with "表" followed by number
346 // Support both "表 1" and "表1" formats
347 if (text.length() < 2) {
348 return false;
349 }
350
351 // Check for "表" character (UTF-8: 0xE8 0xA1 0xA8)
352 if (text.substr(0, 3) != "\xE8\xA1\xA8") {
353 return false;
354 }
355
356 // Check for number after "表"
357 size_t pos = 3;
358 // Skip optional space
359 if (pos < text.length() && text[pos] == ' ') {
360 pos++;
361 }
362
363 // Check for digit
364 if (pos < text.length() && std::isdigit(static_cast<unsigned char>(text[pos]))) {
365 return true;
366 }
367
368 return false;
369}

371pugi::xml_node insert_table_caption(Document* doc,
372 pugi::xml_node after_node,
373 const std::string& description,
374 int table_number) {
375 if (!doc) {
376 return pugi::xml_node{};
377 }
378
379 // Auto-generate table number if not provided
380 if (table_number <= 0) {
381 table_number = get_next_table_number(doc);
382 }
383
384 // Generate full caption text: "表 X description"
385 std::string full_text = "\xE8\xA1\xA8 " + std::to_string(table_number); // "表 "
386 if (!description.empty()) {
387 full_text += " " + description;
388 }
389
390 pugi::xml_document* doc_xml = doc->get_document_xml();
391 if (!doc_xml) {
392 return pugi::xml_node{};
393 }
394
395 pugi::xml_node body = doc_xml->child("w:document").child("w:body");
396 if (!body) {
397 return pugi::xml_node{};
398 }
399
400 // Create new paragraph after the table node
401 pugi::xml_node caption_para;
402 if (after_node) {
403 caption_para = body.insert_child_after("w:p", after_node);
404 } else {
405 caption_para = body.append_child("w:p");
406 }
407
408 if (!caption_para) {
409 return pugi::xml_node{};
410 }
411
412 // Create paragraph properties with center alignment
413 pugi::xml_node p_pr = caption_para.append_child("w:pPr");
414 if (p_pr) {
415 pugi::xml_node jc = p_pr.append_child("w:jc");
416 if (jc) {
417 jc.append_attribute("w:val").set_value("center");
418 }
419
420 // Add spacing before and after
421 pugi::xml_node spacing = p_pr.append_child("w:spacing");
422 if (spacing) {
423 spacing.append_attribute("w:before").set_value(100); // 5pt
424 spacing.append_attribute("w:after").set_value(100); // 5pt
425 }
426 }
427
428 // Create run with text
429 pugi::xml_node run = caption_para.append_child("w:r");
430 if (!run) {
431 return caption_para;
432 }
433
434 // Add run properties with fonts (same as figure caption)
435 pugi::xml_node r_pr = run.append_child("w:rPr");
436 if (r_pr) {
437 // Font settings: SimSun for Chinese, Times New Roman for Western
438 pugi::xml_node r_fonts = r_pr.append_child("w:rFonts");
439 if (r_fonts) {
440 r_fonts.append_attribute("w:ascii").set_value("Times New Roman");
441 r_fonts.append_attribute("w:eastAsia").set_value("SimSun");
442 r_fonts.append_attribute("w:hAnsi").set_value("Times New Roman");
443 }
444
445 // Font size: 10.5pt (21 half-points)
446 pugi::xml_node sz = r_pr.append_child("w:sz");
447 if (sz) {
448 sz.append_attribute("w:val").set_value(21);
449 }
450 pugi::xml_node sz_cs = r_pr.append_child("w:szCs");
451 if (sz_cs) {
452 sz_cs.append_attribute("w:val").set_value(21);
453 }
454 }
455
456 // Add text
457 const pugi::xml_node t = run.append_child("w:t");
458 if (t) {
459 t.text().set(full_text.c_str());
460 }
461
462 return caption_para;
463}
