51void parse_font_from_xml(pugi::xml_node r_pr, Font& font) {
52 if (!r_pr) {
53 return;
54 }
55
56 // Reset defaults to "not set" so only explicitly present properties are serialized
57 font.size = 0;
58 font.name = "";
59 font.name_far_east = "";
60 font.color = Color::auto_color();

62 struct FontBoolFlagMapping {
63 const char* child_name;
64 bool Font::*flag;
65 };
66 static const FontBoolFlagMapping kFontBoolFlagMappings[] = {
67 {"w:b", &Font::bold},
68 {"w:i", &Font::italic},
69 {"w:strike", &Font::strikethrough},
70 };
71 for (const auto& mapping : kFontBoolFlagMappings) {
72 if (r_pr.child(mapping.child_name)) {
73 font.*mapping.flag = true;
74 }
75 }
76
77 if (auto u = r_pr.child("w:u")) {
78 font.underline = string_to_underline_type(u.attribute("w:val").value());
79 }
80
81 if (auto sz = r_pr.child("w:sz")) {
82 font.size = sz.attribute("w:val").as_int() / 2.0;
83 }
84
85 auto r_fonts = r_pr.child("w:rFonts");
86 if (r_fonts) {
87 font.name = r_fonts.attribute("w:ascii").value();
88 if (font.name.empty()) {
89 font.name = r_fonts.attribute("w:hAnsi").value();
90 }
91 font.name_far_east = r_fonts.attribute("w:eastAsia").value();
92 }
93
94 auto color = r_pr.child("w:color");
95 if (color) {
96 font.color = Color::from_hex(color.attribute("w:val").value());
97 }
98
99 auto v_align = r_pr.child("w:vertAlign");
100 if (v_align) {
101 font.script_type = string_to_script_type(v_align.attribute("w:val").value());
102 }
103
104 parse_shading_from_xml(r_pr.child("w:shd"), font.shading);
105}

114static void serialize_style_to_xml(pugi::xml_node styles_root, const Style& style) {
115 pugi::xml_node style_xml;
116
117 if (style.has_preserved_style_xml()) {
118 // Start from a deep copy of the original style XML for maximum fidelity
119 style_xml = styles_root.append_copy(style.get_preserved_style_xml());
120
121 // Overlay DOM-managed attributes
122 style_xml.attribute("w:type").set_value(style_type_to_string(style.get_type()));
123 style_xml.attribute("w:styleId").set_value(style.get_style_id().c_str());
124
125 if (style.get_is_default()) {
126 if (!style_xml.attribute("w:default")) {
127 style_xml.append_attribute("w:default").set_value("1");
128 }
129 } else {
130 style_xml.remove_attribute("w:default");
131 }
132
133 if (!style.get_is_built_in()) {
134 if (!style_xml.attribute("w:customStyle")) {
135 style_xml.append_attribute("w:customStyle").set_value("1");
136 }
137 } else {
138 style_xml.remove_attribute("w:customStyle");
139 }
140
141 // Update name
142 auto name_node = style_xml.child("w:name");
143 if (!style.get_name().empty()) {
144 if (!name_node) {
145 name_node = style_xml.prepend_child("w:name");
146 }
147 name_node.attribute("w:val").set_value(style.get_name().c_str());
148 } else if (name_node) {
149 style_xml.remove_child(name_node);
150 }
151
152 // Update basedOn
153 auto based_on_node = style_xml.child("w:basedOn");
154 if (!style.get_base_style_name().empty()) {
155 if (!based_on_node) {
156 based_on_node = style_xml.prepend_child("w:basedOn");
157 }
158 based_on_node.attribute("w:val").set_value(style.get_base_style_name().c_str());
159 } else if (based_on_node) {
160 style_xml.remove_child(based_on_node);
161 }
162 } else {
163 // No preserved XML: build from scratch (programmatically created style)
164 style_xml = styles_root.append_child("w:style");
165 style_xml.append_attribute("w:type").set_value(style_type_to_string(style.get_type()));
166 style_xml.append_attribute("w:styleId").set_value(style.get_style_id().c_str());
167
168 if (style.get_is_default()) {
169 style_xml.append_attribute("w:default").set_value("1");
170 }
171 if (!style.get_is_built_in()) {
172 style_xml.append_attribute("w:customStyle").set_value("1");
173 }
174
175 if (!style.get_name().empty()) {
176 auto name = style_xml.append_child("w:name");
177 name.append_attribute("w:val").set_value(style.get_name().c_str());
178 }
179
180 if (!style.get_base_style_name().empty()) {
181 auto based_on = style_xml.append_child("w:basedOn");
182 based_on.append_attribute("w:val").set_value(style.get_base_style_name().c_str());
183 }
184 }
185
186 // Determine if DOM has explicit font/paragraph formatting
187 const Font& font = style.get_font();
188 const bool has_font_formatting = font.bold || font.italic || font.strikethrough ||
189 font.underline != UnderlineType::None || font.size > 0 ||
190 !font.name.empty() || font.color != Color::auto_color() ||
191 font.script_type != ScriptType::Normal;
192
193 const ParagraphFormat& para_format = style.get_paragraph_format();
194 const bool has_para_format =
195 para_format.alignment != ParagraphAlignment::Left || para_format.left_indent != 0 ||
196 para_format.right_indent != 0 || para_format.first_line_indent != 0 ||
197 para_format.space_before != 0 || para_format.space_after != 0 ||
198 para_format.line_spacing != 1.15 || para_format.keep_with_next ||
199 para_format.keep_together || para_format.page_break_before || !para_format.widow_control ||
200 para_format.outline_level != OutlineLevel::BodyText;
201
202 // Merge w:rPr: preserve unknown children, overlay managed ones
203 auto r_pr = style_xml.child("w:rPr");
204 if (has_font_formatting) {
205 if (!r_pr) {
206 r_pr = style_xml.append_child("w:rPr");
207 }
208 // Check if original r_pr had szCs before we remove anything
209 const bool original_had_sz_cs = r_pr.child("w:szCs") != nullptr;
210
211 // Remove managed children that we will re-serialize
212 remove_managed_children(r_pr,
213 {"w:b",
214 "w:i",
215 "w:strike",
216 "w:u",
217 "w:sz",
218 "w:szCs",
219 "w:rFonts",
220 "w:color",
221 "w:vertAlign",
222 "w:spacing",
223 "w:w",
224 "w:shd"});
225 // Serialize current DOM font formatting directly into the existing r_pr
226 serialize_font_to_r_pr(r_pr, font, original_had_sz_cs);
227 }
228
229 // Merge w:pPr: preserve unknown children, overlay managed ones
230 auto p_pr = style_xml.child("w:pPr");
231 if (has_para_format) {
232 if (!p_pr) {
233 p_pr = style_xml.append_child("w:pPr");
234 }
235 // Remove managed children that we will re-serialize
236 remove_managed_children(p_pr, {"w:jc", "w:ind", "w:spacing", "w:shd", "w:dropCap"});
237 serialize_style_paragraph_format_to_xml(p_pr, para_format);
238 }

241static std::shared_ptr<Style> parse_style_from_xml(pugi::xml_node style_node, Document* doc) {
242 if (!style_node) {
243 return nullptr;
244 }
245
246 auto style =
247 std::make_shared<Style>(doc, string_to_style_type(style_node.attribute("w:type").value()));
248 style->set_style_id(style_node.attribute("w:styleId").value());
249 style->set_is_default(style_node.attribute("w:default").as_int() != 0);
250 style->set_is_built_in(style_node.attribute("w:customStyle").as_int() == 0);
251
252 auto name = style_node.child("w:name");
253 if (name) {
254 style->set_name(name.attribute("w:val").value());
255 } else {
256 style->set_name(style->get_style_id());
257 }
258
259 auto based_on = style_node.child("w:basedOn");
260 if (based_on) {
261 style->set_base_style_name(based_on.attribute("w:val").value());
262 }
263
264 auto r_pr = style_node.child("w:rPr");
265 if (r_pr) {
266 Font font;
267 parse_font_from_xml(r_pr, font);
268 style->set_font(font);
269 } else {
270 // No explicit r_pr: set font to empty so defaults aren't serialized
271 Font empty_font;
272 empty_font.size = 0;
273 empty_font.name = "";
274 empty_font.name_far_east = "";
275 empty_font.color = Color::auto_color();
276 style->set_font(empty_font);
277 }
278
279 auto p_pr = style_node.child("w:pPr");
280 if (p_pr) {
281 ParagraphFormat format;
282 parse_style_paragraph_format_from_xml(p_pr, format);
283 style->set_paragraph_format(format);
284 } else {
285 // No explicit pPr: set paragraph format to defaults so they aren't serialized
286 style->set_paragraph_format(ParagraphFormat());
287 }
288
289 // Preserve full original style XML for round-trip fidelity
290 style->preserve_style_xml(style_node);
291
292 return style;
293}

299void Document::sync_styles_to_physical() {
300 pugi::xml_document* styles_doc = get_styles();
301 if (!styles_doc) {
302 styles_doc = &create_xml_part("word/styles.xml");
303 add_content_type_override(
304 "/word/styles.xml",
305 "application/vnd.openxmlformats-officedocument.wordprocessingml.styles+xml");
306 }
307
308 auto root = styles_doc->child("w:styles");
309 if (!root) {
310 styles_doc->reset();
311 root = styles_doc->append_child("w:styles");
312 root.append_attribute("xmlns:w").set_value(
313 "http://schemas.openxmlformats.org/wordprocessingml/2006/main");
314 }
315
316 // Remove only w:style nodes, preserve docDefaults, latentStyles, etc.
317 for (auto child = root.first_child(); child;) {
318 auto next = child.next_sibling();
319 if (std::strcmp(child.name(), "w:style") == 0) {
320 root.remove_child(child);
321 }
322 child = next;
323 }
324
325 for (const auto& style : styles()) {
326 serialize_style_to_xml(root, *style);
327 }
328
329 mark_modified("word/styles.xml");
330}

332void Document::sync_styles_from_physical() {
333 pugi::xml_document* styles_doc = get_styles();
334 if (!styles_doc) {
335 // Create default Normal style if styles.xml is missing
336 if (styles_) {
337 styles_->clear();
338 auto normal = std::make_shared<Style>(this, StyleType::Paragraph);
339 normal->set_name("Normal");
340 normal->set_style_id("Normal");
341 normal->set_style_identifier(StyleIdentifier::Normal);
342 normal->set_is_built_in(true);
343 normal->set_is_default(true);
344 styles_->add(normal);
345 }
346 return;
347 }
348
349 if (styles_) {
350 styles_->clear();
351 }
352
353 auto styles_root = styles_doc->child("w:styles");
354 if (!styles_root) {
355 return;
356 }
357
358 for (auto style_node = styles_root.child("w:style"); style_node;
359 style_node = style_node.next_sibling("w:style")) {
360 if (auto style = parse_style_from_xml(style_node, this)) {
361 if (styles_) {
362 styles_->add(style);
363 }
364 }
365 }
366}
