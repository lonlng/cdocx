251struct LoadResult {
252 bool success = false;
253 DocumentIntegrity integrity = DocumentIntegrity::Corrupted;
254 std::vector<LoadError> errors;
255 std::vector<std::string> skipped_files;
256 size_t total_files = 0;
257 size_t loaded_files = 0;
258 double load_time_ms = 0.0;
259
260 bool is_usable() const {
261 return success && (integrity == DocumentIntegrity::Complete ||
262 integrity == DocumentIntegrity::Partial);
263 }
264 bool is_complete() const { return success && integrity == DocumentIntegrity::Complete; }
