20void Document::sync_comments_to_physical() {
21 if (comments_cache_.empty()) {
22 return;
23 }
24
25 auto *comments_xml = get_xml_part("word/comments.xml");
26 if (!comments_xml) {
27 comments_xml = &create_xml_part("word/comments.xml");
28 }
29
30 auto root = comments_xml->child("w:comments");
31 if (!root) {
32 root = comments_xml->prepend_child("w:comments");
33 root.append_attribute("xmlns:w").set_value(
34 "http://schemas.openxmlformats.org/wordprocessingml/2006/main");
35 root.append_attribute("xmlns:w14")
36 .set_value("http://schemas.microsoft.com/office/word/2010/wordml");
37 root.append_attribute("xmlns:w15")
38 .set_value("http://schemas.microsoft.com/office/word/2012/wordml");
39 }
40
41 // Remove existing comment nodes
42 for (auto child = root.first_child(); child;) {
43 auto next = child.next_sibling();
44 if (std::strcmp(child.name(), "w:comment") == 0) {
45 root.remove_child(child);
46 }
47 child = next;
48 }
49
50 for (const auto& comment : comments_cache_) {
51 auto cxml = root.append_child("w:comment");
52 cxml.append_attribute("w:id").set_value(comment->get_id());
53 cxml.append_attribute("w:author").set_value(comment->get_author().c_str());
54 cxml.append_attribute("w:initials").set_value(comment->get_initial().c_str());
55
56 auto time_val = std::chrono::system_clock::to_time_t(comment->get_date_time());
57 cxml.append_attribute("w:date").set_value(time_to_w3cdtf(time_val).c_str());
58
59 // Serialize comment paragraphs
60 for (const auto& child : comment->get_children()) {
61 if (child->node_type() == NodeType::Paragraph) {
62 serialize_paragraph_to_xml(cxml, dynamic_cast<Paragraph*>(child.get()));
63 }
64 }
65 }
66
67 mark_modified("word/comments.xml");
68}

70void Document::sync_comments_from_physical() {
71 auto *comments_xml = get_xml_part("word/comments.xml");
72 if (!comments_xml) {
73 return;
74 }
75
76 comments_cache_.clear();
77
78 auto root = comments_xml->child("w:comments");
79 if (!root) {
80 return;
81 }
82
83 for (auto cxml = root.child("w:comment"); cxml; cxml = cxml.next_sibling("w:comment")) {
84 auto comment = std::make_shared<Comment>(this);
85 comment->set_id(cxml.attribute("w:id").as_int());
86 comment->set_author(cxml.attribute("w:author").value());
87 comment->set_initial(cxml.attribute("w:initials").value());
88
89 const char* date_str = cxml.attribute("w:date").value();
90 if (date_str && *date_str) {
91 auto time = w3cdtf_to_time(date_str);
92 comment->set_date_time(std::chrono::system_clock::from_time_t(time));
93 }
94
95 for (auto para_node = cxml.child("w:p"); para_node;
96 para_node = para_node.next_sibling("w:p")) {
97 auto para = parse_paragraph_from_xml(para_node);
98 if (para) {
99 comment->append_child(para);
100 }
101 }
102
103 comments_cache_.push_back(comment);
104 }
105
106 for (const auto& comment : comments_cache_) {
107 if (comment->get_id() >= next_comment_id_) {
108 next_comment_id_ = comment->get_id() + 1;
109 }
110 }
111}
