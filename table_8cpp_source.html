558void Table::insert_column(int index) {
559 index = std::max(0, index);
560
561 for (const auto& row : get_rows()) {
562 int current_col = 0;
563 int cell_index = 0;
564 bool inserted = false;
565 for (const auto& cell : row->get_cells()) {
566 const int span = cell->get_horizontal_merge_span();
567 if (current_col <= index && index < current_col + span) {
568 // Insert at this position
569 auto new_cell = std::make_shared<Cell>(get_document());
570 new_cell->ensure_minimum();
571 row->insert_child(cell_index, new_cell);
572 inserted = true;
573 break;
574 }
575 current_col += span;
576 ++cell_index;
577 }
578 if (!inserted) {
579 // Append at the end
580 auto new_cell = std::make_shared<Cell>(get_document());
581 new_cell->ensure_minimum();
582 row->append_child(new_cell);

617std::shared_ptr<Cell> Table::merge_cells(const std::shared_ptr<Cell>& start_cell,
618 const std::shared_ptr<Cell>& end_cell) {
619 if (!start_cell || !end_cell || start_cell.get() == end_cell.get()) {
620 return start_cell;
621 }
622
623 auto start_row = start_cell->get_parent_row();
624 auto end_row = end_cell->get_parent_row();
625 if (!start_row || !end_row) {
626 return nullptr;
627 }
628
629 auto table = start_row->get_parent_table();
630 if (!table || table.get() != this) {
631 return nullptr;
632 }
633 if (start_row->get_parent_table().get() != this) {
634 return nullptr;
635 }
636
637 int start_row_idx = start_row->get_row_index();
638 int end_row_idx = end_row->get_row_index();
639 int start_col_idx = start_cell->get_column_index();
640 int end_col_idx = end_cell->get_column_index();
641
642 if (start_row_idx < 0 || end_row_idx < 0 || start_col_idx < 0 || end_col_idx < 0) {
643 return nullptr;
644 }
645 if (start_row_idx > end_row_idx) {
646 std::swap(start_row_idx, end_row_idx);
647 }
648 if (start_col_idx > end_col_idx) {
649 std::swap(start_col_idx, end_col_idx);
650 }
651
652 return merge_cells(start_row_idx, start_col_idx, end_row_idx, end_col_idx);
653}

655std::shared_ptr<Cell> Table::merge_cells(int start_row,
656 int start_col,
657 int end_row,
658 int end_col) const {
659 if (start_row > end_row) {
660 std::swap(start_row, end_row);
661 }
662 if (start_col > end_col) {
663 std::swap(start_col, end_col);
664 }
665
666 const int row_count = get_row_count();
667 if (start_row < 0 || end_row >= row_count) {
668 return nullptr;
669 }
670
671 // Validate that all rows have enough columns (accounting for existing merges)
672 for (int r = start_row; r <= end_row; ++r) {
673 auto row = get_row(r);
674 if (!row) {
675 return nullptr;
676 }
677 int col_count = 0;
678 for (const auto& cell : row->get_cells()) {
679 col_count += cell->get_horizontal_merge_span();
680 }
681 if (end_col >= col_count) {
682 return nullptr;
683 }
684 }
685
686 const int col_span = end_col - start_col + 1;
687 const int row_span = end_row - start_row + 1;
688
689 for (int r = start_row; r <= end_row; ++r) {
690 auto row = get_row(r);
691 auto cells = row->get_cells();
692
693 // Find the cell at start_col and the cells to merge/remove
694 int current_col = 0;
695 std::shared_ptr<Cell> first_cell_in_row;
696 std::vector<std::shared_ptr<Cell>> cells_to_remove;
697 for (const auto& cell : cells) {
698 const int span = cell->get_horizontal_merge_span();
699 if (current_col <= start_col && start_col < current_col + span) {
700 first_cell_in_row = cell;
701 // Collect cells within the horizontal merge range
702 int inner_col = current_col + span;
703 for (size_t i = static_cast<size_t>(cell->get_column_index()) + 1;
704 i < cells.get_count();
705 ++i) {
706 auto next_cell = cells[static_cast<int>(i)];
707 if (inner_col > end_col) {
708 break;
709 }
710 cells_to_remove.push_back(next_cell);
711 inner_col += next_cell->get_horizontal_merge_span();
712 }
713 break;
714 }
715 current_col += span;
716 }
717
718 if (!first_cell_in_row) {
719 continue;
720 }
721
722 // Move content from cells to be removed into the first cell
723 for (const auto& removed_cell : cells_to_remove) {
724 for (const auto& child : removed_cell->get_children()) {
725 first_cell_in_row->append_child(child->clone(true));
726 }
727 row->remove_child(removed_cell);
728 }
729
730 // Set merge properties
731 first_cell_in_row->get_cell_format().horizontal_merge = col_span;
732 if (row_span > 1) {
733 first_cell_in_row->get_cell_format().vertical_merge = true;
734 first_cell_in_row->get_cell_format().vertical_merge_start = (r == start_row);
735 } else {
736 first_cell_in_row->get_cell_format().vertical_merge = false;
737 first_cell_in_row->get_cell_format().vertical_merge_start = false;
738 }
739 }
740
741 auto result = get_cell(start_row, start_col);
742 return result;
743}

745void Table::split_cell(const std::shared_ptr<Cell>& cell, int row_count, int col_count) {
746 if (!cell || row_count < 1 || col_count < 1) {
747 return;
748 }
749 if (row_count == 1 && col_count == 1) {
750 return;
751 }
752
753 auto row = cell->get_parent_row();
754 if (!row) {
755 return;
756 }
757 auto table = row->get_parent_table();
758 if (!table || table.get() != this) {
759 return;
760 }
761
762 const int row_idx = row->get_row_index();
763 const int col_idx = cell->get_column_index();
764 if (row_idx < 0 || col_idx < 0) {
765 return;
766 }
767
768 const int original_col_span = cell->get_horizontal_merge_span();
769 int original_row_span = 1;
770 if (cell->get_cell_format().vertical_merge) {
771 // Count how many rows this cell spans
772 if (cell->get_cell_format().vertical_merge_start) {
773 for (int r = row_idx + 1; r < table->get_row_count(); ++r) {
774 auto next_row = table->get_row(r);
775 auto next_cell = next_row->get_cell(col_idx);
776 if (!next_cell || !next_cell->get_cell_format().vertical_merge ||
777 next_cell->get_cell_format().vertical_merge_start) {
778 break;
779 }
780 original_row_span++;
781 }
782 } else {
783 // Cell is a continuation; find the start and delegate
784 for (int r = row_idx - 1; r >= 0; --r) {
785 auto prev_row = table->get_row(r);
786 auto prev_cell = prev_row->get_cell(col_idx);
787 if (prev_cell && prev_cell->get_cell_format().vertical_merge_start) {
788 split_cell(prev_cell, row_count, col_count);
789 return;
790 }
791 }
792 return;
793 }
794 }
795
796 if (row_count > original_row_span || col_count > original_col_span) {
797 // Cannot split into more parts than the original span
798 return;
799 }
800
801 // For simplicity, we only support splitting back into individual cells
802 // Reset the top-left cell
803 cell->get_cell_format().horizontal_merge = 1;
804 cell->get_cell_format().vertical_merge = false;
805 cell->get_cell_format().vertical_merge_start = false;
806
807 // Insert missing cells in the first row
808 for (int c = 1; c < col_count; ++c) {
809 auto new_cell = std::make_shared<Cell>(get_document());
810 new_cell->ensure_minimum();
811 int insert_pos = col_idx + c;
812 // Clamp insert position
813 insert_pos = std::min(insert_pos, static_cast<int>(row->get_children().size()));
814 row->insert_child(insert_pos, new_cell);
815 }
816
817 // For each continuation row, insert cells
818 for (int r = 1; r < row_count; ++r) {
819 auto current_row = get_row(row_idx + r);
820 if (!current_row) {
821 continue;
822 }
823
824 // Remove the continuation cell if it exists
825 auto cont_cell = current_row->get_cell(col_idx);
826 if (cont_cell) {
827 current_row->remove_child(cont_cell);
828 }
829
830 // Insert new cells
831 for (int c = 0; c < col_count; ++c) {
832 auto new_cell = std::make_shared<Cell>(get_document());
833 new_cell->ensure_minimum();
834 int insert_pos = col_idx + c;
835 insert_pos = std::min(insert_pos, static_cast<int>(current_row->get_children().size()));
836 current_row->insert_child(insert_pos, new_cell);
837 }
838 }
839
840 // Remove any extra continuation rows that were part of the original merge
841 // but are not covered by the new row_count
842 for (int r = row_count; r < original_row_span; ++r) {
843 auto current_row = get_row(row_idx + r);
844 if (!current_row) {
845 continue;
846 }
847 auto cont_cell = current_row->get_cell(col_idx);
848 if (cont_cell && cont_cell->get_cell_format().vertical_merge &&
849 !cont_cell->get_cell_format().vertical_merge_start) {
850 current_row->remove_child(cont_cell);
851 }
852 }
853}
