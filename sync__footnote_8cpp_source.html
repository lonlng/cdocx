20static void sync_notes_to_physical_impl(Document* doc,
21 const char* xml_part,
22 const char* root_name,
23 const char* child_name,
24 const std::vector<std::shared_ptr<Footnote>>& cache) {
25 auto *notes_xml = doc->get_xml_part(xml_part);
26 if (!notes_xml) {
27 if (cache.empty()) {
28 return;
29 }
30 notes_xml = &doc->create_xml_part(xml_part);
31 }
32
33 auto root = notes_xml->child(root_name);
34 if (!root) {
35 root = notes_xml->prepend_child(root_name);
36 root.append_attribute("xmlns:w").set_value(
37 "http://schemas.openxmlformats.org/wordprocessingml/2006/main");
38 root.append_attribute("xmlns:r").set_value(
39 "http://schemas.openxmlformats.org/officeDocument/2006/relationships");
40 }
41
42 // Collect separator nodes to preserve them
43 std::vector<pugi::xml_node> separators;
44 for (auto child = root.first_child(); child;) {
45 auto next = child.next_sibling();
46 const int note_id = child.attribute("w:id").as_int();
47 if (note_id < 0 || note_id == 0) {
48 separators.push_back(child);
49 }
50 root.remove_child(child);
51 child = next;
52 }
53
54 // Re-add separators first, or create them if missing
55 if (separators.empty()) {
56 auto sep = root.prepend_child(child_name);
57 sep.append_attribute("w:id").set_value(-1);
58 sep.append_attribute("w:type").set_value("separator");
59 auto sep_para = sep.append_child("w:p");
60 auto sep_run = sep_para.append_child("w:r");
61 sep_run.append_child("w:separator");
62
63 auto cont_sep = root.prepend_child(child_name);
64 cont_sep.append_attribute("w:id").set_value(0);
65 cont_sep.append_attribute("w:type").set_value("continuationSeparator");
66 auto cont_para = cont_sep.append_child("w:p");
67 auto cont_run = cont_para.append_child("w:r");
68 cont_run.append_child("w:continuationSeparator");
69 } else {
70 for (auto& sep : separators) {
71 root.append_copy(sep);
72 }
73 }
74
75 for (const auto& note : cache) {
76 auto node = root.append_child(child_name);
77 node.append_attribute("w:id").set_value(note->get_id());
78 if (!note->is_auto() && !note->get_reference_mark().empty()) {
79 node.append_attribute("w:type").set_value("normal");
80 }
81 bool first_para = true;
82 for (const auto& child : note->get_children()) {
83 if (child->node_type() == NodeType::Paragraph) {
84 auto para = dynamic_cast<Paragraph*>(child.get());
85 std::shared_ptr<Run> temp_ref_run;
86 if (first_para && !note->is_auto() &&
87 !note->get_reference_mark().empty()) {
88 temp_ref_run = std::make_shared<Run>();
89 temp_ref_run->set_text(note->get_reference_mark());
90 para->insert_child(0, temp_ref_run);
91 }
92 serialize_paragraph_to_xml(node, para);
93 first_para = false;
94 if (temp_ref_run) {
95 para->remove_child(temp_ref_run);
96 }
97 }
98 }
99 }
100
101 doc->mark_modified(xml_part);
102}

104static void sync_notes_from_physical_impl(Document* doc,
105 const char* xml_part,
106 const char* root_name,
107 const char* child_name,
108 std::vector<std::shared_ptr<Footnote>>& cache,
109 int& next_id,
110 FootnoteType type) {
111 auto *notes_xml = doc->get_xml_part(xml_part);
112 if (!notes_xml) {
113 return;
114 }
115
116 cache.clear();
117
118 auto root = notes_xml->child(root_name);
119 if (!root) {
120 return;
121 }
122
123 for (auto node = root.child(child_name); node; node = node.next_sibling(child_name)) {
124 const int id = node.attribute("w:id").as_int();
125 if (id < 0 || id == 0) {
126 continue;
127 }
128
129 auto note = std::make_shared<Footnote>(doc, type);
130 note->set_id(id);
131
132 std::string reference_mark;
133 const char* typeattr = node.attribute("w:type").value();
134 if (typeattr && std::strcmp(typeattr, "normal") == 0) {
135 note->set_auto(false);
136
137 // Extract reference mark from the first run of the first paragraph
138 auto first_para = node.child("w:p");
139 if (first_para) {
140 auto first_run = first_para.child("w:r");
141 if (first_run) {
142 for (auto t = first_run.child("w:t"); t;
143 t = t.next_sibling("w:t")) {
144 reference_mark += t.text().get();
145 }
146 }
147 }
148 }
149
150 for (auto para_node = node.child("w:p"); para_node;
151 para_node = para_node.next_sibling("w:p")) {
152 if (auto para = doc->parse_paragraph_from_xml(para_node)) {
153 note->append_child(para);
154 }
155 }
156
157 if (!reference_mark.empty()) {
158 note->set_reference_mark(reference_mark);
159 // Remove the reference-mark run from the first paragraph's DOM
160 // so get_text() returns only the body text.
161 if (!note->get_children().empty()) {
162 auto* first_para = dynamic_cast<Paragraph*>(note->get_children()[0].get());
163 if (first_para && !first_para->get_children().empty()) {
164 auto first_child = first_para->get_children()[0];
165 if (first_child->node_type() == NodeType::Run) {
166 first_para->remove_child(0);
167 }
168 }
169 }
170 }
171
172 cache.push_back(note);
173 }
174
175 for (const auto& note : cache) {
176 if (note->get_id() >= next_id) {
177 next_id = note->get_id() + 1;
178 }
179 }
180}
