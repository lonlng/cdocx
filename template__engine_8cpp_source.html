395static TemplateTarget resolve_target(const BookmarkNameCache* cache,
396 const std::string& key,
397 TemplateTarget preferred) {
398 if (preferred == TemplateTarget::Auto) {

402 return preferred;

406 const std::string& key,

431static bool insert_formatted_run_after(pugi::xml_node para,
432 pugi::xml_node after_node,
433 const std::string& text,
434 const BookmarkFormat& format) {
435 pugi::xml_node run = para.insert_child_after("w:r", after_node);
436 if (!run) {
437 return false;
438 }
439
440 const bool has_run_props = format.is_valid() || format.bold || format.italic ||
441 format.underline || format.strikethrough ||
442 !format.color.empty();
443 if (has_run_props) {
444 pugi::xml_node run_props = run.append_child("w:rPr");
445
446 if (!format.font_ascii.empty() || !format.font_far_east.empty()) {
447 pugi::xml_node run_fonts = run_props.append_child("w:rFonts");
448 if (!format.font_ascii.empty()) {
449 run_fonts.append_attribute("w:ascii").set_value(format.font_ascii.c_str());
450 }
451 if (!format.font_far_east.empty()) {
452 run_fonts.append_attribute("w:eastAsia").set_value(format.font_far_east.c_str());
453 }
454 }
455
456 if (format.font_size > 0) {
457 append_attr_node(run_props, "w:sz", "w:val", format.font_size);
458 append_attr_node(run_props, "w:szCs", "w:val", format.font_size);
459 }
460
461 if (!format.color.empty()) {
462 append_attr_node(run_props, "w:color", "w:val", format.color.c_str());
463 }
464
465 if (format.bold) {
466 run_props.append_child("w:b");
467 }
468 if (format.italic) {
469 run_props.append_child("w:i");
470 }
471 if (format.underline) {
472 append_attr_node(run_props, "w:u", "w:val", "single");
473 }
474 if (format.strikethrough) {
475 run_props.append_child("w:strike");
476 }
477 }
478
479 run.append_child("w:t").text().set(text.c_str());
480 return true;
481}

506TemplateEngine::Result TemplateEngine::apply(const std::string& key) {
507 last_result_ = Result();
508 if (!doc_ || !doc_->is_open()) {
509 return last_result_;
510 }
511
512 auto it = queue_.find(key);
513 if (it == queue_.end()) {
514 return last_result_;
515 }
516
517 const auto& value = it->second;
518 if (value.is_empty()) {
519 last_result_.skipped++;
520 return last_result_;
521 }
522
523 auto actual = resolve_target(doc_, key, default_target_,
524 delimiter_prefix_, delimiter_suffix_);
525 if (actual == TemplateTarget::Placeholder) {
526 last_result_ = apply_placeholder(key, value);
527 } else {
528 last_result_ = apply_bookmark(key, value);
529 doc_->sync_from_physical_tree();
530 }
531 return last_result_;
532}
533
534TemplateEngine::Result TemplateEngine::apply_if(
535 const std::function<bool(const std::string&)>& predicate) {
536 last_result_ = Result();
537 if (!doc_ || !doc_->is_open() || queue_.empty()) {
538 return last_result_;
539 }
540
541 // Cache bookmark names once for Auto target resolution (avoids O(n*m) DOM sync)
542 std::unique_ptr<BookmarkNameCache> bm_cache;
543 if (default_target_ == TemplateTarget::Auto) {
544 bm_cache = std::make_unique<BookmarkNameCache>(doc_);
545 }
546
547 std::vector<std::pair<std::string, TemplateValue>> placeholders;
548 std::vector<std::pair<std::string, TemplateValue>> bookmarks;
549
550 for (const auto& [key, value] : queue_) {
551 if (!predicate(key)) {
552 last_result_.skipped++;
553 continue;
554 }
555 if (value.is_empty()) {
556 last_result_.skipped++;
557 continue;
558 }
559 auto actual = resolve_target(bm_cache.get(), key, default_target_);
560 if (actual == TemplateTarget::Placeholder) {
561 placeholders.emplace_back(key, value);
562 } else {
563 bookmarks.emplace_back(key, value);
564 }
565 }
566
567 if (!placeholders.empty()) {
568 for (const auto& [key, value] : placeholders) {
569 auto r = apply_placeholder(key, value);
570 last_result_.success += r.success;
571 last_result_.failed += r.failed;
572 }
573 doc_->sync_to_physical_tree();
574 }
575
576 if (!bookmarks.empty()) {
577 if (placeholders.empty()) {
578 doc_->sync_to_physical_tree();
579 }
580 auto collection = doc_->get_bookmarks();
581
582 for (const auto& [key, value] : bookmarks) {
583 auto bm_opt = collection.get(key);
584 if (!bm_opt) {
585 last_result_.failed++;
586 continue;
587 }
588 if (value.is_text()) {
589 const bool ok = apply_text_to_bookmark(
590 *bm_opt, value.text_content(), value.text_format(), default_format_policy_);
591 if (ok) {
592 last_result_.success++;
593 } else {
594 last_result_.failed++;
595 }
596 } else if (value.is_image()) {
597 const bool ok = apply_image_to_bookmark(*bm_opt, value);
598 if (ok) {
599 last_result_.success++;
600 } else {
601 last_result_.failed++;
602 }
603 }
604 }
605 doc_->sync_from_physical_tree();
606
607 // In Auto mode, a key may exist both as a bookmark and as a placeholder.
608 // After bookmark replacement, also try placeholder replacement for any
609 // non-bookmarked instances.
610 if (default_target_ == TemplateTarget::Auto) {
611 for (const auto& [key, value] : bookmarks) {
612 auto r = apply_placeholder(key, value);
613 last_result_.success += r.success;
614 last_result_.failed += r.failed;
615 }
616 doc_->sync_to_physical_tree();
617 }
618 }
619
620 return last_result_;
621}

716bool TemplateEngine::apply_text_to_bookmark(Bookmark& bookmark,
717 const std::string& text,
718 const TemplateFormat& format,
719 FormatPolicy policy) {
720 if (default_action_ == TemplateAction::Insert) {
721 if (!bookmark.is_valid()) {
722 return false;
723 }
724
725 auto paras = bookmark.get_covered_paragraphs();
726 if (paras.empty()) {
727 return false;
728 }
729
730 const pugi::xml_node bookmark_start = find_bookmark_start(paras[0]);
731 if (!bookmark_start) {
732 return false;
733 }
734
735 BookmarkFormat bmf;
736 if (policy == FormatPolicy::Preserve) {
737 bmf = bookmark.get_format();
738 } else {
739 TemplateFormat effective = format;
740 if (effective.is_empty() && !default_format_.is_empty()) {
741 effective = default_format_;
742 }
743 if (!effective.is_empty()) {
744 bmf = effective.to_bookmark_format();
745 } else if (policy == FormatPolicy::Custom) {
746 bmf = bookmark.get_format();
747 }
748 }
749
750 return insert_formatted_run_after(paras[0], bookmark_start, text, bmf);
751 }
752
753 // Replace mode
754 BookmarkReplacer replacer(doc_);
755
756 switch (policy) {
757 case FormatPolicy::Preserve:
758 return replacer.replace_text(bookmark, text);
759
760 case FormatPolicy::Override:
761 case FormatPolicy::Custom: {
762 if (format.is_empty() && policy == FormatPolicy::Custom) {
763 return replacer.replace_text(bookmark, text);
764 }
765 TemplateFormat effective = format;
766 if (effective.is_empty() && !default_format_.is_empty()) {
767 effective = default_format_;
768 }
769 if (effective.is_empty()) {
770 return replacer.replace_text(bookmark, text);
771 }
772 return replacer.replace_text_with_format(bookmark, text,
773 effective.to_bookmark_format());
774 }
775 }
776 return false;
777}

789bool TemplateEngine::apply_image_to_bookmark(Bookmark& bookmark,
790 const TemplateValue& value) {
791 if (default_action_ == TemplateAction::Insert) {
792 if (!bookmark.is_valid()) {
793 return false;
794 }
795
796 auto paras = bookmark.get_covered_paragraphs();
797 if (paras.empty()) {
798 return false;
799 }
800
801 const pugi::xml_node bookmark_start = find_bookmark_start(paras[0]);
802 if (!bookmark_start) {
803 return false;
804 }
805
806 const auto& path = value.image_path();
807 const auto& size = value.image_size();
808 auto align = value.image_alignment();
809
810 if (!std::filesystem::exists(path)) {
811 return false;
812 }
813
814 ImageSize actual_size = size;
815 if (!actual_size.is_valid()) {
816 if (!detect_image_size(path, actual_size)) {
817 actual_size = ImageSize(400, 300);
818 }
819 }
820
821 const std::string rel_id = doc_->add_media_with_rel(path, nullptr);
822 if (rel_id.empty()) {
823 return false;
824 }
825
826 return insert_image_run_after(paras[0], bookmark_start, actual_size, align, rel_id,
827 image_counter_++);
828 }
829
830 // Replace mode
831 BookmarkReplacer replacer(doc_);
832 const auto& path = value.image_path();
833 const auto& size = value.image_size();
834 const auto& caption = value.image_caption();
835 auto align = value.image_alignment();
836

838 return false;
839 }
840
841 if (size.is_valid()) {
842 return replacer.replace_with_image_advanced(bookmark, path, size, caption, align);
843 } else {
844 return replacer.replace_with_image(bookmark, path, caption);
845 }
846}

848bool TemplateEngine::apply_text_to_placeholder(const std::string& key,
849 const std::string& text) {
850 Template tmpl(doc_, delimiter_prefix_, delimiter_suffix_);
851 if (default_action_ == TemplateAction::Insert) {
852 const std::string pattern = delimiter_prefix_ + key + delimiter_suffix_;
853 // Prevent unintended cascading replacements if the replacement text
854 // itself contains the placeholder pattern.
855 if (text.find(pattern) != std::string::npos || text.empty()) {
856 return false;
857 }
858 tmpl.set(key, text + pattern);
859 } else {
860 tmpl.set(key, text);
861 }
862 if (default_scope_ == TemplateScope::First) {
863 return tmpl.replace_first();
864 }
865 tmpl.replace_all();
866 return true;
867}

872 return false;

876 if (default_scope_ == TemplateScope::First) {
877 return tmpl.replace_first();

879 tmpl.replace_all();
