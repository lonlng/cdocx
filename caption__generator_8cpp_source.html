60int CaptionGenerator::count_existing_figures(Document* doc) {
61 if (!doc) {
62 return 0;
63 }
64
65 pugi::xml_document* doc_xml = doc->get_document_xml();
66 if (!doc_xml) {
67 return 0;
68 }
69
70 int count = 0;
71
72 // Get all paragraphs in document body
73 const pugi::xml_node body = doc_xml->child("w:document").child("w:body");
74 if (!body) {
75 return 0;
76 }
77
78 for (pugi::xml_node para = body.child("w:p"); para; para = para.next_sibling("w:p")) {
79 if (is_figure_caption(para)) {
80 count++;
81 }
82 }
83
84 return count;

91bool CaptionGenerator::is_figure_caption(pugi::xml_node para) {
92 if (!para) {
93 return false;
94 }
95
96 // Extract all text from paragraph
97 std::string text;
98 for (pugi::xml_node run = para.child("w:r"); run; run = run.next_sibling("w:r")) {
99 const pugi::xml_node t = run.child("w:t");
100 if (t && t.text()) {
101 text += t.text().get();
102 }
103 }
104
105 // trim whitespace
106 const size_t start = text.find_first_not_of(" \t\n\r");
107 if (start == std::string::npos) {
108 return false;
109 }
110 text = text.substr(start);
111
112 // Check if starts with "图" followed by number
113 // Support both "图 1" and "图1" formats
114 if (text.length() < 2) {
115 return false;
116 }
117
118 // Check for "图" character (UTF-8: 0xE5 0x9B 0xBE)
119 if (text.substr(0, 3) != "\xE5\x9B\xBE") {
120 return false;
121 }
122
123 // Check for number after "图"
124 size_t pos = 3;
125 // Skip optional space
126 if (pos < text.length() && text[pos] == ' ') {
127 pos++;
128 }
129
130 // Check for digit
131 if (pos < text.length() && std::isdigit(static_cast<unsigned char>(text[pos]))) {
132 return true;
133 }
134
135 return false;
136}

138std::string CaptionGenerator::extract_caption_text(pugi::xml_node para) {
139 if (!is_figure_caption(para)) {
140 return "";
141 }
142
143 // Extract all text
144 std::string text;
145 for (pugi::xml_node run = para.child("w:r"); run; run = run.next_sibling("w:r")) {
146 const pugi::xml_node t = run.child("w:t");
147 if (t && t.text()) {
148 text += t.text().get();
149 }
150 }
151
152 // Find position after figure number
153 // Pattern: "图 X " or "图X "
154 size_t pos = text.find("\xE5\x9B\xBE"); // "图"
155 if (pos == std::string::npos) {
156 return "";
157 }
158
159 pos += 3; // Skip "图"
160
161 // Skip optional space
162 if (pos < text.length() && text[pos] == ' ') {
163 pos++;
164 }
165
166 // Skip number
167 while (pos < text.length() && std::isdigit(static_cast<unsigned char>(text[pos]))) {
168 pos++;
169 }
170
171 // Skip space after number
172 if (pos < text.length() && text[pos] == ' ') {
173 pos++;
174 }
175
176 // Return remaining text (description)
177 return text.substr(pos);
178}

184pugi::xml_node CaptionGenerator::create_caption_paragraph(Document* doc,
185 const std::string& full_text,
186 pugi::xml_node insert_after) {
187 if (!doc) {
188 return pugi::xml_node{};
189 }
190
191 pugi::xml_document* doc_xml = doc->get_document_xml();
192 if (!doc_xml) {
193 return pugi::xml_node{};
194 }
195
196 pugi::xml_node body = doc_xml->child("w:document").child("w:body");
197 if (!body) {
198 return pugi::xml_node{};
199 }
200
201 // Create new paragraph
202 pugi::xml_node caption_para;
203 if (insert_after) {
204 caption_para = body.insert_child_after("w:p", insert_after);
205 } else {
206 caption_para = body.append_child("w:p");
207 }
208
209 if (!caption_para) {
210 return pugi::xml_node{};
211 }
212
213 // Create paragraph properties with center alignment
214 pugi::xml_node p_pr = caption_para.append_child("w:pPr");
215 if (p_pr) {
216 pugi::xml_node jc = p_pr.append_child("w:jc");
217 if (jc) {
218 jc.append_attribute("w:val").set_value("center");
219 }
220
221 // Add spacing before and after
222 pugi::xml_node spacing = p_pr.append_child("w:spacing");
223 if (spacing) {
224 spacing.append_attribute("w:before").set_value(100); // 5pt
225 spacing.append_attribute("w:after").set_value(100); // 5pt
226 }
227 }
228
229 // Create run with text
230 pugi::xml_node run = caption_para.append_child("w:r");
231 if (!run) {
232 return caption_para;
233 }
234
235 // Add run properties with fonts
236 pugi::xml_node r_pr = run.append_child("w:rPr");
237 if (r_pr) {
238 const CaptionFormat fmt = CaptionFormat::chinese_default();
239
240 // Font settings
241 pugi::xml_node r_fonts = r_pr.append_child("w:rFonts");
242 if (r_fonts) {
243 r_fonts.append_attribute("w:ascii").set_value(fmt.font_ascii.c_str());
244 r_fonts.append_attribute("w:eastAsia").set_value(fmt.font_east_asia.c_str());
245 r_fonts.append_attribute("w:hAnsi").set_value(fmt.font_hansi.c_str());
246 }
247
248 // Font size
249 pugi::xml_node sz = r_pr.append_child("w:sz");
250 if (sz) {
251 sz.append_attribute("w:val").set_value(fmt.font_size);
252 }
253 pugi::xml_node sz_cs = r_pr.append_child("w:szCs");
254 if (sz_cs) {
255 sz_cs.append_attribute("w:val").set_value(fmt.font_size);
256 }
257 }
258
259 // Add text
260 const pugi::xml_node t = run.append_child("w:t");
261 if (t) {
262 t.text().set(full_text.c_str());
263 }
264
265 return caption_para;
266}

268std::string CaptionGenerator::generate_caption_text(int figure_number,
269 const std::string& description) {
270 // Format: "图 X description"
271 std::string result = "\xE5\x9B\xBE " + std::to_string(figure_number); // "图 "
272
273 if (!description.empty()) {
274 result += " " + description;
275 }
276
277 return result;
278}
